15/// TargetLanguage uniffi_bindgen, with a `clap::ValueEnum` derive.
16#[derive(Copy, Clone, ValueEnum)]
17enum TargetLanguageArg {
18    Kotlin,
19    Swift,
20    Python,
21    Ruby,
22}
23
24impl fmt::Display for TargetLanguageArg {
25    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
26        match self {
27            Self::Kotlin => write!(f, "kotlin"),
28            Self::Swift => write!(f, "swift"),
29            Self::Python => write!(f, "python"),
30            Self::Ruby => write!(f, "ruby"),
31        }
32    }
33}
34
35impl From<TargetLanguageArg> for TargetLanguage {
36    fn from(l: TargetLanguageArg) -> Self {
37        match l {
38            TargetLanguageArg::Kotlin => Self::Kotlin,
39            TargetLanguageArg::Swift => Self::Swift,
40            TargetLanguageArg::Python => Self::Python,
41            TargetLanguageArg::Ruby => Self::Ruby,
42        }
43    }
44}

49/// Scaffolding and bindings generator for Rust
50#[derive(Parser)]
51#[clap(name = "uniffi-bindgen")]
52struct Cli {
53    #[clap(subcommand)]
54    command: Commands,
55}
56
57#[derive(Subcommand)]
58enum Commands {
59    /// Generate foreign language bindings
60    Generate {
61        /// Foreign language(s) for which to build bindings.
62        #[clap(long, short, value_enum)]
63        language: Vec<TargetLanguageArg>,
64
65        /// Directory in which to write generated files. Default is same folder as .udl file.
66        #[clap(long, short)]
67        out_dir: Option<Utf8PathBuf>,
68
69        /// Do not try to format the generated bindings.
70        #[clap(long, short)]
71        no_format: bool,
72
73        /// Path to optional uniffi config file. This config is merged with the `uniffi.toml` config present in each crate, with its values taking precedence.
74        #[clap(long, short)]
75        config: Option<Utf8PathBuf>,
76
77        /// When `--library` is passed, only generate bindings for one crate.
78        /// When `--library` is not passed, use this as the crate name instead of attempting to
79        /// locate and parse Cargo.toml.
80        #[clap(long = "crate")]
81        crate_name: Option<String>,
82
83        /// Path to the UDL file, or cdylib if `library-mode` is specified
84        source: Utf8PathBuf,
85    },
86
87    /// Generate Rust scaffolding code
88    Scaffolding {
89        /// Directory in which to write generated files. Default is same folder as .udl file.
90        #[clap(long, short)]
91        out_dir: Option<Utf8PathBuf>,
92
93        /// Do not try to format the generated bindings.
94        #[clap(long, short)]
95        no_format: bool,
96
97        /// Path to the UDL file.
98        udl_file: Utf8PathBuf,
99    },
100
101    /// Inspect the bindings render pipeline
102    Pipeline(PipelineArgs),
103}
104
105#[derive(Args)]
106struct PipelineArgs {
107    /// Pass in a cdylib path rather than a UDL file
108    #[clap(long = "library")]
109    library_mode: bool,
110
111    /// Path to the UDL file, or cdylib if `library-mode` is specified
112    source: Utf8PathBuf,
113
114    /// When `--library` is passed, only generate bindings for one crate.
115    /// When `--library` is not passed, use this as the crate name instead of attempting to
116    /// locate and parse Cargo.toml.
117    #[clap(long = "crate")]
118    crate_name: Option<String>,
119
120    /// Whether we should exclude dependencies when running "cargo metadata".
121    /// This will mean external types may not be resolved if they are implemented in crates
122    /// outside of this workspace.
123    /// This can be used in environments when all types are in the namespace and fetching
124    /// all sub-dependencies causes obscure platform specific problems.
125    #[clap(long)]
126    metadata_no_deps: bool,
127
128    /// Bindings Language
129    language: TargetLanguageArg,
130
131    /// Only show passes that match <PASS>
132    ///
133    /// Use `last` to only show the last pass, this can be useful when you're writing new pipelines
134    #[clap(short, long)]
135    pass: Option<String>,
136
137    /// Don't show diffs for middle passes
138    #[clap(long)]
139    no_diff: bool,
140
141    /// Only show data for types with name <FILTER_TYPE>
142    #[clap(short = 't', long = "type")]
143    filter_type: Option<String>,
144
145    /// Only show data for items with fields that match <FILTER>
146    #[clap(short = 'n', long = "name")]
147    filter_name: Option<String>,
148}

150pub fn run_main() -> anyhow::Result<()> {
151    let cli = Cli::parse();
152    let metadata_no_deps = false;
153    match cli.command {
154        Commands::Generate {
155            language,
156            out_dir,
157            no_format,
158            config,
159            source,
160            crate_name,
161            ..
162        } => {
163            if language.is_empty() {
164                panic!("please specify at least one language with --language")
165            }
166
167            let mut paths = uniffi_bindgen::BindgenPaths::default();
168            paths.add_layer(crate::config_supplier::NoCargoConfigSupplier);
169
170            generate_with_bindgen_paths(
171                GenerateOptions {
172                    languages: language.into_iter().map(TargetLanguage::from).collect(),
173                    out_dir: out_dir
174                        .expect("--out-dir is required when generating {language} bindings"),
175                    source,
176                    config_override: config,
177                    crate_filter: crate_name,
178                    metadata_no_deps,
179                    format: !no_format,
180                },
181                paths,
182            )?;
183        }
184        Commands::Scaffolding {
185            out_dir,
186            no_format,
187            udl_file,
188        } => {
189            uniffi_bindgen::generate_component_scaffolding(
190                &udl_file,
191                out_dir.as_deref(),
192                !no_format,
193            )?;
194        }
195        Commands::Pipeline(args) => {
196            let mut paths = uniffi_bindgen::BindgenPaths::default();
197            paths.add_layer(crate::config_supplier::NoCargoConfigSupplier);
198
199            let initial_root = if args.library_mode {
200                initial::Root::from_library(paths, &args.source, args.crate_name)?
201            } else {
202                initial::Root::from_udl(paths, &args.source, args.crate_name)?
203            };
204
205            let opts = PrintOptions {
206                pass: args.pass,
207                no_diff: args.no_diff,
208                filter_type: args.filter_type,
209                filter_name: args.filter_name,
210            };
211            match args.language {
212                TargetLanguageArg::Python => python::pipeline().print_passes(initial_root, opts)?,
213                language => unimplemented!("{language} does not use the bindings IR pipeline yet"),
214            };
215        }
216    };
217    Ok(())
218}
