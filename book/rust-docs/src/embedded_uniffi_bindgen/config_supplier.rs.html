5// This is the only significant difference from vanilla UniFFI,
6//
7// We define our own config supplier than parses `Cargo.toml` files directly to avoid a dependency
8// on `cargo`.  This code tries to parse as little as possible to make this work.
9//
10// We could move this code into the main uniffi repo and add a flag to use it instead of the
11// cargo-based config supplier.
12

26impl BindgenPathsLayer for NoCargoConfigSupplier {
27    fn get_udl_path(&self, crate_name: &str, udl_name: &str) -> Option<Utf8PathBuf> {
28        let crate_map = CRATE_MAP.as_ref().expect("Error parsing Cargo.toml files");
29        let crate_root = crate_map.get(crate_name)?;
30        Some(crate_root.join(format!("src/{udl_name}.udl")))
31    }
32}
33
34static CRATE_MAP: LazyLock<Result<HashMap<String, Utf8PathBuf>>> =
35    LazyLock::new(find_workspace_crates);
36
37/// Find all crates in this workspace and return a map of crate_name => crate_root_path
38fn find_workspace_crates() -> Result<HashMap<String, Utf8PathBuf>> {
39    let workspace_toml = find_workspace_toml()?;
40
41    let mut toml_paths_to_process = vec![];
42    for path in workspace_toml.data.workspace.unwrap().members {
43        toml_paths_to_process.extend(join_and_glob(&workspace_toml.dir, path)?)
44    }
45    let mut toml_paths_processed = HashSet::new();
46    let mut map = HashMap::new();
47
48    loop {
49        let Some(crate_dir) = toml_paths_to_process.pop() else {
50            break;
51        };
52        let toml_path = join(&crate_dir, "Cargo.toml")?;
53        if !toml_paths_processed.insert(toml_path.clone()) {
54            continue;
55        }
56
57        let toml = CargoToml::from_path(&toml_path)?;
58        let new_paths = find_other_cargo_toml_paths(&crate_dir, &toml);
59        toml_paths_to_process.extend(new_paths);
60
61        // Add both the package name and library name to the map
62        if let Some(package) = toml.package {
63            map.insert(package.name.replace("-", "_"), crate_dir.clone());
64        }
65
66        if let Some(CargoLibrary { name: Some(name) }) = toml.lib {
67            map.insert(name.replace("-", "_"), crate_dir);
68        }
69    }
70    Ok(map)
71}

73/// Find the workspace Cargo.toml file.
74///
75/// Returns the parsed TOML data plus the directory of the file
76fn find_workspace_toml() -> Result<CargoTomlFile> {
77    let current_dir = camino::Utf8PathBuf::from_path_buf(env::current_dir()?)
78        .map_err(|_| anyhow!("path is not UTF8"))?;
79    let mut dir = current_dir.as_path();
80    loop {
81        let path = dir.join("Cargo.toml");
82        if path.exists() {
83            let toml = CargoToml::from_path(&path)?;
84            if toml.workspace.is_some() {
85                return Ok(CargoTomlFile {
86                    data: toml,
87                    dir: dir.to_path_buf(),
88                });
89            }
90        }
91        dir = dir
92            .parent()
93            .ok_or_else(|| anyhow!("Couldn't find workspace Cargo.toml"))?;
94    }
95}

97/// Process Cargo.toml data and return all crate paths referenced in it
98fn find_other_cargo_toml_paths(crate_dir: &Utf8Path, toml: &CargoToml) -> Vec<Utf8PathBuf> {
99    toml.dependencies
100        .iter()
101        .flat_map(|d| d.values())
102        .filter_map(|dep| match dep {
103            // for servo in particular, Cargo.toml specifies relative paths which do not exist, presumably gated by features.
104            // eg, `servo/components/style_traits/Cargo.toml` references `../atoms`.
105            // So these are just ignored rather than treated as an error.
106            CargoDependency::Detailed { path: Some(path) } => join(crate_dir, path).ok(),
107            _ => None,
108        })
109        .collect()
110}
111
112fn join(dir: &Utf8Path, child: impl AsRef<str>) -> Result<Utf8PathBuf> {
113    let child = child.as_ref();
114    dir.join(child)
115        .canonicalize_utf8()
116        .map_err(|p| anyhow!("Invalid path: {p} {dir}, {child}"))
117}
118
119fn join_and_glob(dir: &Utf8Path, child: impl AsRef<str>) -> Result<Vec<Utf8PathBuf>> {
120    let child = child.as_ref();
121    glob::glob(dir.join(child).as_str())?
122        .map(|entry| anyhow::Ok(Utf8PathBuf::try_from(entry?)?))
123        .map(|path| Ok(path?.canonicalize_utf8()?))
124        .collect()
125}

127#[derive(Debug)]
128struct CargoTomlFile {
129    data: CargoToml,
130    dir: Utf8PathBuf,
131}
132
133#[derive(Debug, Deserialize)]
134struct CargoToml {
135    package: Option<CargoPackage>,
136    lib: Option<CargoLibrary>,
137    workspace: Option<CargoWorkspace>,
138    dependencies: Option<HashMap<String, CargoDependency>>,
139}
140
141impl CargoToml {
142    fn from_path(path: &Utf8Path) -> Result<Self> {
143        let contents = fs::read_to_string(path).context(format!("reading {path}"))?;
144        toml::from_str(&contents).context(format!("parsing {path}"))
145    }
146}
147
148#[derive(Debug, Deserialize)]
149struct CargoPackage {
150    name: String,
151}
152
153#[derive(Debug, Deserialize)]
154struct CargoLibrary {
155    name: Option<String>,
156}
157
158#[derive(Debug, Deserialize)]
159struct CargoWorkspace {
160    members: Vec<Utf8PathBuf>,
161}
162
163#[derive(Debug, Deserialize)]
164#[serde(untagged)]
165enum CargoDependency {
166    #[allow(dead_code)]
167    Simple(String),
168    Detailed {
169        path: Option<Utf8PathBuf>,
170    },
