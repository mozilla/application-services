132    /// Get the pairing URL to navigate to on the Auth side (typically
133    /// a computer).
134    pub fn get_pairing_authority_url(&self) -> Result<String> {
135        // Special case for the production server, we use the shorter firefox.com/pair URL.
136        let content_url_release = Url::parse(config::CONTENT_URL_RELEASE)
137            .expect("hardcoded CONTENT_URL_RELEASE failed to parse");
138        if self.state.config().content_url()? == content_url_release {
139            return Ok("https://firefox.com/pair".to_owned());
140        }
141        // Similarly special case for the China server.
142        let content_url_china = Url::parse(config::CONTENT_URL_CHINA)
143            .expect("hardcoded CONTENT_URL_CHINA failed to parse");
144        if self.state.config().content_url()? == content_url_china {
145            return Ok("https://firefox.com.cn/pair".to_owned());
146        }
147        Ok(self.state.config().pair_url()?.into())
148    }

161    /// Get the "manage account" page URL.
162    /// It is typically used in the application's account status UI,
163    /// to link the user out to a webpage where they can manage
164    /// all the details of their account.
165    ///
166    /// * `entrypoint` - Application-provided string identifying the UI touchpoint
167    ///   through which the page was accessed, for metrics purposes.
168    pub fn get_manage_account_url(&mut self, entrypoint: &str) -> Result<String> {
169        let mut url = self.state.config().settings_url()?;
170        url.query_pairs_mut().append_pair("entrypoint", entrypoint);
171        if self.state.config().redirect_uri == OAUTH_WEBCHANNEL_REDIRECT {
172            url.query_pairs_mut()
173                .append_pair("context", "oauth_webchannel_v1");
174        }
175        self.add_account_identifiers_to_url(url)
176    }
177
178    /// Get the "manage devices" page URL.
179    /// It is typically used in the application's account status UI,
180    /// to link the user out to a webpage where they can manage
181    /// the devices connected to their account.
182    ///
183    /// * `entrypoint` - Application-provided string identifying the UI touchpoint
184    ///   through which the page was accessed, for metrics purposes.
185    pub fn get_manage_devices_url(&mut self, entrypoint: &str) -> Result<String> {
186        let mut url = self.state.config().settings_clients_url()?;
187        url.query_pairs_mut().append_pair("entrypoint", entrypoint);
188        self.add_account_identifiers_to_url(url)
189    }
190
191    fn add_account_identifiers_to_url(&mut self, mut url: Url) -> Result<String> {
192        let profile = self.get_profile(false)?;
193        url.query_pairs_mut()
194            .append_pair("uid", &profile.uid)
195            .append_pair("email", &profile.email);
196        Ok(url.into())
197    }

199    fn get_refresh_token(&self) -> Result<&str> {

202            None => Err(Error::NoRefreshToken),
203        }

210    /// Disconnect from the account and optionally destroy our device record. This will
211    /// leave the account object in a state where it can eventually reconnect to the same user.
212    /// This is a "best effort" infallible method: e.g. if the network is unreachable,
213    /// the device could still be in the FxA devices manager.
214    ///
215    /// **💾 This method alters the persisted account state.**
216    pub fn disconnect(&mut self) {
217        let current_device_result;
218        {
219            current_device_result = self.get_current_device();
220        }
221
222        if let Some(refresh_token) = self.state.refresh_token() {
223            // Delete the current device (which deletes the refresh token), or
224            // the refresh token directly if we don't have a device.
225            let destroy_result = match current_device_result {
226                // If we get an error trying to fetch our device record we'll at least
227                // still try to delete the refresh token itself.
228                Ok(Some(device)) => self.client.destroy_device_record(
229                    self.state.config(),
230                    &refresh_token.token,
231                    &device.id,
232                ),
233                _ => self
234                    .client
235                    .destroy_refresh_token(self.state.config(), &refresh_token.token),
236            };
237            if let Err(e) = destroy_result {
238                crate::warn!("Error while destroying the device: {}", e);
239            }
240        }
241        self.state.disconnect();
242        self.clear_devices_and_attached_clients_cache();
243        self.telemetry = FxaTelemetry::new();
244    }
245
246    /// Update the state based on authentication issues.
247    ///
248    /// **💾 This method alters the persisted account state.**
249    ///
250    /// Call this if you know there's an authentication / authorization issue that requires the
251    /// user to re-authenticated.  It transitions the user to the [FxaRustAuthState.AuthIssues] state.
252    pub fn on_auth_issues(&mut self) {
253        self.state.on_auth_issues();
254        self.clear_devices_and_attached_clients_cache();
255        self.telemetry = FxaTelemetry::new();
256    }

544    #[test]
545    fn test_on_auth_issues() {
546        let config = Config::new("https://stable.dev.lcip.org", "12345678", "https://foo.bar");
547        let mut fxa = FirefoxAccount::with_config(config);
548
549        fxa.state.force_refresh_token(RefreshToken {
550            token: "refresh_token".to_owned(),
551            scopes: HashSet::new(),
552        });
553        fxa.state.force_current_device_id("original-device-id");
554        assert_eq!(fxa.get_auth_state(), FxaRustAuthState::Connected);
555
556        fxa.on_auth_issues();
557        assert_eq!(fxa.get_auth_state(), FxaRustAuthState::AuthIssues);
558
559        fxa.state.complete_oauth_flow(
560            vec![],
561            RefreshToken {
562                token: "refreshtok".to_owned(),
563                scopes: HashSet::default(),
564            },
565            None,
566        );
567        assert_eq!(fxa.get_auth_state(), FxaRustAuthState::Connected);
568
569        // The device ID should be the same as before `on_auth_issues` was called.  This
570        // way, methods like `ensure_capabilities` and `set_device_name`, can re-use it and we
571        // won't try to create a new device record.
572        assert_eq!(fxa.state.current_device_id(), Some("original-device-id"));
573    }

576    fn test_get_auth_state() {
577        let config = Config::new("https://stable.dev.lcip.org", "12345678", "https://foo.bar");
578        let mut fxa = FirefoxAccount::with_config(config);
579
580        fn assert_auth_state(fxa: &FirefoxAccount, correct_state: FxaRustAuthState) {
581            assert_eq!(fxa.get_auth_state(), correct_state);
582
583            let persisted = FirefoxAccount::from_json(&fxa.to_json().unwrap()).unwrap();
584            assert_eq!(persisted.get_auth_state(), correct_state);
585        }
586
587        // The state starts as disconnected
588        assert_auth_state(&fxa, FxaRustAuthState::Disconnected);
589
590        // When we get the refresh tokens the state changes to connected
591        fxa.state.force_refresh_token(RefreshToken {
592            token: "refresh_token".to_owned(),
593            scopes: HashSet::new(),
594        });
595        assert_auth_state(&fxa, FxaRustAuthState::Connected);
596
597        fxa.disconnect();
598        assert_auth_state(&fxa, FxaRustAuthState::Disconnected);
599
600        fxa.disconnect();
601        assert_auth_state(&fxa, FxaRustAuthState::Disconnected);
602    }
