12// If a cached token has less than `OAUTH_MIN_TIME_LEFT` seconds left to live,
13// it will be considered already expired.
14const OAUTH_MIN_TIME_LEFT: u64 = 60;
15
16fn normalize_scopes(scope: &str) -> String {
17    let mut parts: Vec<&str> = scope.split_ascii_whitespace().collect();
18    parts.sort_unstable();
19    parts.dedup();
20    parts.join(" ")
21}

23impl FirefoxAccount {
24    /// Fetch a short-lived access token using the saved refresh token.
25    /// If there is no refresh token held or if it is not authorized for some of the requested
26    /// scopes, this method will error-out and a login flow will need to be initiated
27    /// using `begin_oauth_flow`.
28    ///
29    /// * `scope` - Space-separated list of requested scopes. Order is not significant;
30    ///   the cache is keyed on the normalized (sorted, deduplicated) set, so
31    ///   `"a b"` and `"b a"` are treated as identical requests.
32    /// * `use_cache` - optionally set to false to force a new token request.  The fetched
33    ///   token will still be cached for later `get_access_token` calls.
34    ///
35    /// The result `AccessTokenInfo.key` will only be `Some()` when the token has a single scope
36    /// and that scope has a key. If you request 'sync profile', you don't get the 'sync' key.
37    ///
38    /// **💾 This method may alter the persisted account state.**
39    pub fn get_access_token(&mut self, scope: &str, use_cache: bool) -> Result<AccessTokenInfo> {
40        debug!("get_access_token for scope={scope}, use_cache={use_cache}");
41        let requested = normalize_scopes(scope);
42        if requested.is_empty() {
43            return Err(Error::IllegalState("No scopes requested."));
44        }
45        let requested_set: HashSet<&str> = requested.split(' ').collect();
46
47        if use_cache {
48            if let Some(oauth_info) = self.state.get_cached_access_token(&requested) {
49                if oauth_info.expires_at > util::now_secs() + OAUTH_MIN_TIME_LEFT {
50                    // If the cached key is missing the required sync scoped key, try to fetch it again
51                    if oauth_info.check_missing_sync_scoped_key().is_ok() {
52                        return Ok(oauth_info.clone());
53                    }
54                    // but it missing should be impossible!?
55                    error_support::report_error!(
56                        "fxaclient-lost-scoped-key",
57                        "have cached token without a key"
58                    );
59                }
60            }
61        }
62        let mut requested_scopes: Vec<&str> = requested_set.iter().copied().collect();
63        requested_scopes.sort_unstable();
64        let resp = match self.state.refresh_token() {
65            Some(mut refresh_token) => {
66                let missing: Vec<&str> = requested_scopes
67                    .iter()
68                    .copied()
69                    .filter(|s| !refresh_token.scopes.contains(*s))
70                    .collect();
71                if !missing.is_empty() {
72                    // We don't currently have all scopes - try token exchange to upgrade.
73                    info!(
74                        "refresh token doing exchange; have {:?}, missing {missing:?})",
75                        refresh_token.scopes
76                    );
77                    let exchange_resp = self.client.exchange_token_for_scope(
78                        self.state.config(),
79                        &refresh_token.token,
80                        &missing.join(" "),
81                    )?;
82                    // Update state with the new refresh token that has combined scopes.
83                    if let Some(new_refresh_token) = exchange_resp.refresh_token {
84                        self.state.update_refresh_token(RefreshToken::new(
85                            new_refresh_token,
86                            exchange_resp.scope,
87                        ));
88                    } else {
89                        // A request for a new token succeeding but without a new token is unexpected.
90                        error!("successful response for a new refresh token with additional scopes, but no token was delivered");
91                        // at this stage we are almost certainly still going to fail to get a token...
92                    }
93                    // Get the updated refresh token from state.
94                    refresh_token = match self.state.refresh_token() {
95                        // We had a refresh token, we must either still have the original or maybe a new one,
96                        // but it's impossible for us to not have one at this point.
97                        None => unreachable!("lost the refresh token"),
98                        Some(token) => token,
99                    };
100                }
101                if requested_scopes
102                    .iter()
103                    .all(|s| refresh_token.scopes.contains(*s))
104                {
105                    info!(
106                        "using refresh token to request new token with scope {requested_scopes:?}"
107                    );
108                    self.client.create_access_token_using_refresh_token(
109                        self.state.config(),
110                        &refresh_token.token,
111                        None,
112                        &requested_scopes,
113                    )?
114                } else {
115                    // This should be impossible - if we don't have the scope we would have entered
116                    // the block where we try and get it, that succeeded and we got a new refresh token,
117                    // but still don't have the scope.
118                    error!("New refresh token doesn't have the scopes we requested: {requested}");
119                    return Err(Error::UnexpectedServerResponse);
120                }
121            }
122            None => {
123                match self.state.session_token() {
124                    Some(session_token) => {
125                        info!("using session token to request new token with scope {requested_scopes:?}");
126                        self.client.create_access_token_using_session_token(
127                            self.state.config(),
128                            session_token,
129                            &requested_scopes,
130                        )?
131                    }
132                    None => return Err(Error::NoSessionToken),
133                }
134            }
135        };
136        let since_epoch = SystemTime::now()
137            .duration_since(UNIX_EPOCH)
138            .map_err(|_| Error::IllegalState("Current date before Unix Epoch."))?;
139        let expires_at = since_epoch.as_secs() + resp.expires_in;
140        let key = if requested_scopes.len() == 1 {
141            self.state.get_scoped_key(requested_scopes[0]).cloned()
142        } else {
143            None
144        };
145        let token_info = AccessTokenInfo {
146            scope: resp.scope,
147            token: resp.access_token,
148            key,
149            expires_at,
150        };
151        self.state
152            .add_cached_access_token(&requested, token_info.clone());
153        token_info.check_missing_sync_scoped_key()?;
154        Ok(token_info)
155    }
156
157    /// **💾 This method may alter the persisted account state.**
158    pub fn clear_access_token_cache(&mut self) {
159        self.state.clear_access_token_cache();
160    }

163#[derive(Clone, Serialize, Deserialize)]
164pub struct AccessTokenInfo {
165    pub scope: String,
166    pub token: String,
167    pub key: Option<ScopedKey>,
168    pub expires_at: u64, // seconds since epoch
169}
170
171impl AccessTokenInfo {
172    pub fn check_missing_sync_scoped_key(&self) -> Result<()> {
173        let mut parts = self.scope.split_ascii_whitespace();
174        let first = parts.next();
175        let is_sole_old_sync = first == Some(scopes::OLD_SYNC) && parts.next().is_none();
176        if is_sole_old_sync && self.key.is_none() {
177            Err(Error::SyncScopedKeyMissingInServerResponse)
178        } else {
179            Ok(())
180        }
181    }
182}

184impl TryFrom<AccessTokenInfo> for crate::AccessTokenInfo {
185    type Error = Error;
186    fn try_from(info: AccessTokenInfo) -> Result<Self> {
187        Ok(crate::AccessTokenInfo {
188            scope: info.scope,
189            token: info.token,
190            key: info.key,
191            expires_at: info.expires_at.try_into()?,
192        })
193    }
194}

196impl std::fmt::Debug for AccessTokenInfo {
197    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
198        f.debug_struct("AccessTokenInfo")
199            .field("scope", &self.scope)
200            .field("key", &self.key)
201            .field("expires_at", &self.expires_at)
202            .finish()
203    }
