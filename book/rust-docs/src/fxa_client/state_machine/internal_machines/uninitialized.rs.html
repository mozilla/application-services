5use super::{invalid_transition, Event, InternalStateMachine, State};
6use crate::{Error, FxaEvent, FxaRustAuthState, FxaState, Result};
7
8pub struct UninitializedStateMachine;
9
10// Save some typing
11use Event::*;
12use State::*;
13
14impl InternalStateMachine for UninitializedStateMachine {
15    fn initial_state(&self, event: FxaEvent) -> Result<State> {
16        match event {
17            FxaEvent::Initialize { .. } => Ok(GetAuthState),
18            e => Err(Error::InvalidStateTransition(format!(
19                "Uninitialized -> {e}"
20            ))),
21        }
22    }
23
24    fn next_state(&self, state: State, event: Event) -> Result<State> {
25        Ok(match (state, event) {
26            (GetAuthState, GetAuthStateSuccess { auth_state }) => match auth_state {
27                FxaRustAuthState::Disconnected => Complete(FxaState::Disconnected),
28                FxaRustAuthState::AuthIssues => Complete(FxaState::AuthIssues),
29                FxaRustAuthState::Connected => EnsureDeviceCapabilities,
30            },
31            (EnsureDeviceCapabilities, EnsureDeviceCapabilitiesSuccess) => {
32                Complete(FxaState::Connected)
33            }
34            (EnsureDeviceCapabilities, CallError) => Complete(FxaState::Disconnected),
35            (EnsureDeviceCapabilities, EnsureCapabilitiesAuthError) => CheckAuthorizationStatus,
36
37            // FIXME: we should re-run `ensure_capabilities` in this case, but we don't in order to
38            // match the current firefox-android behavior.
39            // See https://bugzilla.mozilla.org/show_bug.cgi?id=1868418
40            (CheckAuthorizationStatus, CheckAuthorizationStatusSuccess { active: true }) => {
41                Complete(FxaState::Connected)
42            }
43            (CheckAuthorizationStatus, CheckAuthorizationStatusSuccess { active: false })
44            | (CheckAuthorizationStatus, CallError) => Complete(FxaState::AuthIssues),
45            (state, event) => return invalid_transition(state, event),
46        })
47    }
48}

50#[cfg(test)]
51mod test {
52    use super::super::StateMachineTester;
53    use super::*;
54    use crate::{DeviceConfig, DeviceType};
55
56    #[test]
57    fn test_state_machine() {
58        let mut tester = StateMachineTester::new(
59            UninitializedStateMachine,
60            FxaEvent::Initialize {
61                device_config: DeviceConfig {
62                    name: "test-device".to_owned(),
63                    device_type: DeviceType::Mobile,
64                    capabilities: vec![],
65                },
66            },
67        );
68        assert_eq!(tester.state, GetAuthState);
69        assert_eq!(
70            tester.peek_next_state(GetAuthStateSuccess {
71                auth_state: FxaRustAuthState::Disconnected
72            }),
73            Complete(FxaState::Disconnected)
74        );
75        assert_eq!(
76            tester.peek_next_state(GetAuthStateSuccess {
77                auth_state: FxaRustAuthState::AuthIssues
78            }),
79            Complete(FxaState::AuthIssues)
80        );
81
82        tester.next_state(GetAuthStateSuccess {
83            auth_state: FxaRustAuthState::Connected,
84        });
85        assert_eq!(tester.state, EnsureDeviceCapabilities);
86        assert_eq!(
87            tester.peek_next_state(CallError),
88            Complete(FxaState::Disconnected)
89        );
90        assert_eq!(
91            tester.peek_next_state(EnsureDeviceCapabilitiesSuccess),
92            Complete(FxaState::Connected)
93        );
94
95        tester.next_state(EnsureCapabilitiesAuthError);
96        assert_eq!(tester.state, CheckAuthorizationStatus);
97        assert_eq!(
98            tester.peek_next_state(CallError),
99            Complete(FxaState::AuthIssues)
100        );
101        assert_eq!(
102            tester.peek_next_state(CheckAuthorizationStatusSuccess { active: false }),
103            Complete(FxaState::AuthIssues)
104        );
105        assert_eq!(
106            tester.peek_next_state(CheckAuthorizationStatusSuccess { active: true }),
107            Complete(FxaState::Connected)
108        );
109    }
110}
