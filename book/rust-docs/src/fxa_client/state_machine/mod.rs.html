5//! FxA state machine
6//!
7//! This presents a high-level API for logging in, logging out, dealing with authentication token issues, etc.
8
9use error_support::breadcrumb;

11use crate::{internal::FirefoxAccount, DeviceConfig, Error, FxaEvent, FxaState, Result};

16/// Number of state transitions to perform before giving up and assuming the internal state machine
17/// is stuck in an infinite loop
18const MAX_INTERNAL_TRANSITIONS: usize = 20;
19
20use internal_machines::InternalStateMachine;
21use internal_machines::State as InternalState;
22
23impl FirefoxAccount {
24    /// Get the current state
25    pub fn get_state(&self) -> FxaState {
26        self.auth_state.clone()
27    }
28
29    /// Process an event (login, logout, etc).
30    ///
31    /// On success, returns the new state.
32    /// On error, the state will remain the same.
33    pub fn process_event(&mut self, event: FxaEvent) -> Result<FxaState> {
34        let was_in_auth_issues = matches!(self.auth_state, FxaState::AuthIssues);
35
36        let next_state = match &self.auth_state {
37            FxaState::Uninitialized => self.process_event_with_internal_state_machine(
38                internal_machines::UninitializedStateMachine,
39                event,
40            )?,
41            FxaState::Disconnected => self.process_event_with_internal_state_machine(
42                internal_machines::DisconnectedStateMachine,
43                event,
44            )?,
45            FxaState::Authenticating { .. } => self.process_event_with_internal_state_machine(
46                internal_machines::AuthenticatingStateMachine,
47                event,
48            )?,
49            FxaState::Connected => self.process_event_with_internal_state_machine(
50                internal_machines::ConnectedStateMachine,
51                event,
52            )?,
53            FxaState::AuthIssues => self.process_event_with_internal_state_machine(
54                internal_machines::AuthIssuesStateMachine,
55                event,
56            )?,
57        };
58        if !was_in_auth_issues && matches!(next_state, FxaState::AuthIssues) {
59            self.on_auth_issues();
60        }
61        Ok(next_state)
62    }
63
64    fn process_event_with_internal_state_machine<T: InternalStateMachine>(
65        &mut self,
66        state_machine: T,
67        event: FxaEvent,
68    ) -> Result<FxaState> {
69        let device_config = self.handle_state_machine_initialization(&event)?;
70
71        breadcrumb!("FxaStateMachine.process_event starting: {event}");
72        let mut internal_state = state_machine.initial_state(event)?;
73        let mut count = 0;
74        // Loop through internal state transitions until we reach a terminal state
75        //
76        // See `README.md` for details.
77        loop {
78            count += 1;
79            if count > MAX_INTERNAL_TRANSITIONS {
80                breadcrumb!("FxaStateMachine.process_event finished (MAX_INTERNAL_TRANSITIONS)");
81                return Err(Error::StateMachineLogicError(
82                    "infinite loop detected".to_owned(),
83                ));
84            }
85            match internal_state {
86                InternalState::Complete(new_state) => {
87                    breadcrumb!("FxaStateMachine.process_event finished (Complete({new_state}))");
88                    self.auth_state = new_state.clone();
89                    return Ok(new_state);
90                }
91                InternalState::Cancel => {
92                    breadcrumb!("FxaStateMachine.process_event finished (Cancel)");
93                    return Ok(self.auth_state.clone());
94                }
95                state => {
96                    let event = state.make_call(self, &device_config)?;
97                    let event_msg = event.to_string();
98                    internal_state = state_machine.next_state(state, event)?;
99                    breadcrumb!("FxaStateMachine.process_event {event_msg} -> {internal_state}")
100                }
101            }
102        }
103    }
104
105    /// Handles initialization before we process an event
106    ///
107    /// This checks that the first event we see is `FxaEvent::Initialize` and it returns the
108    /// `DeviceConfig` from that event.
109    fn handle_state_machine_initialization(&mut self, event: &FxaEvent) -> Result<DeviceConfig> {
110        match &event {
111            FxaEvent::Initialize { device_config } => match self.device_config {
112                Some(_) => Err(Error::InvalidStateTransition(
113                    "Initialize already sent".to_owned(),
114                )),
115                None => {
116                    self.device_config = Some(device_config.clone());
117                    Ok(device_config.clone())
118                }
119            },
120            _ => match &self.device_config {
121                Some(device_config) => Ok(device_config.clone()),
122                None => Err(Error::InvalidStateTransition(
123                    "Initialize not yet sent".to_owned(),
124                )),
125            },
126        }
127    }
128}
