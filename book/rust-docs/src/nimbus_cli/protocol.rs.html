5use serde_json::Value;
6
7/// This is the protocol that each app understands.
8///
9/// It is sent to the apps via the start_app command.
10///
11/// Any change to this protocol requires changing the Kotlin and Swift code,
12/// and perhaps the app code itself.
13#[derive(Clone, Debug, Default)]
14pub(crate) struct StartAppProtocol<'a> {
15    pub(crate) reset_db: bool,
16    pub(crate) experiments: Option<&'a Value>,
17    pub(crate) log_state: bool,
18}
