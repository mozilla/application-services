11impl LaunchableApp {
12    pub(crate) fn copy_to_clipboard(
13        &self,
14        app_protocol: &StartAppProtocol,
15        open: &AppOpenArgs,
16    ) -> Result<usize> {
17        let url = self.longform_url(app_protocol, open)?;
18        let len = url.len();
19        if let Err(e) = set_clipboard(url) {
20            anyhow::bail!("Can't copy URL to clipboard: {}", e)
21        };
22
23        Ok(len)
24    }
25
26    pub(crate) fn longform_url(
27        &self,
28        app_protocol: &StartAppProtocol,
29        open: &AppOpenArgs,
30    ) -> Result<String> {
31        let deeplink = match (&open.deeplink, self.app_opening_deeplink()) {
32            (Some(deeplink), _) => deeplink.to_owned(),
33            (_, Some(deeplink)) => join_query(deeplink, "--nimbus-cli&--is-launcher"),
34            _ => anyhow::bail!("A deeplink must be provided"),
35        };
36
37        let url = longform_deeplink_url(deeplink.as_str(), app_protocol)?;
38
39        self.prepend_scheme(url.as_str())
40    }
41
42    fn app_opening_deeplink(&self) -> Option<&str> {
43        match self {
44            Self::Android { open_deeplink, .. } => open_deeplink.as_deref(),
45            Self::Ios { .. } => Some("noop"),
46        }
47    }
48
49    pub(crate) fn deeplink(&self, open: &AppOpenArgs) -> Result<Option<String>> {
50        let deeplink = &open.deeplink;
51        if deeplink.is_none() {
52            return Ok(None);
53        }
54        let deeplink = self.prepend_scheme(deeplink.as_ref().unwrap())?;
55        Ok(Some(deeplink))
56    }
57
58    fn prepend_scheme(&self, deeplink: &str) -> Result<String> {
59        Ok(if deeplink.contains("://") {
60            deeplink.to_string()
61        } else {
62            let scheme = self.mandatory_scheme()?;
63            format!("{scheme}://{deeplink}")
64        })
65    }
66
67    fn mandatory_scheme(&self) -> Result<&str> {
68        match self {
69            Self::Android { scheme, .. } | Self::Ios { scheme, .. } => scheme
70                .as_deref()
71                .ok_or_else(|| anyhow!("A scheme is not defined for this app")),
72        }
73    }
74}

76// The following are the special query percent encode set.
77// https://url.spec.whatwg.org/#query-percent-encode-set
78const QUERY: &AsciiSet = &CONTROLS
79    .add(b' ')
80    .add(b'"')
81    .add(b'<')
82    .add(b'>')
83    .add(b'#')
84    .add(b'\'')
85    // Additionally, we've added '{' and '}' to make  sure iOS simctl works with it.
86    .add(b'{')
87    .add(b'}')
88    // Then some belt and braces: we're quoting a single query attribute value.
89    .add(b':')
90    .add(b'/')
91    .add(b'?')
92    .add(b'&');
93
94/// Construct a URL from the deeplink and the protocol object.
95pub(crate) fn longform_deeplink_url(
96    deeplink: &str,
97    app_protocol: &StartAppProtocol,
98) -> Result<String> {
99    let StartAppProtocol {
100        reset_db,
101        experiments,
102        log_state,
103        jexl_expression,
104    } = app_protocol;
105    if !reset_db && experiments.is_none() && !log_state && jexl_expression.is_none() {
106        return Ok(deeplink.to_string());
107    }
108
109    let mut parts: Vec<_> = Default::default();
110    if !deeplink.contains("--nimbus-cli") {
111        parts.push("--nimbus-cli".to_string());
112    }
113    if let Some(v) = experiments {
114        let json = serde_json::to_string(v)?;
115        let string = percent_encoding::utf8_percent_encode(&json, QUERY).to_string();
116        parts.push(format!("--experiments={string}"));
117    }
118
119    if *reset_db {
120        parts.push("--reset-db".to_string());
121    }
122    if *log_state {
123        parts.push("--log-state".to_string());
124    }
125    if let Some(jexl_expr) = jexl_expression {
126        let string = percent_encoding::utf8_percent_encode(jexl_expr, QUERY).to_string();
127        parts.push(format!("--eval-jexl={string}"));
128    }
129
130    Ok(join_query(deeplink, &parts.join("&")))
131}
132
133fn join_query(url: &str, item: &str) -> String {
134    let suffix = if url.contains('?') { '&' } else { '?' };
135    format!("{url}{suffix}{item}")
136}

138fn set_clipboard(contents: String) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
139    use copypasta::{ClipboardContext, ClipboardProvider};
140    let mut ctx = ClipboardContext::new()?;
141    ctx.set_contents(contents)?;
142    Ok(())
143}

151    #[test]
152    fn test_url_noop() -> Result<()> {
153        let p = StartAppProtocol {
154            reset_db: false,
155            experiments: None,
156            log_state: false,
157            jexl_expression: None,
158        };
159        assert_eq!("host".to_string(), longform_deeplink_url("host", &p)?);
160        assert_eq!(
161            "host?query=1".to_string(),
162            longform_deeplink_url("host?query=1", &p)?
163        );
164        Ok(())
165    }

248    #[test]
249    fn test_deeplink_has_is_launcher_param_if_no_deeplink_is_specified() -> Result<()> {
250        let app =
251            LaunchableApp::try_from_app_channel_device(Some("fenix"), Some("developer"), None)?;
252
253        // No payload, or command line param for deeplink.
254        let payload: StartAppProtocol = Default::default();
255        let open: AppOpenArgs = Default::default();
256        assert_eq!(
257            "fenix-dev://open?--nimbus-cli&--is-launcher".to_string(),
258            app.longform_url(&payload, &open)?
259        );
260
261        // A command line param for deeplink.
262        let open = AppOpenArgs {
263            deeplink: Some("deeplink".to_string()),
264            ..Default::default()
265        };
266        assert_eq!(
267            "fenix-dev://deeplink".to_string(),
268            app.longform_url(&payload, &open)?
269        );
270
271        // A parameter from the payload, but no deeplink.
272        let payload = StartAppProtocol {
273            log_state: true,
274            ..Default::default()
275        };
276        assert_eq!(
277            "fenix-dev://open?--nimbus-cli&--is-launcher&--log-state".to_string(),
278            app.longform_url(&payload, &Default::default())?
279        );
280
281        // A deeplink from the command line, and an extra param from the payload.
282        let open = AppOpenArgs {
283            deeplink: Some("deeplink".to_string()),
284            ..Default::default()
285        };
286        assert_eq!(
287            "fenix-dev://deeplink?--nimbus-cli&--log-state".to_string(),
288            app.longform_url(&payload, &open)?
289        );
290
291        Ok(())
