5use std::collections::hash_map::DefaultHasher;
6use std::hash::{Hash, Hasher};
7use viaduct::Request;
8
9#[derive(Clone, Debug, PartialEq)]
10pub struct RequestHash(String);
11
12impl From<&Request> for RequestHash {
13    fn from(request: &Request) -> Self {
14        let mut hasher = DefaultHasher::new();
15        request.method.hash(&mut hasher);
16        request.url.hash(&mut hasher);
17
18        let mut headers: Vec<_> = request.headers.clone().into_iter().collect();
19        headers.sort_by_key(|header| header.name.to_ascii_lowercase());
20        for header in headers {
21            header.name.hash(&mut hasher);
22            header.value.hash(&mut hasher);
23        }
24
25        // Strip context_id before hashing â€” it rotates on client re-instantiation
26        // and should not invalidate cached responses.
27        // NOTE: This couples ads-client domain logic to the cache. When the cache
28        // is extracted into its own module, this should move to the caller side.
29        let body_for_hash = request.body.as_deref().and_then(|bytes| {
30            serde_json::from_slice::<serde_json::Value>(bytes)
31                .ok()
32                .map(|mut value| {
33                    if let Some(obj) = value.as_object_mut() {
34                        obj.remove("context_id");
35                    }
36                    serde_json::to_vec(&value).unwrap_or_else(|_| bytes.to_vec())
37                })
38        });
39        body_for_hash.hash(&mut hasher);
40
41        RequestHash(format!("{:x}", hasher.finish()))
42    }
43}
44
45impl From<&str> for RequestHash {
46    fn from(s: &str) -> Self {
47        RequestHash(s.to_string())
48    }
49}
50
51impl From<String> for RequestHash {
52    fn from(s: String) -> Self {
53        RequestHash(s)
54    }
55}
56
57impl std::fmt::Display for RequestHash {
58    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
59        write!(f, "{}", self.0)
60    }
61}

68    fn create_test_request(url: &str, body: &[u8]) -> Request {
69        Request {
70            method: Method::Get,
71            url: url.parse().unwrap(),
72            headers: Headers::new(),
73            body: Some(body.to_vec()),
74        }
75    }

77    #[test]
78    fn test_request_hashing() {
79        let request1 = create_test_request("https://example.com/api1", b"body1");
80        let request2 = create_test_request("https://example.com/api2", b"body2");
81        let request3 = create_test_request("https://example.com/api", b"body");
82
83        let hash1 = RequestHash::from(&request1);
84        let hash2 = RequestHash::from(&request2);
85        let hash3 = RequestHash::from(&request3);
86
87        assert_ne!(hash1.to_string(), hash2.to_string());
88        assert_ne!(hash1.to_string(), hash3.to_string());
89        assert_ne!(hash2.to_string(), hash3.to_string());
90
91        let same_request = create_test_request("https://example.com/api", b"body");
92        let hash4 = RequestHash::from(&same_request);
93        assert_eq!(hash3.to_string(), hash4.to_string());
94    }

104        h1.insert("X-Test", "1").unwrap();
105        let req1 = Request {
106            headers: h1,
107            ..req_base.clone()
108        };
109
110        let req_base = create_test_request(base_url, body);
111        let mut h2 = Headers::new();
112        h2.insert("X-Test", "1").unwrap();
113        h2.insert("Accept", "text/plain").unwrap();
114        let req2 = Request {
115            headers: h2,
116            ..req_base.clone()
117        };

119        let req_base = create_test_request(base_url, body);
120        let mut h3 = Headers::new();

122        h3.insert("x-test", "1").unwrap();
