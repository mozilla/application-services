5use rusqlite::Connection;
6use sql_support::open_database;
7use std::time::Duration;
8
9pub struct HttpCacheConnectionInitializer {}
10
11impl open_database::ConnectionInitializer for HttpCacheConnectionInitializer {
12    const NAME: &'static str = "http_cache";
13    const END_VERSION: u32 = 1;
14
15    fn prepare(&self, conn: &Connection, _db_empty: bool) -> open_database::Result<()> {
16        conn.execute_batch("PRAGMA journal_mode=wal;")?;
17        conn.busy_timeout(Duration::from_secs(5))?;
18        Ok(())
19    }
20
21    fn init(&self, tx: &rusqlite::Transaction<'_>) -> open_database::Result<()> {
22        const SCHEMA: &str = "
23            CREATE TABLE IF NOT EXISTS http_cache (
24                cached_at INTEGER NOT NULL,
25                expires_at INTEGER NOT NULL,
26                request_hash TEXT NOT NULL,
27                response_body BLOB NOT NULL,
28                response_headers BLOB,
29                response_status INTEGER NOT NULL,
30                size_bytes INTEGER NOT NULL,
31                ttl_seconds INTEGER NOT NULL,
32                PRIMARY KEY (request_hash)
33            );
34            CREATE INDEX IF NOT EXISTS idx_http_cache_cached_at ON http_cache(cached_at);
35            CREATE INDEX IF NOT EXISTS idx_http_cache_expires_at ON http_cache(expires_at);
36        ";
37        tx.execute_batch(SCHEMA)?;
38        Ok(())
39    }
40
41    fn upgrade_from(
42        &self,
43        conn: &rusqlite::Transaction<'_>,
44        version: u32,
45    ) -> open_database::Result<()> {
46        match version {
47            0 => {
48                // Version 0 means we need to create the initial schema
49                self.init(conn)
50            }
51            _ => Err(open_database::Error::IncompatibleVersion(version)),
52        }
53    }
54}
