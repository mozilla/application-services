14const DEFAULT_MAX_SIZE: ByteSize = ByteSize::mib(10);
15const DEFAULT_TTL: Duration = Duration::from_secs(300);
16
17const MIN_CACHE_SIZE: ByteSize = ByteSize::kib(1);
18const MAX_CACHE_SIZE: ByteSize = ByteSize::mib(100);
19const MIN_TTL: Duration = Duration::from_secs(1);
20const MAX_TTL: Duration = Duration::from_secs(60 * 60 * 24 * 7); // 7 days
21
22#[derive(Debug, thiserror::Error)]
23pub enum Error {
24    #[error("Database path cannot be empty")]
25    EmptyDbPath,
26    #[error("Database error: {0}")]
27    Database(#[from] open_database::Error),
28    #[error(
29        "Maximum cache size must be between {min_size} and {max_size}, got {size_bytes} bytes"
30    )]
31    InvalidMaxSize {
32        max_size: String,
33        min_size: String,
34        size_bytes: u64,
35    },
36    #[error("TTL must be between {min_ttl} and {max_ttl}, got {ttl} seconds")]
37    InvalidTtl {
38        max_ttl: String,
39        min_ttl: String,
40        ttl: u64,
41    },
42}
43
44#[derive(Debug)]
45pub struct HttpCacheBuilder {
46    db_path: PathBuf,
47    max_size: Option<ByteSize>,
48    default_ttl: Option<Duration>,
49}
50
51impl HttpCacheBuilder {
52    pub fn new(db_path: impl Into<PathBuf>) -> Self {
53        Self {
54            db_path: db_path.into(),
55            max_size: None,
56            default_ttl: None,
57        }
58    }
59
60    pub fn max_size(mut self, max_size: ByteSize) -> Self {
61        self.max_size = Some(max_size);
62        self
63    }
64
65    pub fn default_ttl(mut self, ttl: Duration) -> Self {
66        self.default_ttl = Some(ttl);
67        self
68    }
69
70    fn validate(&self) -> Result<(), Error> {
71        if self.db_path.to_string_lossy().trim().is_empty() {
72            return Err(Error::EmptyDbPath);
73        }
74
75        if let Some(max_size) = self.max_size {
76            if max_size < MIN_CACHE_SIZE || max_size > MAX_CACHE_SIZE {
77                return Err(Error::InvalidMaxSize {
78                    size_bytes: max_size.as_u64(),
79                    min_size: MIN_CACHE_SIZE.to_string(),
80                    max_size: MAX_CACHE_SIZE.to_string(),
81                });
82            }
83        }
84
85        if let Some(ttl) = self.default_ttl {
86            if !(MIN_TTL..=MAX_TTL).contains(&ttl) {
87                return Err(Error::InvalidTtl {
88                    ttl: ttl.as_secs(),
89                    min_ttl: format!("{} seconds", MIN_TTL.as_secs()),
90                    max_ttl: format!("{} seconds", MAX_TTL.as_secs()),
91                });
92            }
93        }
94
95        Ok(())
96    }
97
98    pub fn build(self) -> Result<HttpCache, Error> {
99        self.validate()?;
100
101        let initializer = HttpCacheConnectionInitializer {};
102        let conn = if cfg!(test) {
103            open_database::open_memory_database(&initializer)?
104        } else {
105            open_database::open_database(&self.db_path, &initializer)?
106        };
107
108        let max_size = self.max_size.unwrap_or(DEFAULT_MAX_SIZE);
109        let store = HttpCacheStore::new(conn);
110        let default_ttl = self.default_ttl.unwrap_or(DEFAULT_TTL);
111
112        Ok(HttpCache {
113            max_size,
114            store,
115            default_ttl,
116        })
117    }
118}

125    fn test_cache_builder_with_defaults() {
126        let builder = HttpCacheBuilder::new("test.db".to_string());
127        assert_eq!(builder.db_path, PathBuf::from("test.db"));
