14#[uniffi::export(with_foreign)]
15pub trait MozAdsTelemetry: Send + Sync {
16    fn record_build_cache_error(&self, label: String, value: String);
17    fn record_client_error(&self, label: String, value: String);
18    fn record_client_operation_total(&self, label: String);
19    fn record_deserialization_error(&self, label: String, value: String);
20    fn record_http_cache_outcome(&self, label: String, value: String);
21}
22
23pub struct NoopMozAdsTelemetry;
24
25impl MozAdsTelemetry for NoopMozAdsTelemetry {
26    fn record_build_cache_error(&self, _label: String, _value: String) {}
27    fn record_client_error(&self, _label: String, _value: String) {}
28    fn record_client_operation_total(&self, _label: String) {}
29    fn record_deserialization_error(&self, _label: String, _value: String) {}
30    fn record_http_cache_outcome(&self, _label: String, _value: String) {}
31}
32
33#[derive(Clone)]
34pub struct MozAdsTelemetryWrapper {
35    inner: Arc<dyn MozAdsTelemetry>,
36}
37
38impl MozAdsTelemetryWrapper {
39    pub fn new(inner: Arc<dyn MozAdsTelemetry>) -> Self {
40        Self { inner }
41    }
42
43    pub fn noop() -> Self {
44        Self {
45            inner: Arc::new(NoopMozAdsTelemetry),
46        }
47    }
48}
49
50impl Telemetry for MozAdsTelemetryWrapper {
51    fn record(&self, event: &dyn Any) {
52        if let Some(cache_outcome) = event.downcast_ref::<CacheOutcome>() {
53            self.inner.record_http_cache_outcome(
54                match cache_outcome {
55                    CacheOutcome::Hit => "hit".to_string(),
56                    CacheOutcome::LookupFailed(_) => "lookup_failed".to_string(),
57                    CacheOutcome::NoCache => "no_cache".to_string(),
58                    CacheOutcome::MissNotCacheable => "miss_not_cacheable".to_string(),
59                    CacheOutcome::MissStored => "miss_stored".to_string(),
60                    CacheOutcome::StoreFailed(_) => "store_failed".to_string(),
61                    CacheOutcome::CleanupFailed(_) => "cleanup_failed".to_string(),
62                },
63                match cache_outcome {
64                    CacheOutcome::LookupFailed(e) => e.to_string(),
65                    CacheOutcome::StoreFailed(e) => e.to_string(),
66                    CacheOutcome::CleanupFailed(e) => e.to_string(),
67                    _ => "".to_string(),
68                },
69            );
70            return;
71        }
72        if let Some(client_op) = event.downcast_ref::<ClientOperationEvent>() {
73            self.inner.record_client_operation_total(match client_op {
74                ClientOperationEvent::New => "new".to_string(),
75                ClientOperationEvent::RecordClick => "record_click".to_string(),
76                ClientOperationEvent::RecordImpression => "record_impression".to_string(),
77                ClientOperationEvent::ReportAd => "report_ad".to_string(),
78                ClientOperationEvent::RequestAds => "request_ads".to_string(),
79            });
80            return;
81        }
82        if let Some(cache_builder_error) = event.downcast_ref::<HttpCacheBuilderError>() {
83            self.inner.record_build_cache_error(
84                match cache_builder_error {
85                    HttpCacheBuilderError::EmptyDbPath => "empty_db_path".to_string(),
86                    HttpCacheBuilderError::Database(_) => "database_error".to_string(),
87                    HttpCacheBuilderError::InvalidMaxSize { .. } => "invalid_max_size".to_string(),
88                    HttpCacheBuilderError::InvalidTtl { .. } => "invalid_ttl".to_string(),
89                },
90                format!("{}", cache_builder_error),
91            );
92            return;
93        }
94        if let Some(record_click_error) = event.downcast_ref::<RecordClickError>() {
95            self.inner.record_client_error(
96                "record_click".to_string(),
97                format!("{}", record_click_error),
98            );
99            return;
100        }
101        if let Some(record_impression_error) = event.downcast_ref::<RecordImpressionError>() {
102            self.inner.record_client_error(
103                "record_impression".to_string(),
104                format!("{}", record_impression_error),
105            );
106            return;
107        }
108        if let Some(report_ad_error) = event.downcast_ref::<ReportAdError>() {
109            self.inner
110                .record_client_error("report_ad".to_string(), format!("{}", report_ad_error));
111            return;
112        }
113        if let Some(request_ads_error) = event.downcast_ref::<RequestAdsError>() {
114            self.inner
115                .record_client_error("request_ads".to_string(), format!("{}", request_ads_error));
116            return;
117        }
118        if let Some(json_error) = event.downcast_ref::<serde_json::Error>() {
119            self.inner.record_deserialization_error(
120                "invalid_ad_item".to_string(),
121                format!("{}", json_error),
122            );
123            return;
124        }
125        eprintln!("Unsupported telemetry event type: {:?}", event.type_id());
126        #[cfg(test)]
127        panic!("Unsupported telemetry event type: {:?}", event.type_id());
128    }
129}
