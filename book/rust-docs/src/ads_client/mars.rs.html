20static MARS_API_ENDPOINT_PROD: Lazy<Url> =
21    Lazy::new(|| Url::parse("https://ads.mozilla.org/v1/").expect("hardcoded URL must be valid"));
22
23#[cfg(feature = "dev")]
24static MARS_API_ENDPOINT_STAGING: Lazy<Url> =
25    Lazy::new(|| Url::parse("https://ads.allizom.org/v1/").expect("hardcoded URL must be valid"));
26
27impl Environment {
28    pub fn into_mars_url(self) -> &'static Url {
29        match self {
30            Environment::Prod => &MARS_API_ENDPOINT_PROD,
31            #[cfg(feature = "dev")]
32            Environment::Staging => &MARS_API_ENDPOINT_STAGING,
33        }
34    }
35}

37#[cfg_attr(test, mockall::automock)]
38pub trait MARSClient: Sync + Send {
39    fn fetch_ads(
40        &self,
41        request: &AdRequest,
42        cache_policy: &RequestCachePolicy,
43    ) -> Result<AdResponse, FetchAdsError>;
44    fn record_impression(&self, callback: Url) -> Result<(), RecordImpressionError>;
45    fn record_click(&self, callback: Url) -> Result<(), RecordClickError>;
46    fn report_ad(&self, callback: Url) -> Result<(), ReportAdError>;
47    fn get_context_id(&self) -> context_id::ApiResult<String>;
48    fn cycle_context_id(&mut self) -> context_id::ApiResult<String>;
49    fn get_mars_endpoint(&self) -> &Url;
50    fn clear_cache(&self) -> Result<(), HttpCacheError>;
51}
52
53pub struct DefaultMARSClient {
54    context_id_component: ContextIDComponent,
55    endpoint: Url,
56    http_cache: Option<HttpCache>,
57}
58
59impl DefaultMARSClient {
60    pub fn new(
61        context_id: String,
62        environment: Environment,
63        http_cache: Option<HttpCache>,
64    ) -> Self {
65        let endpoint = environment.into_mars_url().clone();
66
67        Self {
68            context_id_component: ContextIDComponent::new(
69                &context_id,
70                0,
71                false,
72                Box::new(DefaultContextIdCallback),
73            ),
74            endpoint,
75            http_cache,
76        }
77    }
78
79    #[cfg(test)]
80    pub fn new_with_endpoint(
81        context_id: String,
82        endpoint: String,
83        http_cache: Option<HttpCache>,
84    ) -> Self {
85        Self {
86            context_id_component: ContextIDComponent::new(
87                &context_id,
88                0,
89                false,
90                Box::new(DefaultContextIdCallback),
91            ),
92            endpoint: Url::parse(endpoint.as_str()).unwrap(),
93            http_cache,
94        }
95    }
96
97    fn make_callback_request(&self, callback: Url) -> Result<(), CallbackRequestError> {
98        let request = Request::get(callback);
99        let response = request.send()?;
100        check_http_status_for_error(&response).map_err(Into::into)
101    }
102}
103
104impl MARSClient for DefaultMARSClient {
105    fn cycle_context_id(&mut self) -> context_id::ApiResult<String> {
106        let old_context_id = self.get_context_id()?;
107        self.context_id_component.force_rotation()?;
108        Ok(old_context_id)
109    }
110
111    fn get_context_id(&self) -> context_id::ApiResult<String> {
112        self.context_id_component.request(0)
113    }
114
115    fn get_mars_endpoint(&self) -> &Url {
116        &self.endpoint
117    }
118
119    fn fetch_ads(
120        &self,
121        ad_request: &AdRequest,
122        cache_policy: &RequestCachePolicy,
123    ) -> Result<AdResponse, FetchAdsError> {
124        let base = self.get_mars_endpoint();
125        let url = base.join("ads")?;
126        let request = Request::post(url).json(ad_request);
127
128        if let Some(cache) = self.http_cache.as_ref() {
129            let outcome = cache.send_with_policy(&request, cache_policy)?;
130
131            // TODO: observe cache outcome for metrics/logging.
132            match &outcome.cache_outcome {
133                CacheOutcome::Hit => {}
134                CacheOutcome::LookupFailed(_err) => {}
135                CacheOutcome::NoCache => {}
136                CacheOutcome::MissNotCacheable => {}
137                CacheOutcome::MissStored => {}
138                CacheOutcome::StoreFailed(_err) => {}
139                CacheOutcome::CleanupFailed(_err) => {}
140            }
141            check_http_status_for_error(&outcome.response)?;
142            let response_json: AdResponse = outcome.response.json()?;
143            Ok(response_json)
144        } else {
145            let response = request.send()?;
146            check_http_status_for_error(&response)?;
147            let response_json: AdResponse = response.json()?;
148            Ok(response_json)
149        }
150    }
151
152    fn record_impression(&self, callback: Url) -> Result<(), RecordImpressionError> {
153        Ok(self.make_callback_request(callback)?)
154    }
155
156    fn record_click(&self, callback: Url) -> Result<(), RecordClickError> {
157        Ok(self.make_callback_request(callback)?)
158    }
159
160    fn report_ad(&self, callback: Url) -> Result<(), ReportAdError> {
161        Ok(self.make_callback_request(callback)?)
162    }
163
164    fn clear_cache(&self) -> Result<(), HttpCacheError> {
165        if let Some(cache) = &self.http_cache {
166            cache.clear()?;
167        }
168        Ok(())
169    }
170}

265    fn test_fetch_ads_cache_hit_skips_network() {
266        viaduct_dev::init_backend_dev();
267        let expected = get_example_happy_ad_response();
268        let _m = mock("POST", "/ads")
269            .with_status(200)
270            .with_header("content-type", "application/json")
271            .with_body(serde_json::to_string(&expected).unwrap())
272            .expect(1) // only first request goes to network
273            .create();
274
275        let client = create_test_client(mockito::server_url());
276        let ad_request = make_happy_ad_request();
277
278        // First call should be a miss then warm the cache
279        assert_eq!(
280            client
281                .fetch_ads(&ad_request, &RequestCachePolicy::default())
282                .unwrap(),
283            expected
284        );
285        // Second call should be a hit
286        assert_eq!(
287            client
288                .fetch_ads(&ad_request, &RequestCachePolicy::default())
289                .unwrap(),
290            expected
291        );
292    }
