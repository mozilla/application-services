16const ENDPOINT: &str = "/ads";
17
18#[derive(Debug, PartialEq, Serialize)]
19pub struct AdRequest {
20    pub context_id: String,
21    /// Skipped to exclude from the request body
22    #[serde(skip)]
23    pub environment: Environment,
24    #[serde(skip_serializing_if = "HashMap::is_empty")]
25    pub flags: AdRequestFlags,
26    #[serde(skip)]
27    pub headers: Headers,
28    #[serde(skip)]
29    pub ohttp: bool,
30    pub placements: Vec<AdPlacementRequest>,
31}
32
33/// Hash implementation intentionally excludes `context_id` as it rotates
34/// on client re-instantiation and should not invalidate cached responses.
35/// `headers` are also excluded as they are request metadata, not cache keys.
36/// `flags` is hashed only when set so non-flag callers keep prior cache keys.
37/// If response shape ever varies, add a version to this hash for variant tracking.
38impl Hash for AdRequest {
39    fn hash<H: Hasher>(&self, state: &mut H) {
40        ENDPOINT.hash(state);
41        self.environment.hash(state);
42        if !self.flags.is_empty() {
43            // HashMap is unordered — sort by key for a stable hash.
44            let mut sorted: Vec<_> = self.flags.iter().collect();
45            sorted.sort_unstable_by_key(|(k, _)| k.as_str());
46            for (k, v) in sorted {
47                k.hash(state);
48                v.hash(state);
49            }
50        }
51        self.ohttp.hash(state);
52        self.placements.hash(state);
53    }
54}
55
56impl From<AdRequest> for Request {
57    fn from(ad_request: AdRequest) -> Self {
58        let url = ad_request.environment.into_url(ENDPOINT);
59        let mut request = Request::post(url).json(&ad_request);
60        request.headers.extend(ad_request.headers);
61        request
62    }
63}
64
65impl AdRequest {
66    pub fn try_new(
67        context_id: String,
68        environment: Environment,
69        flags: AdRequestFlags,
70        ohttp: bool,
71        placements: Vec<AdPlacementRequest>,
72    ) -> Result<Self, BuildRequestError> {
73        if placements.is_empty() {
74            return Err(BuildRequestError::EmptyConfig);
75        };
76
77        let mut request = AdRequest {
78            context_id,
79            environment,
80            flags,
81            headers: Headers::new(),
82            ohttp,
83            placements: vec![],
84        };
85
86        let mut used_placement_ids: HashSet<String> = HashSet::new();
87
88        for ad_placement_request in placements {
89            if used_placement_ids.contains(&ad_placement_request.placement) {
90                return Err(BuildRequestError::DuplicatePlacementId {
91                    placement_id: ad_placement_request.placement.clone(),
92                });
93            }
94
95            request.placements.push(AdPlacementRequest {
96                content: ad_placement_request
97                    .content
98                    .map(|iab_content| AdContentCategory {
99                        categories: iab_content.categories,
100                        taxonomy: iab_content.taxonomy,
101                    }),
102                count: ad_placement_request.count,
103                placement: ad_placement_request.placement.clone(),
104            });
105
106            used_placement_ids.insert(ad_placement_request.placement.clone());
107        }
108
109        Ok(request)
110    }
111}

522    #[test]
523    fn test_endpoint_const_participates_in_hash() {
524        use std::collections::hash_map::DefaultHasher;
525
526        let request = AdRequest::try_new(
527            TEST_CONTEXT_ID.to_string(),
528            Environment::Test,
529            AdRequestFlags::default(),
530            false,
531            vec![AdPlacementRequest {
532                content: None,
533                count: 1,
534                placement: "tile_1".to_string(),
535            }],
536        )
537        .unwrap();
538
539        let mut full = DefaultHasher::new();
540        request.hash(&mut full);
541
542        let mut without_endpoint = DefaultHasher::new();
543        request.environment.hash(&mut without_endpoint);
544        request.ohttp.hash(&mut without_endpoint);
545        request.placements.hash(&mut without_endpoint);
546
547        assert_ne!(
548            full.finish(),
549            without_endpoint.finish(),
550            "ENDPOINT must contribute to AdRequest hash",
551        );
552    }
