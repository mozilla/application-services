14pub type AdsClientApiResult<T> = std::result::Result<T, MozAdsClientApiError>;

16#[derive(Debug, thiserror::Error, uniffi::Error)]
17pub enum MozAdsClientApiError {
18    #[error("Something unexpected occurred.")]
19    Other { reason: String },
20}
21
22impl From<context_id::ApiError> for MozAdsClientApiError {
23    fn from(err: context_id::ApiError) -> Self {
24        MozAdsClientApiError::Other {
25            reason: err.to_string(),
26        }
27    }
28}
29
30impl GetErrorHandling for ComponentError {
31    type ExternalError = MozAdsClientApiError;
32
33    fn get_error_handling(&self) -> ErrorHandling<Self::ExternalError> {
34        ErrorHandling::convert(MozAdsClientApiError::Other {
35            reason: self.to_string(),
36        })
37    }
38}
