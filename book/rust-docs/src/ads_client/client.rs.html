24const DEFAULT_TTL_SECONDS: u64 = 300;
25const DEFAULT_MAX_CACHE_SIZE_MIB: u64 = 10;
26const DEFAULT_ROTATION_DAYS: u8 = 3;

28pub trait ContextIdProvider: Send + Sync {
29    fn context_id(&self) -> context_id::ApiResult<String>;
30}
31
32impl ContextIdProvider for ContextIDComponent {
33    fn context_id(&self) -> context_id::ApiResult<String> {
34        self.request(DEFAULT_ROTATION_DAYS)
35    }

38pub struct AdsClient<T>
39where
40    T: Clone + Telemetry,
41{
42    client: MARSClient<T>,
43    context_id_provider: Box<dyn ContextIdProvider>,
44    telemetry: T,
45}
46
47impl<T> AdsClient<T>
48where
49    T: Clone + Telemetry,
50{
51    pub fn new(client_config: AdsClientConfig<T>) -> Self {
52        let context_id_provider = client_config.context_id_provider.unwrap_or_else(|| {
53            Box::new(ContextIDComponent::new(
54                &Uuid::new_v4().to_string(),
55                0,
56                cfg!(test),
57                Box::new(DefaultContextIdCallback),
58            ))
59        });
60
61        let telemetry = client_config.telemetry;
62        let environment = client_config.environment;
63
64        // Configure the cache if a path is provided.
65        // Defaults for ttl and cache size are also set if unspecified.
66        let http_cache = client_config.cache_config.and_then(|cache_cfg| {
67            let default_cache_ttl = Duration::from_secs(
68                cache_cfg
69                    .default_cache_ttl_seconds
70                    .unwrap_or(DEFAULT_TTL_SECONDS),
71            );
72            let max_cache_size =
73                ByteSize::mib(cache_cfg.max_size_mib.unwrap_or(DEFAULT_MAX_CACHE_SIZE_MIB));
74
75            match HttpCache::builder(cache_cfg.db_path)
76                .max_size(max_cache_size)
77                .default_ttl(default_cache_ttl)
78                .build()
79            {
80                Ok(cache) => Some(cache),
81                Err(e) => {
82                    telemetry.record(&e);
83                    None
84                }
85            }
86        });
87
88        let client = MARSClient::new(environment, http_cache, telemetry.clone());
89        telemetry.record(&ClientOperationEvent::New);
90        Self {
91            client,
92            context_id_provider,
93            telemetry: telemetry.clone(),
94        }
95    }
96

160    pub fn request_image_ads(
161        &self,
162        ad_placement_requests: Vec<AdPlacementRequest>,
163        options: Option<CachePolicy>,
164        ohttp: bool,
165    ) -> Result<HashMap<String, AdImage>, RequestAdsError> {
166        let response = self
167            .request_ads::<AdImage>(ad_placement_requests, options, ohttp)
168            .inspect_err(|e| {
169                self.telemetry.record(e);
170            })?;
171        self.telemetry.record(&ClientOperationEvent::RequestAds);
172        Ok(response.take_first())
173    }
174
175    pub fn request_spoc_ads(
176        &self,
177        ad_placement_requests: Vec<AdPlacementRequest>,
178        options: Option<CachePolicy>,
179        ohttp: bool,
180    ) -> Result<HashMap<String, Vec<AdSpoc>>, RequestAdsError> {
181        let result = self.request_ads::<AdSpoc>(ad_placement_requests, options, ohttp);
182        result
183            .inspect_err(|e| {
184                self.telemetry.record(e);
185            })
186            .map(|response| {
187                self.telemetry.record(&ClientOperationEvent::RequestAds);
188                response.data
189            })
190    }
191
192    pub fn request_tile_ads(
193        &self,
194        ad_placement_requests: Vec<AdPlacementRequest>,
195        options: Option<CachePolicy>,
196        ohttp: bool,
197    ) -> Result<HashMap<String, AdTile>, RequestAdsError> {
198        let result = self.request_ads::<AdTile>(ad_placement_requests, options, ohttp);
199        result
200            .inspect_err(|e| {
201                self.telemetry.record(e);
202            })
203            .map(|response| {
204                self.telemetry.record(&ClientOperationEvent::RequestAds);
205                response.take_first()
206            })
207    }
208
209    fn request_ads<A>(
210        &self,
211        placements: Vec<AdPlacementRequest>,
212        options: Option<CachePolicy>,
213        ohttp: bool,
214    ) -> Result<AdResponse<A>, RequestAdsError>
215    where
216        A: AdResponseValue,
217    {
218        let context_id = self.get_context_id()?;
219        let cache_policy = options.unwrap_or_default();
220        let (mut response, request_hash) =
221            self.client
222                .fetch_ads::<A>(context_id, placements, cache_policy, ohttp)?;
223        response.enrich_callbacks(&request_hash);
224        Ok(response)
225    }
226}
