24const DEFAULT_TTL_SECONDS: u64 = 300;
25const DEFAULT_MAX_CACHE_SIZE_MIB: u64 = 10;
26
27pub struct AdsClient {
28    client: Box<dyn MARSClient>,
29}
30
31impl AdsClient {
32    pub fn new(client_config: Option<AdsClientConfig>) -> Self {
33        let context_id = Uuid::new_v4().to_string();
34
35        let client_config = client_config.unwrap_or_default();
36
37        // Configure the cache if a path is provided.
38        // Defaults for ttl and cache size are also set if unspecified.
39        if let Some(cache_cfg) = client_config.cache_config {
40            let default_cache_ttl = Duration::from_secs(
41                cache_cfg
42                    .default_cache_ttl_seconds
43                    .unwrap_or(DEFAULT_TTL_SECONDS),
44            );
45            let max_cache_size =
46                ByteSize::mib(cache_cfg.max_size_mib.unwrap_or(DEFAULT_MAX_CACHE_SIZE_MIB));
47
48            let http_cache = HttpCache::builder(cache_cfg.db_path)
49                .max_size(max_cache_size)
50                .default_ttl(default_cache_ttl)
51                .build()
52                .ok(); // TODO: handle error with telemetry
53
54            let client = Box::new(DefaultMARSClient::new(
55                context_id,
56                client_config.environment,
57                http_cache,
58            ));
59            return Self { client };
60        }
61
62        let client = Box::new(DefaultMARSClient::new(
63            context_id,
64            client_config.environment,
65            None,
66        ));
67        Self { client }
68    }
69
70    pub fn request_ads(
71        &self,
72        ad_placement_requests: Vec<AdPlacementRequest>,
73        options: Option<RequestCachePolicy>,
74    ) -> Result<HashMap<String, Vec<Ad>>, RequestAdsError> {
75        let ad_request = AdRequest::build(self.client.get_context_id()?, ad_placement_requests)?;
76        let cache_policy = options.unwrap_or_default();
77        let response = self.client.fetch_ads(&ad_request, &cache_policy)?;
78        let placements = response.build_placements(&ad_request)?;
79        Ok(placements)
80    }
81
82    pub fn record_impression(&self, impression_url: Url) -> Result<(), RecordImpressionError> {
83        self.client.record_impression(impression_url)
84    }
85
86    pub fn record_click(&self, click_url: Url) -> Result<(), RecordClickError> {
87        self.client.record_click(click_url)
88    }
89
90    pub fn report_ad(&self, report_url: Url) -> Result<(), ReportAdError> {
91        self.client.report_ad(report_url)?;
92        Ok(())
93    }
94
95    pub fn cycle_context_id(&mut self) -> context_id::ApiResult<String> {
96        self.client.cycle_context_id()
97    }
98
99    pub fn clear_cache(&self) -> Result<(), HttpCacheError> {
100        self.client.clear_cache()
101    }
102}
