17// From https://searchfox.org/mozilla-central/rev/ea63a0888d406fae720cf24f4727d87569a8cab5/services/sync/modules/constants.js#75
18const URI_LENGTH_MAX: usize = 65536;
19
20const MAX_TITLE_CHAR_LENGTH: usize = 512; // We put an upper limit on title sizes for tabs to reduce memory
21const MAX_PAYLOAD_SIZE: usize = 2 * 1024 * 1024; // 2MB, server now 2.5MB? Maybe going to 10? This should come from server config somehow.
22                                                 // How many entries we keep in the `urlHistory` field. In practice only 1 ever supplied, so trimming
23                                                 // this many is always a noop, but we should maybe bring this back?
24const TAB_ENTRIES_LIMIT: usize = 5;
25
26// We try our best to fit as many tabs in a payload as possible, this includes
27// limiting the url history entries, title character count and finally drop enough tabs
28// until we have small enough payload that the server will accept
29pub fn prepare_for_upload(record: &mut TabsRecord) {
30    let mut sanitized_tabs = std::mem::take(&mut record.tabs)
31        .into_iter()
32        .filter_map(|mut tab| {
33            if tab.url_history.is_empty() || !is_url_syncable(&tab.url_history[0]) {
34                return None;
35            }
36            let mut sanitized_history = Vec::with_capacity(TAB_ENTRIES_LIMIT);
37            for url in tab.url_history {
38                if sanitized_history.len() == TAB_ENTRIES_LIMIT {
39                    break;
40                }
41                if is_url_syncable(&url) {
42                    sanitized_history.push(url);
43                }
44            }
45            tab.url_history = sanitized_history;
46            // Potentially truncate the title to some limit
47            tab.title = slice_up_to(tab.title, MAX_TITLE_CHAR_LENGTH);
48            Some(tab)
49        })
50        .collect::<Vec<_>>();
51    // Sort the tabs so when we trim tabs it's the oldest tabs
52    sanitized_tabs.sort_by(|a, b| b.last_used.cmp(&a.last_used));
53    // deduct tab group and window info from the total.
54    let used = payload_support::compute_serialized_size(&record.windows)
55        .unwrap_or_default()
56        .saturating_add(
57            payload_support::compute_serialized_size(&record.tab_groups).unwrap_or_default(),
58        );
59    let size = MAX_PAYLOAD_SIZE.saturating_sub(used);
60    trim_tabs_length(&mut sanitized_tabs, size);
61    record.tabs = sanitized_tabs;
62    info!("prepare_for_upload found {} tabs", record.tabs.len());
63}

65// Try to keep in sync with https://searchfox.org/mozilla-central/rev/2ad13433da20a0749e1e9a10ec0ab49b987c2c8e/modules/libpref/init/all.js#3927
66fn is_url_syncable(url: &str) -> bool {
67    url.len() <= URI_LENGTH_MAX
68        && !(url.starts_with("about:")
69            || url.starts_with("resource:")
70            || url.starts_with("chrome:")
71            || url.starts_with("wyciwyg:")
72            || url.starts_with("blob:")
73            || url.starts_with("file:")
74            || url.starts_with("moz-extension:")
75            || url.starts_with("data:"))
76}

78/// Trim the amount of tabs in a list to fit the specified memory size.
79/// If trimming the tab length fails for some reason, just return the untrimmed tabs.
80fn trim_tabs_length(tabs: &mut Vec<TabsRecordTab>, payload_size_max_bytes: usize) {
81    if let Some(count) = payload_support::try_fit_items(tabs, payload_size_max_bytes).as_some() {
82        tabs.truncate(count.get());
83    }
84}

86// Similar to places/utils.js
87// This method ensures we safely truncate a string up to a certain max_len while
88// respecting char bounds to prevent rust panics. If we do end up truncating, we
89// append an ellipsis to the string
90fn slice_up_to(s: String, max_len: usize) -> String {
91    if max_len >= s.len() {
92        return s;
93    }
94
95    let ellipsis = '\u{2026}';
96    // Ensure we leave space for the ellipsis while still being under the max
97    let mut idx = max_len - ellipsis.len_utf8();
98    while !s.is_char_boundary(idx) {
99        idx -= 1;
100    }
101    let mut new_str = s[..idx].to_string();
102    new_str.push(ellipsis);
103    new_str
104}

112    fn test_is_url_syncable() {
113        assert!(is_url_syncable("https://bobo.com"));
114        assert!(is_url_syncable("ftp://bobo.com"));
115        assert!(!is_url_syncable("about:blank"));
116        // XXX - this smells wrong - we should insist on a valid complete URL?

121    #[test]
122    fn test_trimming_tab_title() {
123        error_support::init_for_tests();
124        let mut record = TabsRecord {
125            id: "me".to_string(),
126            client_name: "name".to_string(),
127            tabs: vec![TabsRecordTab {
128                title: "a".repeat(MAX_TITLE_CHAR_LENGTH + 10), // Fill a string more than max
129                url_history: vec!["https://foo.bar".to_owned()],
130                ..Default::default()
131            }],
132            ..Default::default()
133        };
134        prepare_for_upload(&mut record);
135
136        let ellipsis_char = '\u{2026}';
137        let mut truncated_title = "a".repeat(MAX_TITLE_CHAR_LENGTH - ellipsis_char.len_utf8());
138        truncated_title.push(ellipsis_char);
139        assert_eq!(
140            record.tabs,
141            vec![
142                // title trimmed to 50 characters
143                TabsRecordTab {
144                    title: truncated_title, // title was trimmed to only max char length
145                    url_history: vec!["https://foo.bar".to_owned()],
146                    ..Default::default()
147                },
148            ]
149        );
150    }

152    #[test]
153    fn test_utf8_safe_title_trim() {
154        error_support::init_for_tests();
155        let mut record = TabsRecord {
156            id: "me".to_string(),
157            client_name: "name".to_string(),
158            tabs: vec![
159                TabsRecordTab {
160                    title: "üòç".repeat(MAX_TITLE_CHAR_LENGTH + 10), // Fill a string more than max
161                    url_history: vec!["https://foo.bar".to_owned()],
162                    ..Default::default()
163                },
164                TabsRecordTab {
165                    title: "„Çí".repeat(MAX_TITLE_CHAR_LENGTH + 5), // Fill a string more than max
166                    url_history: vec!["https://foo_jp.bar".to_owned()],
167                    ..Default::default()
168                },
169            ],
170            ..Default::default()
171        };
172        prepare_for_upload(&mut record);
173        // check we truncated correctly.
174        let ellipsis_char = '\u{2026}';
175        // (MAX_TITLE_CHAR_LENGTH - ellipsis / "üòç" bytes)
176        let mut truncated_title = "üòç".repeat(127);
177        // (MAX_TITLE_CHAR_LENGTH - ellipsis / "„Çí" bytes)
178        let mut truncated_jp_title = "„Çí".repeat(169);
179        truncated_title.push(ellipsis_char);
180        truncated_jp_title.push(ellipsis_char);
181        assert_eq!(
182            record.tabs,
183            vec![
184                TabsRecordTab {
185                    title: truncated_title, // title was trimmed to only max char length
186                    url_history: vec!["https://foo.bar".to_owned()],
187                    ..Default::default()
188                },
189                TabsRecordTab {
190                    title: truncated_jp_title, // title was trimmed to only max char length
191                    url_history: vec!["https://foo_jp.bar".to_owned()],
192                    ..Default::default()
193                },
194            ]
195        );
196        // We should be less than max
197        assert!(record.tabs[0].title.chars().count() <= MAX_TITLE_CHAR_LENGTH);
198        assert!(record.tabs[1].title.chars().count() <= MAX_TITLE_CHAR_LENGTH);
199    }

203        error_support::init_for_tests();
204        let example_tab = TabsRecordTab {
205            title: "‚ù§Ô∏è".repeat(MAX_TITLE_CHAR_LENGTH),
206            url_history: vec!["‚û°Ô∏è".repeat(250)],
207            icon: Some("‚ò∫Ô∏è".repeat(250)),

212            client_name: "name".to_string(),
213            ..Default::default()
214        };
215
216        // Given the example, we know that we can fit 440 tabs of this size.
217        for _ in 0..440 {
218            record.tabs.push(example_tab.clone());
219        }
220        let had = record.tabs.len();
221        prepare_for_upload(&mut record);

226        prepare_for_upload(&mut record);
227        assert_eq!(record.tabs.len(), had);
228        assert!(compute_serialized_size(&record).unwrap() <= MAX_PAYLOAD_SIZE);
229    }
