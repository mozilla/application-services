15#[derive(Debug, Clone, Hash, PartialEq, Eq, Serialize, Deserialize, Default)]
16#[serde(rename_all = "camelCase")]
17pub struct TabsRecordTab {
18    pub title: String,
19    pub url_history: Vec<String>,
20    pub icon: Option<String>,
21    pub last_used: i64, // Seconds since epoch!
22    #[serde(default, skip_serializing_if = "skip_if_default")]
23    pub inactive: bool,
24    #[serde(default, skip_serializing_if = "skip_if_default")]
25    pub pinned: bool,
26    #[serde(default, skip_serializing_if = "skip_if_default")]
27    pub index: u32, // the position
28    #[serde(default, skip_serializing_if = "skip_if_default")]
29    pub tab_group_id: String,
30    #[serde(default, skip_serializing_if = "skip_if_default")]
31    pub window_id: String,
32}

34impl From<RemoteTab> for TabsRecordTab {
35    fn from(tab: RemoteTab) -> Self {
36        Self {
37            title: tab.title,
38            url_history: tab.url_history,
39            icon: tab.icon,
40            last_used: tab.last_used.checked_div(1000).unwrap_or_default(),
41            inactive: tab.inactive,
42            pinned: tab.pinned,
43            index: tab.index,
44            tab_group_id: tab.tab_group_id,
45            window_id: tab.window_id,
46        }
47    }
48}

130#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
131#[cfg_attr(test, derive(Default))]
132#[serde(rename_all = "camelCase")]
133/// This struct mirrors what is stored on the server as the top-level payload.
134pub struct TabsRecord {
135    // `String` instead of `SyncGuid` because `SyncGuid` is optimized for short uids,
136    // but we expect these to be long "xxx-xxx-xxx-xxx" FxA device uids.
137    pub id: String,
138    pub client_name: String,
139    pub tabs: Vec<TabsRecordTab>,
140    #[serde(default, skip_serializing_if = "skip_if_default")]
141    pub tab_groups: HashMap<String, TabsRecordTabGroup>,
142    #[serde(default, skip_serializing_if = "skip_if_default")]
143    pub windows: HashMap<String, TabsRecordWindow>,
144}

146#[cfg(test)]
147pub mod test {
148    use super::*;
149    use serde_json::json;

151    #[test]
152    fn test_payload() {
153        let payload = json!({
154            "id": "JkeBPC50ZI0m",
155            "clientName": "client name",
156            "tabs": [{
157                "title": "the title",
158                "urlHistory": [
159                    "https://mozilla.org/"
160                ],
161                "icon": "https://mozilla.org/icon",
162                "lastUsed": 1643764207
163            }]
164        });
165        let record: TabsRecord = serde_json::from_value(payload).expect("should work");
166        assert_eq!(record.id, "JkeBPC50ZI0m");
167        assert_eq!(record.client_name, "client name");
168        assert_eq!(record.tabs.len(), 1);
169        assert_eq!(record.windows.len(), 0);
170        assert_eq!(record.tab_groups.len(), 0);
171        let tab = &record.tabs[0];
172        assert_eq!(tab.title, "the title");
173        assert_eq!(tab.icon, Some("https://mozilla.org/icon".to_string()));
174        assert_eq!(tab.last_used, 1643764207);
175        assert!(!tab.inactive);
176    }
177
178    #[test]
179    fn test_roundtrip() {
180        let tab = TabsRecord {
181            id: "JkeBPC50ZI0m".into(),
182            client_name: "client name".into(),
183            tabs: vec![TabsRecordTab {
184                title: "the title".into(),
185                url_history: vec!["https://mozilla.org/".into()],
186                icon: Some("https://mozilla.org/icon".into()),
187                last_used: 1643764207,
188                inactive: true,
189                ..Default::default()
190            }],
191            tab_groups: HashMap::new(),
192            windows: HashMap::new(),
193        };
194        let round_tripped =
195            serde_json::from_value(serde_json::to_value(tab.clone()).unwrap()).unwrap();
196        assert_eq!(tab, round_tripped);
197    }

199    #[test]
200    fn test_extra_fields() {
201        let payload = json!({
202            "id": "JkeBPC50ZI0m",
203            // Let's say we agree on new tabs to record, we want old versions to
204            // ignore them!
205            "ignoredField": "??",
206            "foo": [1, 2, 3],
207            "bar": [{"id": 1}],
208            "clientName": "client name",
209            "tabs": [{
210                "title": "the title",
211                "urlHistory": [
212                    "https://mozilla.org/"
213                ],
214                "icon": "https://mozilla.org/icon",
215                "lastUsed": 1643764207,
216                // Ditto - make sure we ignore unexpected fields in each tab.
217                "ignoredField": "??",
218            }]
219        });
220        let record: TabsRecord = serde_json::from_value(payload).unwrap();
221        // The point of this test is really just to ensure the deser worked, so
222        // just check the ID.
223        assert_eq!(record.id, "JkeBPC50ZI0m");
224    }

227    fn test_windows_tab_groups() {
228        let payload = json!({
229            "id": "JkeBPC50ZI0m",
230            "clientName": "client name",
231            "tabs": [{
232                "title": "the title",
233                "urlHistory": [

242                    "lastUsed": 1,
