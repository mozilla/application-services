5use serde_derive::*;
6use std::collections::HashMap;
7use sync_guid::Guid;
8
9// Known record formats.
10
11#[derive(Serialize, Deserialize, Debug, Clone)]
12pub struct MetaGlobalEngine {
13    pub version: usize,
14    #[serde(rename = "syncID")]
15    pub sync_id: Guid,
16}
17
18#[derive(Serialize, Deserialize, Debug, Clone)]
19pub struct MetaGlobalRecord {
20    #[serde(rename = "syncID")]
21    pub sync_id: Guid,
22    #[serde(rename = "storageVersion")]
23    pub storage_version: usize,
24    #[serde(default)]
25    pub engines: HashMap<String, MetaGlobalEngine>,
26    #[serde(default)]
27    pub declined: Vec<String>,
28}
29
30#[derive(Deserialize, Serialize, Clone, Debug, Eq, PartialEq)]
31pub struct CryptoKeysRecord {
32    pub id: Guid,
33    pub collection: String,
34    pub default: [String; 2],
35    pub collections: HashMap<String, [String; 2]>,
36}

38#[cfg(test)]
39#[test]
40fn test_deserialize_meta_global() {
41    let record = serde_json::json!({
42        "syncID": "abcd1234abcd",
43        "storageVersion": 1,
44    })
45    .to_string();
46    let r = serde_json::from_str::<MetaGlobalRecord>(&record).unwrap();
47    assert_eq!(r.sync_id, "abcd1234abcd");
48    assert_eq!(r.storage_version, 1);
49    assert_eq!(r.engines.len(), 0);
50    assert_eq!(r.declined.len(), 0);
51}
