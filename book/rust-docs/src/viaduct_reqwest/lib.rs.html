10// Note: we don't `use` things from reqwest or the viaduct crate because
11// it would be rather confusing given that we have the same name for
12// most things as them.
13
14static CLIENT: Lazy<reqwest::blocking::Client> = Lazy::new(|| {
15    let settings = GLOBAL_SETTINGS.read();
16    let mut builder = reqwest::blocking::ClientBuilder::new()
17        .timeout(settings.read_timeout)
18        .connect_timeout(settings.connect_timeout)
19        .redirect(if settings.follow_redirects {
20            reqwest::redirect::Policy::default()
21        } else {
22            reqwest::redirect::Policy::none()
23        });
24    if cfg!(target_os = "ios") {
25        // The FxA servers rely on the UA agent to filter
26        // some push messages directed to iOS devices.
27        // This is obviously a terrible hack and we should
28        // probably do https://github.com/mozilla/application-services/issues/1326
29        // instead, but this will unblock us for now.
30        builder = builder.user_agent("Firefox-iOS-FxA/24");
31    }
32    // Note: no cookie or cache support.
33    builder
34        .build()
35        .expect("Failed to initialize global reqwest::Client")
36});

38#[allow(clippy::unnecessary_wraps)] // not worth the time to untangle
39fn into_reqwest(
40    request: viaduct::Request,
41) -> Result<reqwest::blocking::Request, viaduct::ViaductError> {
42    let method = match request.method {
43        viaduct::Method::Get => reqwest::Method::GET,
44        viaduct::Method::Head => reqwest::Method::HEAD,
45        viaduct::Method::Post => reqwest::Method::POST,
46        viaduct::Method::Put => reqwest::Method::PUT,
47        viaduct::Method::Delete => reqwest::Method::DELETE,
48        viaduct::Method::Connect => reqwest::Method::CONNECT,
49        viaduct::Method::Options => reqwest::Method::OPTIONS,
50        viaduct::Method::Trace => reqwest::Method::TRACE,
51        viaduct::Method::Patch => reqwest::Method::PATCH,
52    };
53    let mut result = reqwest::blocking::Request::new(method, request.url);
54    for h in request.headers {
55        use reqwest::header::{HeaderName, HeaderValue};
56        // Unwraps should be fine, we verify these in `Header`
57        let value = HeaderValue::from_str(h.value()).unwrap();
58        result
59            .headers_mut()
60            .insert(HeaderName::from_bytes(h.name().as_bytes()).unwrap(), value);
61    }
62    *result.body_mut() = request.body.map(reqwest::blocking::Body::from);
63    Ok(result)
64}

66pub struct ReqwestBackend;
67impl Backend for ReqwestBackend {
68    fn send(&self, request: viaduct::Request) -> Result<viaduct::Response, viaduct::ViaductError> {
69        viaduct::note_backend("reqwest (untrusted)");
70        let request_method = request.method;
71        let req = into_reqwest(request)?;
72        let mut resp = CLIENT
73            .execute(req)
74            .map_err(|e| viaduct::ViaductError::NetworkError(e.to_string()))?;
75        let status = resp.status().as_u16();
76        let url = resp.url().clone();
77        let mut body = Vec::with_capacity(resp.content_length().unwrap_or_default() as usize);
78        resp.read_to_end(&mut body).map_err(|e| {
79            error!("Failed to get body from response: {:?}", e);
80            viaduct::ViaductError::NetworkError(e.to_string())
81        })?;
82        let mut headers = viaduct::Headers::with_capacity(resp.headers().len());
83        for (k, v) in resp.headers() {
84            let val = String::from_utf8_lossy(v.as_bytes()).to_string();
85            let hname = match viaduct::HeaderName::new(k.as_str().to_owned()) {
86                Ok(name) => name,
87                Err(e) => {
88                    // Ignore headers with invalid names, since nobody can look for them anyway.
89                    warn!("Server sent back invalid header name: '{}'", e);
90                    continue;
91                }
92            };
93            // Not using Header::new since the error it returns is for request headers.
94            headers.insert_header(viaduct::Header::new_unchecked(hname, val));
95        }
96        Ok(viaduct::Response {
97            request_method,
98            url,
99            status,
100            headers,
101            body,
102        })
103    }
104}

106static INIT_REQWEST_BACKEND: Once = Once::new();
107
108pub fn use_reqwest_backend() {
109    INIT_REQWEST_BACKEND.call_once(|| {
110        viaduct::set_backend(Box::leak(Box::new(ReqwestBackend)))
111            .expect("Backend already set (FFI)");
112    })
113}
