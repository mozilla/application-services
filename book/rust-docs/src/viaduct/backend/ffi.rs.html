14impl From<crate::Request> for msg_types::Request {
15    fn from(request: crate::Request) -> Self {
16        let settings = GLOBAL_SETTINGS.read();
17        msg_types::Request {
18            url: request.url.to_string(),
19            body: request.body,
20            // Real weird that this needs to be specified as an i32, but
21            // it certainly makes it convenient for us...
22            method: request.method as i32,
23            headers: request.headers.into(),
24            follow_redirects: settings.follow_redirects,
25            use_caches: settings.use_caches,
26            connect_timeout_secs: settings.connect_timeout.map_or(0, |d| d.as_secs() as i32),
27            read_timeout_secs: settings.read_timeout.map_or(0, |d| d.as_secs() as i32),
28        }
29    }
30}

32macro_rules! backend_error {
33    ($($args:tt)*) => {{
34        let msg = format!($($args)*);
35        error!("{}", msg);
36        ViaductError::BackendError(msg)
37    }};

40pub struct FfiBackend;
41impl Backend for FfiBackend {
42    fn send(&self, request: crate::Request) -> Result<crate::Response, ViaductError> {
43        use ffi_support::IntoFfi;
44        use prost::Message;
45        super::note_backend("FFI (trusted)");
46
47        let method = request.method;
48        let fetch = callback_holder::get_callback().ok_or(ViaductError::BackendNotInitialized)?;
49        let proto_req: msg_types::Request = request.into();
50        let buf = proto_req.into_ffi_value();
51        let response = unsafe { fetch(buf) };
52        // This way we'll Drop it if we panic, unlike if we just got a slice into
53        // it. Besides, we already own it.
54        let response_bytes = response.destroy_into_vec();
55
56        let response: msg_types::Response = match Message::decode(response_bytes.as_slice()) {
57            Ok(v) => v,
58            Err(e) => {
59                panic!(
60                    "Failed to parse protobuf returned from fetch callback! {}",
61                    e
62                );
63            }
64        };
65
66        if let Some(exn) = response.exception_message {
67            return Err(ViaductError::NetworkError(format!("Java error: {:?}", exn)));
68        }
69        let status = response
70            .status
71            .ok_or_else(|| backend_error!("Missing HTTP status"))?;
72
73        if status < 0 || status > i32::from(u16::MAX) {
74            return Err(backend_error!("Illegal HTTP status: {}", status));
75        }
76
77        let mut headers = crate::Headers::with_capacity(response.headers.len());
78        for (name, val) in response.headers {
79            let hname = match crate::HeaderName::new(name) {
80                Ok(name) => name,
81                Err(e) => {
82                    // Ignore headers with invalid names, since nobody can look for them anyway.
83                    warn!("Server sent back invalid header name: '{}'", e);
84                    continue;
85                }
86            };
87            // Not using Header::new since the error it returns is for request headers.
88            headers.insert_header(crate::Header::new_unchecked(hname, val));
89        }
90
91        let url = url::Url::parse(
92            &response
93                .url
94                .ok_or_else(|| backend_error!("Response has no URL"))?,
95        )
96        .map_err(|e| backend_error!("Response has illegal URL: {}", e))?;
97
98        Ok(crate::Response {
99            url,
100            request_method: method,
101            body: response.body.unwrap_or_default(),
102            status: status as u16,
103            headers,
104        })
105    }
106}

108/// Type of the callback we need callers on the other side of the FFI to
109/// provide.
110///
111/// Takes and returns a ffi_support::ByteBuffer. (TODO: it would be nice if we could
112/// make this take/return pointers, so that we could use JNA direct mapping. Maybe
113/// we need some kind of ThinBuffer?)
114///
115/// This is a bit weird, since it requires us to allow code on the other side of
116/// the FFI to allocate a ByteBuffer from us, but it works.
117///
118/// The code on the other side of the FFI is responsible for freeing the ByteBuffer
119/// it's passed using `viaduct_destroy_bytebuffer`.
120type FetchCallback = unsafe extern "C" fn(ByteBuffer) -> ByteBuffer;
121
122/// Module that manages get/set of the global fetch callback pointer.
123mod callback_holder {
124    use super::FetchCallback;
125    use crate::error;
126    use std::sync::atomic::{AtomicUsize, Ordering};
127
128    /// Note: We only assign to this once.
129    static CALLBACK_PTR: AtomicUsize = AtomicUsize::new(0);
130
131    // Overly-paranoid sanity checking to ensure that these types are
132    // convertible between each-other. `transmute` actually should check this for
133    // us too, but this helps document the invariants we rely on in this code.
134    //
135    // Note that these are guaranteed by
136    // https://rust-lang.github.io/unsafe-code-guidelines/layout/function-pointers.html
137    // and thus this is a little paranoid.
138    ffi_support::static_assert!(
139        STATIC_ASSERT_USIZE_EQ_FUNC_SIZE,
140        std::mem::size_of::<usize>() == std::mem::size_of::<FetchCallback>()
141    );
142
143    ffi_support::static_assert!(
144        STATIC_ASSERT_USIZE_EQ_OPT_FUNC_SIZE,
145        std::mem::size_of::<usize>() == std::mem::size_of::<Option<FetchCallback>>()
146    );
147
148    /// Get the function pointer to the FetchCallback. Panics if the callback
149    /// has not yet been initialized.
150    pub(super) fn get_callback() -> Option<FetchCallback> {
151        let ptr_value = CALLBACK_PTR.load(Ordering::SeqCst);
152        unsafe { std::mem::transmute::<usize, Option<FetchCallback>>(ptr_value) }
153    }
154
155    /// Set the function pointer to the FetchCallback. Returns false if we did nothing because the callback had already been initialized
156    pub(super) fn set_callback(h: FetchCallback) -> bool {
157        let as_usize = h as usize;
158        match CALLBACK_PTR.compare_exchange(0, as_usize, Ordering::SeqCst, Ordering::SeqCst) {
159            Ok(_) => true,
160            Err(_) => {
161                // This is an internal bug, the other side of the FFI should ensure
162                // it sets this only once. Note that this is actually going to be
163                // before logging is initialized in practice, so there's not a lot
164                // we can actually do here.
165                error!("Bug: Initialized CALLBACK_PTR multiple times");
166                false
167            }
168        }
169    }
170}

172/// Return a ByteBuffer of the requested size. This is used to store the
173/// response from the callback.
174#[no_mangle]
175pub extern "C" fn viaduct_alloc_bytebuffer(sz: i32) -> ByteBuffer {
176    let mut error = ffi_support::ExternError::default();
177    let buffer =
178        ffi_support::call_with_output(&mut error, || ByteBuffer::new_with_size(sz.max(0) as usize));
179    error.consume_and_log_if_error();
180    buffer
181}

183#[no_mangle]
184pub extern "C" fn viaduct_log_error(s: FfiStr<'_>) {
185    let mut error = ffi_support::ExternError::default();
186    ffi_support::call_with_output(&mut error, || error!("Viaduct Ffi Error: {}", s.as_str()));
187    error.consume_and_log_if_error();
188}
189
190#[no_mangle]
191pub extern "C" fn viaduct_initialize(callback: FetchCallback) -> u8 {
192    ffi_support::abort_on_panic::call_with_output(|| callback_holder::set_callback(callback))
193}

195/// Allows connections to the hard-coded address the Android Emulator uses for
196/// localhost. It would be easy to support allowing the address to be passed in,
197/// but we've made a decision to avoid that possible footgun. The expectation is
198/// that this will only be called in debug builds or if the app can determine it
199/// is in the emulator, but the Rust code doesn't know that, so we can't check.
200#[no_mangle]
201pub extern "C" fn viaduct_allow_android_emulator_loopback() {
202    let mut error = ffi_support::ExternError::default();
203    ffi_support::call_with_output(&mut error, || {
204        allow_android_emulator_loopback();
205    });
206    error.consume_and_log_if_error();
207}
