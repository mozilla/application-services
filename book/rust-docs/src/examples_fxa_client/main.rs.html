54#[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord, ValueEnum)]
55enum Server {
56    /// Official server
57    Release,
58    /// China server
59    China,
60    /// stable dev sever
61    Stable,
62    /// staging dev sever
63    Stage,
64    /// local dev sever
65    LocalDev,
66    /// custom sever URL
67    Custom,
68}

70#[derive(Subcommand)]
71enum Command {
72    Devices(devices::DeviceArgs),
73    SendTab(send_tab::SendTabArgs),
74    /// Get a new access token for a scope
75    ///
76    /// This can be used to test the token exchange API for relay
77    /// (https://bugzilla.mozilla.org/show_bug.cgi?id=2012143).
78    ///
79    /// * Start with a fresh account by using the `cargo fxa disconnect`
80    /// * Run `cargo fxa --log --trace get-access-token https://identity.mozilla.com/apps/relay`.
81    ///   * You should see a token exchange request in the trace logs.
82    /// * Run `cargo fxa --log --trace get-access-token https://identity.mozilla.com/apps/relay`
83    ///   again.
84    ///   * This time there shouldn't be a token exchange request, since the refresh token now has
85    ///     the relay scope.
86    GetAccessToken {
87        scope: String,
88    },
89    Disconnect,
90}

92fn main() -> Result<()> {
93    let cli = Cli::parse();
94    nss::ensure_initialized();
95    viaduct_hyper::viaduct_init_backend_hyper()?;
96    if cli.log {
97        if cli.trace {
98            init_logging_with("fxa_client=trace");
99        } else if cli.debug {
100            init_logging_with("fxa_client=debug");
101        } else if cli.info {
102            init_logging_with("fxa_client=info");
103        } else {
104            init_logging_with("fxa_client=warn");
105        }
106    }
107
108    let scopes: &[&str] = if cli.session_scope {
109        &[fxa_creds::SYNC_SCOPE, fxa_creds::SESSION_SCOPE]
110    } else {
111        &[fxa_creds::SYNC_SCOPE]
112    };
113
114    println!();
115    let account = load_account(&cli, scopes)?;
116    match cli.command {
117        Command::Devices(args) => devices::run(&account, args),
118        Command::SendTab(args) => send_tab::run(&account, args),
119        Command::GetAccessToken { scope } => {
120            println!("Requesting access token with scope: {scope}");
121            account.get_access_token(&scope, false)?;
122            println!("Saving account with updated token");
123            persist_fxa_state(&account)?;
124            println!("Success");
125            Ok(())
126        }
127        Command::Disconnect => {
128            account.disconnect();
129            Ok(())
130        }
131    }?;
132
133    Ok(())
134}

136impl Cli {
137    fn server(&self) -> Result<FxaServer> {
138        Ok(match &self.server {
139            None => FxaServer::Release,
140            Some(Server::Release) => FxaServer::Release,
141            Some(Server::Stable) => FxaServer::Stable,
142            Some(Server::Stage) => FxaServer::Stage,
143            Some(Server::China) => FxaServer::China,
144            Some(Server::LocalDev) => FxaServer::LocalDev,
145            Some(Server::Custom) => FxaServer::Custom {
146                url: match &self.custom_url {
147                    Some(url) => url.clone(),
148                    None => bail!("--custom-url missing"),
149                },
150            },
151        })
152    }
153}
154
155fn load_account(cli: &Cli, scopes: &[&str]) -> Result<FirefoxAccount> {
156    let server = cli.server()?;
157    let redirect_uri = format!("{}/oauth/success/a2270f727f45f648", server.content_url());
158    let config = FxaConfig {
159        server,
160        redirect_uri,
161        client_id: CLIENT_ID.into(),
162        token_server_url_override: None,
163    };
164    fxa_creds::get_cli_fxa(config, &credentials_path(), scopes).map(|cli| cli.account)
165}
166
167pub fn persist_fxa_state(acct: &FirefoxAccount) -> Result<()> {
168    let json = acct.to_json().unwrap();
169    Ok(fs::write(credentials_path(), json)?)
170}
171
172fn credentials_path() -> String {
173    cli_support::cli_data_path(CREDENTIALS_FILENAME)
