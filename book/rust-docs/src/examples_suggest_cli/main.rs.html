21#[derive(Debug, Parser)]
22#[command(about, long_about = None)]
23struct Cli {

25    remote_settings_server: Option<RemoteSettingsServerArg>,
26    #[arg(short = 'b')]
27    remote_settings_bucket: Option<String>,

31    #[command(subcommand)]
32    command: Commands,
33}

35#[derive(Clone, Debug, ValueEnum)]
36enum RemoteSettingsServerArg {
37    Prod,
38    Stage,
39    Dev,
40}

42#[derive(Debug, Subcommand)]
43enum Commands {
44    /// Ingest data
45    Ingest {
46        #[clap(long, short, action)]
47        reingest: bool,
48        #[clap(long, short)]
49        providers: Vec<SuggestionProviderArg>,
50    },
51    /// Query against ingested data
52    Query {
53        #[arg(long, action)]
54        fts_match_info: bool,
55        #[clap(long, short)]
56        provider: Option<SuggestionProviderArg>,
57        /// Input to search
58        input: String,
59        #[clap(long, short)]
60        amp_matching_strategy: Option<AmpMatchingStrategyArg>,
61    },
62}

64#[derive(Clone, Debug, ValueEnum)]
65enum AmpMatchingStrategyArg {
66    /// Use keyword matching, without keyword expansion
67    NoKeyword = 1, // Use `1` as the starting discriminant, since the JS code assumes this.
68    /// Use FTS matching
69    Fts,
70    /// Use FTS matching against the title
71    FtsTitle,
72}
73
74impl From<AmpMatchingStrategyArg> for AmpMatchingStrategy {
75    fn from(val: AmpMatchingStrategyArg) -> Self {
76        match val {
77            AmpMatchingStrategyArg::NoKeyword => AmpMatchingStrategy::NoKeywordExpansion,
78            AmpMatchingStrategyArg::Fts => AmpMatchingStrategy::FtsAgainstFullKeywords,
79            AmpMatchingStrategyArg::FtsTitle => AmpMatchingStrategy::FtsAgainstTitle,
80        }
81    }
82}

95impl From<SuggestionProviderArg> for SuggestionProvider {

97        match value {

99            SuggestionProviderArg::Wikipedia => Self::Wikipedia,
100            SuggestionProviderArg::Amo => Self::Amo,
101            SuggestionProviderArg::Yelp => Self::Yelp,
102            SuggestionProviderArg::Mdn => Self::Mdn,
103            SuggestionProviderArg::Weather => Self::Weather,
104            SuggestionProviderArg::Fakespot => Self::Fakespot,
105        }

109fn main() -> Result<()> {
110    let cli = Cli::parse();
111    cli_support::init_logging_with(if cli.verbose {
112        DEFAULT_LOG_FILTER_VERBOSE
113    } else {
114        DEFAULT_LOG_FILTER
115    });
116    nss::ensure_initialized();
117    viaduct_hyper::init_backend_hyper()?;
118    let store = build_store(&cli)?;
119    match cli.command {
120        Commands::Ingest {
121            reingest,
122            providers,
123        } => ingest(&store, reingest, providers, cli.verbose),
124        Commands::Query {
125            provider,
126            input,
127            fts_match_info,
128            amp_matching_strategy,
129        } => query(
130            &store,
131            provider,
132            input,
133            fts_match_info,
134            amp_matching_strategy,
135            cli.verbose,
136        ),
137    };
138    Ok(())
139}

141fn build_store(cli: &Cli) -> Result<Arc<SuggestStore>> {
142    Ok(Arc::new(SuggestStoreBuilder::default())
143        .data_path(cli_support::cli_data_path(DB_FILENAME))
144        .remote_settings_service(build_remote_settings_service(cli))
145        .build()?)
146}
147
148fn build_remote_settings_service(cli: &Cli) -> Arc<RemoteSettingsService> {
149    let config = RemoteSettingsConfig2 {
150        server: cli.remote_settings_server.as_ref().map(|s| match s {
151            RemoteSettingsServerArg::Dev => RemoteSettingsServer::Dev,
152            RemoteSettingsServerArg::Stage => RemoteSettingsServer::Stage,
153            RemoteSettingsServerArg::Prod => RemoteSettingsServer::Prod,
154        }),
155        bucket_name: cli.remote_settings_bucket.clone(),
156        app_context: None,
157    };
158    let storage_dir = cli_support::cli_data_subdir("remote-settings-data");
159    Arc::new(RemoteSettingsService::new(storage_dir, config))
160}

162fn ingest(
163    store: &SuggestStore,
164    reingest: bool,
165    providers: Vec<SuggestionProviderArg>,
166    verbose: bool,
167) {
168    if reingest {
169        print_header("Reingesting data");
170        store.force_reingest();
171    } else {
172        print_header("Ingesting data");
173    }
174    let constraints = if providers.is_empty() {
175        SuggestIngestionConstraints::all_providers()
176    } else {
177        SuggestIngestionConstraints {
178            providers: Some(providers.into_iter().map(Into::into).collect()),
179            ..SuggestIngestionConstraints::default()
180        }
181    };
182
183    let metrics = store
184        .ingest(constraints)
185        .unwrap_or_else(|e| panic!("Error in ingest: {e}"));
186    if verbose && !metrics.ingestion_times.is_empty() {
187        print_header("Ingestion times");
188        let mut ingestion_times = metrics.ingestion_times;
189        let download_times: HashMap<String, u64> = metrics
190            .download_times
191            .into_iter()
192            .map(|s| (s.label, s.value))
193            .collect();
194
195        ingestion_times.sort_by_key(|s| s.value);
196        ingestion_times.reverse();
197        for sample in ingestion_times {
198            let label = &sample.label;
199            let ingestion_time = sample.value / 1000;
200            let download_time = download_times.get(label).unwrap_or(&0) / 1000;
201
202            println!(
203                "{label:30} Download: {download_time:>5}ms    Ingestion: {ingestion_time:>5}ms"
204            );
205        }
206    }
207    print_header("Done");
208}

210fn query(
211    store: &SuggestStore,
212    provider: Option<SuggestionProviderArg>,
213    input: String,
214    fts_match_info: bool,
215    amp_matching_strategy: Option<AmpMatchingStrategyArg>,
216    verbose: bool,
217) {
218    let query = SuggestionQuery {
219        providers: match provider {
220            Some(provider) => vec![provider.into()],
221            None => SuggestionProvider::all().to_vec(),
222        },
223        keyword: input,
224        provider_constraints: Some(SuggestionProviderConstraints {
225            amp_alternative_matching: amp_matching_strategy.map(Into::into),
226            ..SuggestionProviderConstraints::default()
227        }),
228        ..SuggestionQuery::default()
229    };
230    let mut results = store
231        .query_with_metrics(query)
232        .unwrap_or_else(|e| panic!("Error querying store: {e}"));
233    if results.suggestions.is_empty() {
234        print_header("No Results");
235    } else {
236        print_header("Results");
237        let count = results.suggestions.len();
238        for suggestion in results.suggestions {
239            let title = suggestion.title();
240            let url = suggestion.url().unwrap_or("[no-url]");
241            let icon = if suggestion.icon_data().is_some() {
242                "with icon"
243            } else {
244                "no icon"
245            };
246            println!("* {title} ({url}) ({icon})");
247            if fts_match_info {
248                if let Some(match_info) = suggestion.fts_match_info() {
249                    println!("   {match_info:?}")
250                } else {
251                    println!("   <no match info>");
252                }
253                println!("+ {} other sugestions", count - 1);
254                break;
255            }
256        }
257    }
258    if verbose {
259        print_header("Query times");
260        results.query_times.sort_by_key(|s| s.value);
261        results.query_times.reverse();
262        for s in results.query_times {
263            println!("{:33} Time: {:>5}us", s.label, s.value);
264        }
265    }
266}

268fn print_header(msg: impl Into<String>) {
269    let mut msg = msg.into();
270    if msg.len() % 2 == 1 {
271        msg.push(' ');
272    }
273    let width = (70 - msg.len() - 2) / 2;
274    println!();
275    println!("{} {msg} {}", "=".repeat(width), "=".repeat(width));
276}
