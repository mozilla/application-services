11/// The validity of a tag.
12#[derive(Clone, Copy, Debug, Eq, PartialEq)]
13pub enum ValidatedTag<'a> {
14    /// The tag is invalid.
15    Invalid(&'a str),
16
17    /// The tag is valid, but normalized to remove leading and trailing
18    /// whitespace.
19    Normalized(&'a str),
20
21    /// The original tag is valid.
22    Original(&'a str),
23}
24
25impl<'a> ValidatedTag<'a> {
26    /// Returns `true` if the original tag is valid; `false` if it's invalid or
27    /// normalized.
28    pub fn is_original(&self) -> bool {
29        matches!(&self, ValidatedTag::Original(_))
30    }
31
32    /// Returns the tag string if the tag is valid or normalized, or an error
33    /// if the tag is invalid.
34    pub fn ensure_valid(&self) -> Result<&'a str> {
35        match self {
36            ValidatedTag::Invalid(_) => Err(InvalidPlaceInfo::InvalidTag.into()),
37            ValidatedTag::Normalized(t) | ValidatedTag::Original(t) => Ok(t),
38        }
39    }
40}
41
42/// Checks the validity of the specified tag.
43pub fn validate_tag(tag: &str) -> ValidatedTag<'_> {
44    // Drop empty and oversized tags.
45    let t = tag.trim();
46    if t.is_empty() || t.len() > TAG_LENGTH_MAX {
47        ValidatedTag::Invalid(tag)
48    } else if t.len() != tag.len() {
49        ValidatedTag::Normalized(t)
50    } else {
51        ValidatedTag::Original(t)
52    }
53}

55/// Tags the specified URL.
56///
57/// # Arguments
58///
59/// * `conn` - A database connection on which to operate.
60///
61/// * `url` - The URL to tag.
62///
63/// * `tag` - The tag to add for the URL.
64///
65/// # Returns
66///
67/// There is no success return value.
68pub fn tag_url(db: &PlacesDb, url: &Url, tag: &str) -> Result<()> {
69    let tag = validate_tag(tag).ensure_valid()?;
70    let tx = db.begin_transaction()?;
71
72    // This function will not create a new place.
73    // Fetch the place id, so we (a) avoid creating a new tag when we aren't
74    // going to reference it and (b) to avoid a sub-query.
75    let place_id = match fetch_page_info(db, url)? {
76        Some(info) => info.page.row_id,
77        None => return Err(InvalidPlaceInfo::NoSuchUrl.into()),
78    };
79
80    db.execute_cached(
81        "INSERT OR IGNORE INTO moz_tags(tag, lastModified)
82         VALUES(:tag, now())",
83        &[(":tag", &tag)],
84    )?;
85
86    db.execute_cached(
87        "INSERT OR IGNORE INTO moz_tags_relation(tag_id, place_id)
88         VALUES((SELECT id FROM moz_tags WHERE tag = :tag), :place_id)",
89        &[
90            (":tag", &tag as &dyn rusqlite::ToSql),
91            (":place_id", &place_id),
92        ],
93    )?;
94    tx.commit()?;
95    Ok(())
96}

98/// Remove the specified tag from the specified URL.
99///
100/// # Arguments
101///
102/// * `conn` - A database connection on which to operate.
103///
104/// * `url` - The URL from which the tag should be removed.
105///
106/// * `tag` - The tag to remove from the URL.
107///
108/// # Returns
109///
110/// There is no success return value - the operation is ignored if the URL
111/// does not have the tag.
112pub fn untag_url(db: &PlacesDb, url: &Url, tag: &str) -> Result<()> {
113    let tag = validate_tag(tag).ensure_valid()?;
114    db.execute_cached(
115        "DELETE FROM moz_tags_relation
116         WHERE tag_id = (SELECT id FROM moz_tags
117                         WHERE tag = :tag)
118         AND place_id = (SELECT id FROM moz_places
119                         WHERE url_hash = hash(:url)
120                         AND url = :url)",
121        &[(":tag", &tag), (":url", &url.as_str())],
122    )?;
123    Ok(())
124}

169/// Retrieves a list of URLs which have the specified tag.
170///
171/// # Arguments
172///
173/// * `conn` - A database connection on which to operate.
174///
175/// * `tag` - The tag to query.
176///
177/// # Returns
178///
179/// * A Vec<Url> with all URLs which have the tag, ordered by the frecency of
180///   the URLs.
181pub fn get_urls_with_tag(db: &PlacesDb, tag: &str) -> Result<Vec<Url>> {
182    let tag = validate_tag(tag).ensure_valid()?;
183
184    let mut stmt = db.prepare(
185        "SELECT p.url FROM moz_places p
186         JOIN moz_tags_relation r ON r.place_id = p.id
187         JOIN moz_tags t ON t.id = r.tag_id
188         WHERE t.tag = :tag
189         ORDER BY p.frecency",
190    )?;
191
192    let rows = stmt.query_and_then(&[(":tag", &tag)], |row| row.get::<_, String>("url"))?;
193    let mut urls = Vec::new();
194    for row in rows {
195        urls.push(Url::parse(&row?)?);
196    }
197    Ok(urls)
198}

200/// Retrieves a list of tags for the specified URL.
201///
202/// # Arguments
203///
204/// * `conn` - A database connection on which to operate.
205///
206/// * `url` - The URL to query.
207///
208/// # Returns
209///
210/// * A Vec<String> with all tags for the URL, sorted by the last modified
211///   date of the tag (latest to oldest)
212pub fn get_tags_for_url(db: &PlacesDb, url: &Url) -> Result<Vec<String>> {
213    let mut stmt = db.prepare(
214        "SELECT t.tag
215         FROM moz_tags t
216         JOIN moz_tags_relation r ON r.tag_id = t.id
217         JOIN moz_places h ON h.id = r.place_id
218         WHERE url_hash = hash(:url) AND url = :url
219         ORDER BY t.lastModified DESC",
220    )?;
221    let rows = stmt.query_and_then(&[(":url", &url.as_str())], |row| {
222        row.get::<_, String>("tag")
223    })?;
224    let mut tags = Vec::new();
225    for row in rows {
226        tags.push(row?);
227    }
228    Ok(tags)
229}

264    fn test_validate_tag() {
265        let v = validate_tag("foo");
266        assert_eq!(v, ValidatedTag::Original("foo"));
267        assert!(v.is_original());
268        assert_eq!(v.ensure_valid().expect("should work"), "foo");
269
270        let v = validate_tag(" foo ");
271        assert_eq!(v, ValidatedTag::Normalized("foo"));
272        assert!(!v.is_original());
273        assert_eq!(v.ensure_valid().expect("should work"), "foo");
274
275        let v = validate_tag("");
276        assert_eq!(v, ValidatedTag::Invalid(""));
277        assert!(!v.is_original());
278        assert!(v.ensure_valid().is_err());

281        assert_eq!(

283            ValidatedTag::Normalized("foo bar")

286        assert!(validate_tag(&"f".repeat(101)).ensure_valid().is_err());

302        tag_url(&conn, &url1, "common").expect("should work");

307        assert_eq!(get_foreign_count(&conn, &url2), 1);

309        assert_eq!(get_foreign_count(&conn, &url2), 2);

355            true,
356        );
357        assert_eq!(count.unwrap().unwrap(), 0);

359        // places should still exist.
