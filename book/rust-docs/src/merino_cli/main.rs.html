5extern crate libsqlite3_sys;
6
7use anyhow::Result;
8use clap::{Parser, Subcommand};
9use merino::curated_recommendations::models::request::{
10    CuratedRecommendationsConfig, CuratedRecommendationsRequest,
11};
12use merino::curated_recommendations::CuratedRecommendationsClient;
13use merino::suggest::{SuggestClient, SuggestConfig, SuggestOptions};
14use viaduct::{configure_ohttp_channel, OhttpConfig};
15
16#[derive(Debug, Parser)]
17struct Cli {
18    /// Optional base host (defaults to prod)
19    #[arg(long)]
20    base_host: Option<String>,
21
22    #[command(subcommand)]
23    command: Commands,
24}
25
26#[derive(Debug, Subcommand)]
27enum Commands {
28    /// Fetch curated recommendations
29    Recommendations {
30        /// Required user agent header
31        #[arg(long)]
32        user_agent: String,
33
34        /// JSON string of type CuratedRecommendationsRequest
35        #[clap(long)]
36        json: Option<String>,
37
38        /// Path to a JSON file containing the request
39        #[clap(long, value_name = "FILE")]
40        json_file: Option<std::path::PathBuf>,
41    },
42    /// Fetch suggestions
43    Suggest {
44        /// OHTTP relay URL
45        #[arg(long, default_value = "https://ohttp-merino.mozilla.fastly-edge.com")]
46        relay_url: String,
47
48        /// OHTTP gateway host
49        #[arg(long, default_value = "ohttp-gateway-merino.services.mozilla.com")]
50        gateway_host: String,
51
52        /// Search query
53        #[arg(long)]
54        query: String,
55
56        /// List of providers (e.g. --providers wikipedia --providers adm)
57        #[arg(long)]
58        providers: Option<Vec<String>>,
59
60        /// Source identifier sent to Merino (e.g urlbar, new tab. defaults to unknown)
61        #[arg(long)]
62        source: Option<String>,
63
64        /// Country code (e.g. "US")
65        #[arg(long)]
66        country: Option<String>,
67
68        /// Region code (e.g. "CA")
69        #[arg(long)]
70        region: Option<String>,
71
72        /// City name
73        #[arg(long)]
74        city: Option<String>,
75
76        /// Client variants (e.g. --client-variants control --client-variants treatment)
77        #[arg(long)]
78        client_variants: Option<Vec<String>>,
79
80        /// Request type (e.g location | weather)
81        #[arg(long)]
82        request_type: Option<String>,
83
84        /// Accept-Language header value (e.g. "en-US")
85        #[arg(long)]
86        accept_language: Option<String>,
87    },
88}
89
90fn main() -> Result<()> {
91    viaduct_hyper::viaduct_init_backend_hyper()?;
92
93    let cli = Cli::parse();
94
95    match cli.command {
96        Commands::Recommendations {
97            user_agent,
98            json,
99            json_file,
100        } => {
101            let config = CuratedRecommendationsConfig {
102                base_host: cli.base_host,
103                user_agent_header: user_agent,
104            };
105            let client = CuratedRecommendationsClient::new(config)?;
106            let json_data = match (json_file, json) {
107                (Some(path), _) => std::fs::read_to_string(path)?,
108                (None, Some(raw)) => raw,
109                (None, None) => anyhow::bail!("You must provide either --json or --json-file"),
110            };
111            let request: CuratedRecommendationsRequest = serde_json::from_str(&json_data)?;
112            let response = client.get_curated_recommendations(&request)?;
113            println!("{}", serde_json::to_string_pretty(&response)?);
114        }
115        Commands::Suggest {
116            relay_url,
117            gateway_host,
118            query,
119            providers,
120            source,
121            country,
122            region,
123            city,
124            client_variants,
125            request_type,
126            accept_language,
127        } => {
128            configure_ohttp_channel(
129                "merino".to_string(),
130                OhttpConfig {
131                    relay_url,
132                    gateway_host,
133                },
134            )?;
135            let config = SuggestConfig {
136                base_host: cli.base_host,
137            };
138            let client = SuggestClient::new(config)?;
139            let options = SuggestOptions {
140                providers,
141                source,
142                country,
143                region,
144                city,
145                client_variants,
146                request_type,
147                accept_language,
148            };
149            match client.get_suggestions(query, options)? {
150                Some(response) => {
151                    let json: serde_json::Value = serde_json::from_str(&response)?;
152                    println!("{}", serde_json::to_string_pretty(&json)?);
153                }
154                None => println!("No suggestions available (204 No Content)"),
155            }
156        }
157    }
158
159    Ok(())
160}
