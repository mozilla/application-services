5//! Remote Settings integration for Firefox Relay
6//!
7//! This module handles fetching allowlist and denylist data from Remote Settings
8//! to determine which sites should show Relay email mask suggestions.
9//!
10//! ## Public Suffix List (PSL) Usage
11//!
12//! Mobile clients should use PSL libraries to extract domains before calling this API:
13//! - **URL**: `https://mail.google.com/inbox` (full URL)
14//! - **Host**: `mail.google.com` (hostname from URL)
15//! - **Domain**: `google.com` (registrable domain via PSL, also known as eTLD+1)
16//!
17//! The PSL ensures proper domain matching:
18//! - `mail.google.com` → domain `google.com` ✓
19//! - `evil-google.com` → domain `evil-google.com` ✗ (won't match rules for google.com)
20//! - `mysite.github.io` → domain `mysite.github.io` ✓ (github.io is in PSL)
21//!
22//! ## Caching Behavior
23//!
24//! Remote Settings automatically caches collection data in SQLite:
25//! - **First call**: Fetches from server and caches (network request)
26//! - **Subsequent calls**: Reads from cache (fast, no network)
27//! - **Updates**: Mobile apps should call `RemoteSettingsService.sync()` periodically
28//!   to refresh all collections, or individual clients will sync on first access
29//!
30//! This makes the component self-sufficient - it will automatically fetch data when needed
31//! and cache it for fast subsequent access.
32

37/// Remote Settings collection names for Relay
38#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
39pub enum Collection {
40    Allowlist,
41    Denylist,
42}

44impl Collection {
45    pub fn name(&self) -> &'static str {
46        match self {
47            Self::Allowlist => "fxrelay-allowlist",
48            Self::Denylist => "fxrelay-denylist",
49        }
50    }
51}

53/// Client for fetching Relay data from Remote Settings

58pub struct RelayRemoteSettingsClient {

74    /// Determines whether Firefox Relay should be offered for a given site.
75    ///
76    /// This function implements the Relay visibility decision logic based on three factors:
77    /// 1. Whether the site is on the denylist
78    /// 2. Whether the user is an existing Relay user
79    /// 3. Whether the site is on the allowlist
80    ///
81    /// # Decision Logic (in order)
82    ///
83    /// 1. If the site is on the **denylist**, don't show Relay (regardless of user status)
84    /// 2. If the site is NOT on the denylist and the user **is** a Relay user, show Relay
85    /// 3. If the site is NOT on the denylist and the user is **not** a Relay user, check the allowlist:
86    ///    - If the site is on the **allowlist**, show Relay (to promote Relay to new users)
87    ///    - Otherwise, don't show Relay
88    ///
89    /// # Fail-Safe Behavior
90    ///
91    /// If Remote Settings is unavailable:
92    /// - If denylist fetch fails: don't show Relay (conservative: avoid showing on potentially blocked sites)
93    /// - If allowlist fetch fails for non-Relay users: assume empty (conservative: don't promote without data)
94    ///
95    /// # Arguments
96    /// * `host` - The full hostname (e.g., "mail.google.com", "www.example.com")
97    /// * `domain` - The registrable domain from PSL (e.g., "google.com", "example.co.uk")
98    /// * `is_relay_user` - Whether the user already has a Relay account
99    ///
100    /// # Public Suffix List (PSL)
101    ///
102    /// Mobile clients should use PSL to extract the domain:
103    /// - **Host**: Full hostname like "mail.google.com"
104    /// - **Domain**: Registrable domain (eTLD+1) like "google.com"
105    /// - Swift: Use `URL.host` and extract domain via PSL library
106    /// - Kotlin: Use `URL.host` and extract domain via PSL library
107    ///
108    /// # Returns
109    /// `true` if Relay should be shown for this site, `false` otherwise
110    ///
111    /// # Determining if User is a Relay User
112    ///
113    /// Mobile clients should check if the user is a Relay user by calling
114    /// `FirefoxAccount.getAttachedClients()` and checking if Relay is in the
115    /// list of attached clients.
116    ///
117    /// # Example
118    /// ```kotlin
119    /// val url = URL("https://mail.google.com/inbox")
120    /// val host = url.host // "mail.google.com"
121    /// val domain = PublicSuffixList.getRegistrableDomain(host) // "google.com"
122    ///
123    /// val rsClient = RelayRemoteSettingsClient(remoteSettingsService)
124    /// val attachedClients = fxAccount.getAttachedClients()
125    /// val isRelayUser = attachedClients.any { it.clientId == RELAY_CLIENT_ID }
126    ///
127    /// if (rsClient.shouldShowRelay(host, domain, isRelayUser)) {
128    ///     // Show Relay UI
129    /// }
130    /// ```
131    pub fn should_show_relay(&self, host: String, domain: String, is_relay_user: bool) -> bool {
132        // Validate inputs
133        if host.is_empty() || domain.is_empty() {
134            log::debug!("Empty host or domain, not showing Relay");
135            return false;
136        }
137
138        // Log the site and user status we're checking
139        log::debug!(
140            "Checking if Relay should be shown for host: {}, domain: {}, is_relay_user: {}",
141            host,
142            domain,
143            is_relay_user
144        );
145
146        // Fetch denylist with error handling
147        let denylist = match self.get_records(Collection::Denylist) {
148            Ok(list) => list,
149            Err(e) => {
150                log::warn!("Failed to fetch denylist, failing conservatively: {}", e);
151                // Fail-safe: if we can't fetch denylist, don't show Relay
152                // (conservative default - better to not show Relay than risk showing on blocked sites)
153                return false;
154            }
155        };
156
157        // Step 1: Check denylist - if site is denied, never show Relay
158        if !denylist.is_empty() && self.is_site_in_list(&denylist, &host, &domain) {
159            log::debug!(
160                "Site {} ({}) is in denylist, not showing Relay",
161                host,
162                domain
163            );
164            return false;
165        }
166
167        // Step 2: If site is not on denylist and user is already a Relay user, show Relay
168        if is_relay_user {
169            log::debug!(
170                "Site {} ({}) is not in denylist and user is a Relay user, showing Relay",
171                host,
172                domain
173            );
174            return true;
175        }
176
177        // Step 3: User is not a Relay user, check allowlist to determine if we should promote Relay
178        let allowlist = match self.get_records(Collection::Allowlist) {
179            Ok(list) => list,
180            Err(e) => {
181                log::warn!("Failed to fetch allowlist, assuming empty: {}", e);
182                // Fail-safe: if we can't fetch allowlist, don't promote to new users
183                // (conservative default - don't show Relay promotion without data)
184                vec![]
185            }
186        };
187
188        // If allowlist is empty, don't show Relay promotion
189        if allowlist.is_empty() {
190            log::debug!(
191                "Allowlist is empty, not showing Relay promotion for non-Relay user on site {} ({})",
192                host,
193                domain
194            );
195            return false;
196        }
197
198        // Check if site is in allowlist
199        let in_allowlist = self.is_site_in_list(&allowlist, &host, &domain);
200        if in_allowlist {
201            log::debug!(
202                "Site {} ({}) is in allowlist, showing Relay promotion to non-Relay user",
203                host,
204                domain
205            );
206        } else {
207            log::debug!(
208                "Site {} ({}) is not in allowlist, not showing Relay promotion to non-Relay user",
209                host,
210                domain
211            );
212        }
213        in_allowlist
214    }

217impl RelayRemoteSettingsClient {
218    /// Gets the Remote Settings client for a specific collection
219    fn client_for_collection(&self, collection: Collection) -> &RemoteSettingsClient {
220        match collection {
221            Collection::Allowlist => &self.allowlist_client,
222            Collection::Denylist => &self.denylist_client,
223        }
224    }
225
226    /// Checks if a site matches an entry in the list using PSL-aware matching.
227    ///
228    /// Matches on exact domain, exact host, or subdomain relationships.
229    fn is_site_in_list(&self, list: &[String], host: &str, domain: &str) -> bool {
230        for entry in list {
231            // Exact domain or host match
232            if entry == domain || entry == host {
233                return true;
234            }
235
236            // Entry is a subdomain of our domain
237            if entry.ends_with(domain) && entry.len() > domain.len() {
238                let prefix_end = entry.len() - domain.len();
239                if entry.as_bytes().get(prefix_end - 1) == Some(&b'.') {
240                    return true;
241                }
242            }
243
244            // Our host is a subdomain of the entry
245            if host.ends_with(entry) && host.len() > entry.len() {
246                let prefix_end = host.len() - entry.len();
247                if host.as_bytes().get(prefix_end - 1) == Some(&b'.') {
248                    return true;
249                }
250            }
251        }
252
253        false
254    }
255
256    /// Fetches records from a specific Remote Settings collection
257    fn get_records(&self, collection: Collection) -> Result<Vec<String>> {
258        let client = self.client_for_collection(collection);
259
260        // Get records from Remote Settings
261        // Use sync_if_empty=true so the first call will fetch from server if cache is empty,
262        // and subsequent calls will use the cached data (fast, no network)
263        let records = match client.get_records(true) {
264            Some(records) => records,
265            None => {
266                log::debug!("No records found for collection: {}", collection.name());
267                return Ok(vec![]);
268            }
269        };
270
271        // If records list is empty, return empty vec (this is OK - not an error)
272        if records.is_empty() {
273            log::debug!("Empty records list for collection: {}", collection.name());
274            return Ok(vec![]);
275        }
276
277        // Parse the records from Remote Settings
278        // Remote Settings records have a `fields` property with a "domain" field
279        let domains: Vec<String> = records
280            .iter()
281            .filter_map(|record| match record.fields.get("domain") {
282                Some(value) => value.as_str().map(String::from),
283                None => {
284                    log::warn!(
285                        "Record missing 'domain' field in {}: {:?}",
286                        collection.name(),
287                        record.id
288                    );
289                    None
290                }
291            })
292            .collect();
293
294        Ok(domains)
295    }
