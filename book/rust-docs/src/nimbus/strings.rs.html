10#[allow(dead_code)]
11pub fn fmt<T: serde::Serialize>(template: &str, context: &T) -> Result<String> {
12    let obj: Value = match serde_json::to_value(context) {
13        Ok(v) => v,
14        Err(e) => {
15            return Err(NimbusError::JSONError(
16                "obj = nimbus::strings::fmt::serde_json::to_value".into(),
17                e.to_string(),
18            ));
19        }
20    };
21
22    fmt_with_value(template, &obj)
23}

25#[allow(dead_code)]
26pub fn fmt_with_value(template: &str, value: &Value) -> Result<String> {
27    if let Value::Object(map) = value {
28        Ok(fmt_with_map(template, map))
29    } else {
30        Err(NimbusError::EvaluationError(
31            "Can only format json objects".to_string(),
32        ))
33    }
34}

36pub fn fmt_with_map(input: &str, context: &Map<String, Value>) -> String {
37    let mut output = String::with_capacity(input.len());
38
39    let mut iter = iter_graphemes(input);
40    let mut last_index = 0;
41
42    // This is exceedingly simple; never refer to this as a parser.
43    while let Some((index, c)) = iter.next() {
44        if c == "{" {
45            let open_index = index;
46            for (index, c) in iter.by_ref() {
47                if c == "}" {
48                    let close_index = index;
49                    let field_name = &input[open_index + 1..close_index];
50
51                    // If we decided to embed JEXL into this templating language,
52                    // this would be the place to put it.
53                    // However, we'd likely want to make this be able to detect balanced braces,
54                    // which this does not.
55                    let replace_string = match context.get(field_name) {
56                        Some(Value::Bool(v)) => v.to_string(),
57                        Some(Value::String(v)) => v.to_string(),
58                        Some(Value::Number(v)) => v.to_string(),
59                        _ => format!("{{{v}}}", v = field_name),
60                    };
61
62                    output.push_str(&input[last_index..open_index]);
63                    output.push_str(&replace_string);
64
65                    // +1 skips the closing }
66                    last_index = close_index + 1;
67                    break;
68                }
69            }
70        }
71    }
72
73    output.push_str(&input[last_index..input.len()]);
74
75    output
76}

78/// Iterate over graphemes
79///
80/// Each iteration will yield the start index of the grapheme and it's content.
81///
82/// This is intended to be equivalent to `grapheme_indices(true)` from the `unicode_segmentation`
83/// crate.
84fn iter_graphemes(input: &str) -> impl Iterator<Item = (usize, &str)> {
85    let mut last_idx = None;
86    let segmenter = GraphemeClusterSegmenter::new();
87    segmenter
88        .segment_str(input)
89        .filter_map(move |idx| {
90            let next = last_idx.map(|last_idx| (last_idx, idx));
91            last_idx = Some(idx);
92            next
93        })
94        .map(|(last_idx, idx)| (last_idx, &input[last_idx..idx]))
95}

97#[cfg(test)]
98mod unit_tests {
99    use serde_json::json;
100
101    use super::*;
102
103    #[test]
104    fn smoke_tests() {
105        let c = json!({
106            "string": "STRING".to_string(),
107            "number": 42,
108            "boolean": true,
109        });
110        let c = c.as_object().unwrap();
111
112        assert_eq!(
113            fmt_with_map("A {string}, a {number}, a {boolean}.", c),
114            "A STRING, a 42, a true.".to_string()
115        );
116    }
117
118    #[test]
119    fn test_unicode_boundaries() {
120        let c = json!({
121            "empty": "".to_string(),
122            "unicode": "a̐éö̲".to_string(),
123            "a̐éö̲": "unicode".to_string(),
124        });
125        let c = c.as_object().unwrap();
126
127        assert_eq!(fmt_with_map("fîré{empty}ƒøüX", c), "fîréƒøüX".to_string());
128        assert_eq!(fmt_with_map("a̐éö̲{unicode}a̐éö̲", c), "a̐éö̲a̐éö̲a̐éö̲".to_string());
129        assert_eq!(
130            fmt_with_map("is this {a̐éö̲}?", c),
131            "is this unicode?".to_string()
132        );
133    }
134
135    #[test]
136    fn test_pathological_cases() {
137        let c = json!({
138            "empty": "".to_string(),
139            "foo}́": "foo-with-accented-brace",
140        });
141        let c = c.as_object().unwrap();
142
143        assert_eq!(
144            fmt_with_map("A {notthere}.", c),
145            "A {notthere}.".to_string()
146        );
147        assert_eq!(
148            fmt_with_map("aa { unclosed", c),
149            "aa { unclosed".to_string()
150        );
151        // }́ shouldn't close the name since we only match on graphemes
152        assert_eq!(
153            fmt_with_map("aa {foo}́}", c),
154            "aa foo-with-accented-brace".to_string()
155        );
156    }
157}
