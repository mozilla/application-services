9#[allow(dead_code)]
10pub fn fmt<T: serde::Serialize>(template: &str, context: &T) -> Result<String> {
11    let obj: Value = match serde_json::to_value(context) {
12        Ok(v) => v,
13        Err(e) => {
14            return Err(NimbusError::JSONError(
15                "obj = nimbus::strings::fmt::serde_json::to_value".into(),
16                e.to_string(),
17            ))
18        }
19    };
20
21    fmt_with_value(template, &obj)
22}

24#[allow(dead_code)]
25pub fn fmt_with_value(template: &str, value: &Value) -> Result<String> {
26    if let Value::Object(map) = value {
27        Ok(fmt_with_map(template, map))
28    } else {
29        Err(NimbusError::EvaluationError(
30            "Can only format json objects".to_string(),
31        ))
32    }
33}

35pub fn fmt_with_map(input: &str, context: &Map<String, Value>) -> String {
36    let mut output = String::with_capacity(input.len());
37
38    let mut iter = iter_graphemes(input);
39    let mut last_index = 0;
40
41    // This is exceedingly simple; never refer to this as a parser.
42    while let Some((index, c)) = iter.next() {
43        if c == "{" {
44            let open_index = index;
45            for (index, c) in iter.by_ref() {
46                if c == "}" {
47                    let close_index = index;
48                    let field_name = &input[open_index + 1..close_index];
49
50                    // If we decided to embed JEXL into this templating language,
51                    // this would be the place to put it.
52                    // However, we'd likely want to make this be able to detect balanced braces,
53                    // which this does not.
54                    let replace_string = match context.get(field_name) {
55                        Some(Value::Bool(v)) => v.to_string(),
56                        Some(Value::String(v)) => v.to_string(),
57                        Some(Value::Number(v)) => v.to_string(),
58                        _ => format!("{{{v}}}", v = field_name),
59                    };
60
61                    output.push_str(&input[last_index..open_index]);
62                    output.push_str(&replace_string);
63
64                    // +1 skips the closing }
65                    last_index = close_index + 1;
66                    break;
67                }
68            }
69        }
70    }
71
72    output.push_str(&input[last_index..input.len()]);
73
74    output
75}

77/// Iterate over graphemes
78///
79/// Each iteration will yield the start index of the grapheme and it's content.
80///
81/// This is intended to be equivalent to `grapheme_indices(true)` from the `unicode_segmentation`
82/// crate.
83fn iter_graphemes(input: &str) -> impl Iterator<Item = (usize, &str)> {
84    let mut last_idx = None;
85    let segmenter = GraphemeClusterSegmenter::new();
86    segmenter
87        .segment_str(input)
88        .filter_map(move |idx| {
89            let next = last_idx.map(|last_idx| (last_idx, idx));
90            last_idx = Some(idx);
91            next
92        })
93        .map(|(last_idx, idx)| (last_idx, &input[last_idx..idx]))
94}

96#[cfg(test)]
97mod unit_tests {
98    use serde_json::json;
99
100    use super::*;
101
102    #[test]
103    fn smoke_tests() {
104        let c = json!({
105            "string": "STRING".to_string(),
106            "number": 42,
107            "boolean": true,
108        });
109        let c = c.as_object().unwrap();
110
111        assert_eq!(
112            fmt_with_map("A {string}, a {number}, a {boolean}.", c),
113            "A STRING, a 42, a true.".to_string()
114        );
115    }
116
117    #[test]
118    fn test_unicode_boundaries() {
119        let c = json!({
120            "empty": "".to_string(),
121            "unicode": "a̐éö̲".to_string(),
122            "a̐éö̲": "unicode".to_string(),
123        });
124        let c = c.as_object().unwrap();
125
126        assert_eq!(fmt_with_map("fîré{empty}ƒøüX", c), "fîréƒøüX".to_string());
127        assert_eq!(fmt_with_map("a̐éö̲{unicode}a̐éö̲", c), "a̐éö̲a̐éö̲a̐éö̲".to_string());
128        assert_eq!(
129            fmt_with_map("is this {a̐éö̲}?", c),
130            "is this unicode?".to_string()
131        );
132    }
133
134    #[test]
135    fn test_pathological_cases() {
136        let c = json!({
137            "empty": "".to_string(),
138            "foo}́": "foo-with-accented-brace",
139        });
140        let c = c.as_object().unwrap();
141
142        assert_eq!(
143            fmt_with_map("A {notthere}.", c),
144            "A {notthere}.".to_string()
145        );
146        assert_eq!(
147            fmt_with_map("aa { unclosed", c),
148            "aa { unclosed".to_string()
149        );
150        // }́ shouldn't close the name since we only match on graphemes
151        assert_eq!(
152            fmt_with_map("aa {foo}́}", c),
153            "aa foo-with-accented-brace".to_string()
154        );
155    }
156}
