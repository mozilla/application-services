5//! This is a simple HTTP client that uses viaduct to retrieve experiment data from the server.
6//! Currently configured to use Kinto and the old schema, although that would change once we start
7//! working on the real Nimbus schema.
8//!
9//! In the future we might replace this with a more fully-feature Remote Settings client, such as:
10//!
11//!   https://github.com/mozilla-services/remote-settings-client
12//!   Issue: https://github.com/mozilla/application-services/issues/3475
13//!
14//! But the simple subset implemented here meets our needs for now.
15
16use std::sync::Arc;
17
18use remote_settings::{RemoteSettingsClient, RemoteSettingsError};
19use serde_json::json;
20
21use crate::NimbusError;
22use crate::error::Result;
23use crate::schema::parse_experiments;
24use crate::stateful::client::{Experiment, SettingsClient};
25
26impl SettingsClient for Arc<RemoteSettingsClient> {
27    fn get_experiments_metadata(&self) -> Result<String> {
28        unimplemented!();
29    }
30
31    fn fetch_experiments(&self) -> Result<Vec<Experiment>> {
32        self.sync()?;
33
34        let records = self.get_records(false).ok_or(RemoteSettingsError::Other {
35            reason: "Unable to fetch experiment records".to_owned(),
36        })?;
37        let wrapped_data = json!({ "data": records });
38        let resp = serde_json::to_string(&wrapped_data).map_err(|e| {
39            NimbusError::JSONError(
40                "SettingsClient::fetch_experiments resp = serde_json::to_string".to_owned(),
41                e.to_string(),
42            )
43        })?;
44        parse_experiments(&resp)
45    }
46}
