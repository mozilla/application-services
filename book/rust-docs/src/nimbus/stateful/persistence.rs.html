101/// Enumeration of the different stores within our database.
102///
103/// Our rkv database contains a number of different "stores", and the items
104/// in each store correspond to a particular type of object at the Rust level.
105pub enum StoreId {
106    /// Store containing the set of known experiments, as read from the server.
107    ///
108    /// Keys in the `Experiments` store are experiment identifier slugs, and their
109    /// corresponding values are  serialized instances of the [`Experiment`] struct
110    /// representing the last known state of that experiment.
111    Experiments,
112    /// Store containing the set of known experiment enrollments.
113    ///
114    /// Keys in the `Enrollments` store are experiment identifier slugs, and their
115    /// corresponding values are serialized instances of the [`ExperimentEnrollment`]
116    /// struct representing the current state of this client's enrollment (or not)
117    /// in that experiment.
118    Enrollments,
119    /// Store containing miscellaneous metadata about this client instance.
120    ///
121    /// Keys in the `Meta` store are string constants, and their corresponding values
122    /// are serialized items whose type depends on the constant. Known constraints
123    /// include:
124    ///   * "db_version":   u16, the version number of the most revent migration
125    ///     applied to this database.
126    ///   * "nimbus-id":    String, the randomly-generated identifier for the
127    ///     current client instance.
128    ///   * "user-opt-in-experiments":  bool, whether the user has explicitly opted in or out
129    ///     of participating in experiments.
130    ///   * "user-opt-in-rollouts":  bool, whether the user has explicitly opted in or out
131    ///     of participating in rollouts.
132    ///   * "installation-date": a UTC DateTime string, defining the date the consuming app was
133    ///     installed
134    ///   * "update-date": a UTC DateTime string, defining the date the consuming app was
135    ///     last updated
136    ///   * "app-version": String, the version of the app last persisted
137    Meta,
138    /// Store containing pending updates to experiment data.
139    ///
140    /// The `Updates` store contains a single key "pending-experiment-updates", whose
141    /// corresponding value is a serialized `Vec<Experiment>` of new experiment data
142    /// that has been received from the server but not yet processed by the application.
143    Updates,
144    /// Store containing collected counts of behavior events for targeting purposes.
145    ///
146    /// Keys in the `EventCounts` store are strings representing the identifier for
147    /// the event and their corresponding values represent a serialized instance of a
148    /// [`MultiIntervalCounter`] struct that contains a set of configurations and data
149    /// for the different time periods that the data will be aggregated on.
150    EventCounts,
151}

164    pub fn put<T: serde::Serialize + for<'de> serde::Deserialize<'de>>(
165        &self,
166        writer: &mut Writer,
167        key: &str,
168        persisted_data: &T,
169    ) -> Result<()> {
170        let persisted_json = match serde_json::to_string(persisted_data) {
171            Ok(v) => v,
172            Err(e) => return Err(NimbusError::JSONError("persisted_json = nimbus::stateful::persistence::SingleStore::put::serde_json::to_string".into(), e.to_string()))
173        };
174        self.store
175            .put(writer, key, &rkv::Value::Json(&persisted_json))?;
176        Ok(())
177    }

190    // Some "get" functions that cooperate with transactions (ie, so we can
191    // get what we've written to the transaction before it's committed).
192    // It's unfortunate that these are duplicated with the DB itself, but the
193    // traits used by rkv make this tricky.
194    pub fn get<'r, T, R>(&self, reader: &'r R, key: &str) -> Result<Option<T>>
195    where
196        R: Readable<'r>,
197        T: serde::Serialize + for<'de> serde::Deserialize<'de>,
198    {
199        let persisted_data = self.store.get(reader, key)?;
200        match persisted_data {
201            Some(data) => {
202                if let rkv::Value::Json(data) = data {
203                    Ok(Some(match serde_json::from_str::<T>(data) {
204                        Ok(v) => v,
205                        Err(e) => return Err(NimbusError::JSONError("match persisted_data nimbus::stateful::persistence::SingleStore::get::serde_json::from_str".into(), e.to_string()))
206                    }))
207                } else {
208                    Err(NimbusError::InvalidPersistedData)
209                }
210            }
211            None => Ok(None),
212        }
213    }

215    /// Fork of collect_all that simply drops records that fail to read
216    /// rather than simply returning an error up the stack.  This likely
217    /// wants to be just a parameter to collect_all, but for now....
218    pub fn try_collect_all<'r, T, R>(&self, reader: &'r R) -> Result<Vec<T>>
219    where
220        R: Readable<'r>,
221        T: serde::Serialize + for<'de> serde::Deserialize<'de>,
222    {
223        let mut result = Vec::new();
224        let mut iter = self.store.iter_start(reader)?;
225        while let Some(Ok((_, data))) = iter.next() {
226            if let rkv::Value::Json(data) = data {
227                let unserialized = serde_json::from_str::<T>(data);
228                match unserialized {
229                    Ok(value) => result.push(value),
230                    Err(e) => {
231                        // If there is an error, we won't push this onto the
232                        // result Vec, but we won't blow up the entire
233                        // deserialization either.
234                        warn!(
235                            "try_collect_all: discarded a record while deserializing with: {:?}",
236                            e
237                        );
238                        warn!(
239                            "try_collect_all:   data that failed to deserialize: {:?}",
240                            data
241                        );
242                    }
243                };
244            }
245        }
246        Ok(result)
247    }
248
249    pub fn collect_all<'r, T, R>(&self, reader: &'r R) -> Result<Vec<T>>
250    where
251        R: Readable<'r>,
252        T: serde::Serialize + for<'de> serde::Deserialize<'de>,
253    {
254        let mut result = Vec::new();
255        let mut iter = self.store.iter_start(reader)?;
256        while let Some(Ok((_, data))) = iter.next() {
257            if let rkv::Value::Json(data) = data {
258                result.push(match serde_json::from_str::<T>(data) {
259                    Ok(v) => v,
260                    Err(e) => return Err(NimbusError::JSONError("rkv::Value::Json(data) nimbus::stateful::persistence::SingleStore::collect_all::serde_json::from_str".into(), e.to_string()))
261                });
262            }
263        }
264        Ok(result)
265    }
266}

345    fn maybe_upgrade(&self) -> Result<()> {
346        debug!("entered maybe upgrade");
347        let mut writer = self.rkv.write()?;
348        let current_version = self
349            .meta_store
350            .get::<u16, _>(&writer, DB_KEY_DB_VERSION)?
351            .unwrap_or(0);
352
353        if current_version == DB_VERSION {
354            info!("Already at version {}, no upgrade needed", DB_VERSION);
355            return Ok(());
356        }
357
358        if current_version == 0 || current_version > DB_VERSION {
359            info!(
360                "maybe_upgrade: current_version: {}, DB_VERSION: {}; wiping most stores",
361                current_version, DB_VERSION
362            );
363            self.clear_experiments_and_enrollments(&mut writer)?;
364            self.updates_store.clear(&mut writer)?;
365            self.meta_store
366                .put(&mut writer, DB_KEY_DB_VERSION, &DB_VERSION)?;
367            writer.commit()?;
368            return Ok(());
369        }
370
371        if current_version == 1 {
372            info!("Migrating database from v1 to v2");
373            if let Err(e) = self.migrate_v1_to_v2(&mut writer) {
374                error_support::report_error!(
375                    "nimbus-database-migration",
376                    "Error migrating database v1 to v2: {:?}. Wiping experiments and enrollments",
377                    e
378                );
379                self.clear_experiments_and_enrollments(&mut writer)?;
380            }
381        }
382
383        if current_version == 2 {
384            info!("Migrating database from v2 to v3");
385            if let Err(e) = self.migrate_v2_to_v3(&mut writer) {
386                error_support::report_error!(
387                    "nimbus-database-migration",
388                    "Error migrating database v2 to v3: {:?}. Wiping experiments and enrollments",
389                    e
390                );
391                self.clear_experiments_and_enrollments(&mut writer)?;
392            }
393        }
394
395        // It is safe to clear the update store (i.e. the pending experiments) on all schema upgrades
396        // as it will be re-filled from the server on the next `fetch_experiments()`.
397        // The current contents of the update store may cause experiments to not load, or worse,
398        // accidentally unenroll.
399        self.updates_store.clear(&mut writer)?;
400        self.meta_store
401            .put(&mut writer, DB_KEY_DB_VERSION, &DB_VERSION)?;
402        writer.commit()?;
403        debug!("maybe_upgrade: transaction committed");
404        Ok(())
405    }

421    /// Migrates a v1 database to v2
422    ///
423    /// Note that any Err returns from this function (including stuff
424    /// propagated up via the ? operator) will cause maybe_update (our caller)
425    /// to assume that this is unrecoverable and wipe the database, removing
426    /// people from any existing enrollments and blowing away their experiment
427    /// history, so that they don't get left in an inconsistent state.
428    fn migrate_v1_to_v2(&self, writer: &mut Writer) -> Result<()> {
429        info!("Upgrading from version 1 to version 2");
430
431        // use try_collect_all to read everything except records that serde
432        // returns deserialization errors on.  Some logging of those errors
433        // happens, but it's not ideal.
434        let reader = self.read()?;
435
436        // XXX write a test to verify that we don't need to gc any
437        // enrollments that don't have experiments because the experiments
438        // were discarded either during try_collect_all (these wouldn't have been
439        // detected during the filtering phase) or during the filtering phase
440        // itself.  The test needs to run evolve_experiments, as that should
441        // correctly drop any orphans, even if the migrators aren't perfect.
442
443        let enrollments: Vec<ExperimentEnrollment> =
444            self.enrollment_store.try_collect_all(&reader)?;
445        let experiments: Vec<Experiment> = self.experiment_store.try_collect_all(&reader)?;
446
447        // figure out which experiments have records that need to be dropped
448        // and log that we're going to drop them and why
449        let empty_string = "".to_string();
450        let slugs_with_experiment_issues: HashSet<String> = experiments
451            .iter()
452            .filter_map(
453                    |e| {
454                let branch_with_empty_feature_ids =
455                    e.branches.iter().find(|b| b.feature.is_none() || b.feature.as_ref().unwrap().feature_id.is_empty());
456                if branch_with_empty_feature_ids.is_some() {
457                    warn!("{:?} experiment has branch missing a feature prop; experiment & enrollment will be discarded", &e.slug);
458                    Some(e.slug.to_owned())
459                } else if e.feature_ids.is_empty() || e.feature_ids.contains(&empty_string) {
460                    warn!("{:?} experiment has invalid feature_ids array; experiment & enrollment will be discarded", &e.slug);
461                    Some(e.slug.to_owned())
462                } else {
463                    None
464                }
465            })
466            .collect();
467        let slugs_to_discard: HashSet<_> = slugs_with_experiment_issues;
468
469        // filter out experiments to be dropped
470        let updated_experiments: Vec<Experiment> = experiments
471            .into_iter()
472            .filter(|e| !slugs_to_discard.contains(&e.slug))
473            .collect();
474        debug!("updated experiments = {:?}", updated_experiments);
475
476        // filter out enrollments to be dropped
477        let updated_enrollments: Vec<ExperimentEnrollment> = enrollments
478            .into_iter()
479            .filter(|e| !slugs_to_discard.contains(&e.slug))
480            .collect();
481        debug!("updated enrollments = {:?}", updated_enrollments);
482
483        // rewrite both stores
484        self.experiment_store.clear(writer)?;
485        for experiment in updated_experiments {
486            self.experiment_store
487                .put(writer, &experiment.slug, &experiment)?;
488        }
489
490        self.enrollment_store.clear(writer)?;
491        for enrollment in updated_enrollments {
492            self.enrollment_store
493                .put(writer, &enrollment.slug, &enrollment)?;
494        }
495        debug!("exiting migrate_v1_to_v2");
496
497        Ok(())
498    }

500    /// Migrates a v2 database to v3
501    ///
502    /// Separates global user participation into experiments and rollouts participation.
503    /// For privacy: if user opted out globally, they remain opted out of experiments.
504    fn migrate_v2_to_v3(&self, writer: &mut Writer) -> Result<()> {
505        info!("Upgrading from version 2 to version 3");
506
507        let meta_store = &self.meta_store;
508
509        // Get the old global participation flag
510        let old_global_participation = meta_store
511            .get::<bool, _>(writer, DB_KEY_GLOBAL_USER_PARTICIPATION)?
512            .unwrap_or(true); // Default was true
513
514        // Set new separate flags based on privacy requirements:
515        // - If user opted out globally, they stay opted out of experiments
516        // - If user opted out globally, they stay opted out of rollouts (per requirement #3)
517        meta_store.put(
518            writer,
519            DB_KEY_EXPERIMENT_PARTICIPATION,
520            &old_global_participation,
521        )?;
522        meta_store.put(
523            writer,
524            DB_KEY_ROLLOUT_PARTICIPATION,
525            &old_global_participation,
526        )?;
527
528        // Remove the old global participation key if it exists
529        if meta_store
530            .get::<bool, _>(writer, DB_KEY_GLOBAL_USER_PARTICIPATION)?
531            .is_some()
532        {
533            meta_store.delete(writer, DB_KEY_GLOBAL_USER_PARTICIPATION)?;
534        }
535
536        info!(
537            "Migration v2->v3: experiments_participation={}, rollouts_participation={}",
538            old_global_participation, old_global_participation
539        );
540
541        Ok(())
542    }

556    pub fn open_rkv<P: AsRef<Path>>(path: P) -> Result<Rkv> {
557        let path = std::path::Path::new(path.as_ref()).join("db");
558        debug!("open_rkv: path =  {:?}", path.display());
559        fs::create_dir_all(&path)?;
560        let rkv = match rkv_new(&path) {
561            Ok(rkv) => Ok(rkv),
562            Err(rkv_error) => {
563                match rkv_error {
564                    // For some errors we just delete the DB and start again.
565                    StoreError::DatabaseCorrupted | StoreError::FileInvalid => {
566                        // On one hand this seems a little dangerous, but on
567                        // the other hand avoids us knowing about the
568                        // underlying implementation (ie, how do we know what
569                        // files might exist in all cases?)
570                        warn!(
571                            "Database at '{}' appears corrupt - removing and recreating",
572                            path.display()
573                        );
574                        fs::remove_dir_all(&path)?;
575                        fs::create_dir_all(&path)?;
576                        // TODO: Once we have glean integration we want to
577                        // record telemetry here.
578                        rkv_new(&path)
579                    }
580                    // All other errors are fatal.
581                    _ => Err(rkv_error),
582                }
583            }
584        }?;
585        debug!("Database initialized");
586        Ok(rkv)
587    }

596    /// writer.
597    pub fn write(&self) -> Result<Writer<'_>> {
598        Ok(self.rkv.write()?)
599    }

601    /// Function used to retrieve persisted data outside of a transaction.
602    /// It allows retrieval of any serializable and deserializable data
603    /// Currently only supports JSON data
604    // Only available for tests; product code should always be using transactions.
605    ///
606    /// # Arguments
607    /// - `key`: A key for the data stored in the underlying database
608    #[cfg(test)]
609    pub fn get<T: serde::Serialize + for<'de> serde::Deserialize<'de>>(
610        &self,
611        store_id: StoreId,
612        key: &str,
613    ) -> Result<Option<T>> {
614        let reader = self.rkv.read()?;
615        let persisted_data = self.get_store(store_id).store.get(&reader, key)?;
616        match persisted_data {
617            Some(data) => {
618                if let rkv::Value::Json(data) = data {
619                    Ok(Some(match serde_json::from_str::<T>(data) {
620                        Ok(v) => v,
621                        Err(e) => return Err(NimbusError::JSONError("rkv::Value::Json(data) nimbus::stateful::persistence::Database::get::serde_json::from_str".into(), e.to_string()))
622                    }))
623                } else {
624                    Err(NimbusError::InvalidPersistedData)
625                }
626            }
627            None => Ok(None),
628        }
629    }

631    // Function for collecting all items in a store outside of a transaction.
632    // Only available for tests; product code should always be using transactions.
633    // Iters are a bit tricky - would be nice to make them generic, but this will
634    // do for our use-case.
635    #[cfg(test)]
636    pub fn collect_all<T: serde::Serialize + for<'de> serde::Deserialize<'de>>(
637        &self,
638        store_id: StoreId,
639    ) -> Result<Vec<T>> {
640        let mut result = Vec::new();
641        let reader = self.rkv.read()?;
642        let mut iter = self.get_store(store_id).store.iter_start(&reader)?;
643        while let Some(Ok((_, data))) = iter.next() {
644            if let rkv::Value::Json(data) = data {
645                result.push(match serde_json::from_str::<T>(data) {
646                    Ok(v) => v,
647                    Err(e) => return Err(NimbusError::JSONError("rkv::Value::Json(data) nimbus::stateful::persistence::Database::collect_all::serde_json::from_str".into(), e.to_string()))
648                });
649            }
650        }
651        Ok(result)
652    }
