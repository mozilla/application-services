5//! This module implements the sampling logic required to hash,
6//! randomize and pick branches using pre-set ratios.

12const HASH_BITS: u32 = 48;
13const HASH_LENGTH: u32 = HASH_BITS / 4;

15/// Sample by splitting the input space into a series of buckets, checking
16/// if the given input is in a range of buckets
17///
18/// The range to check is defined by a start point and length, and can wrap around
19/// the input space. For example, if there are 100 buckets, and we ask to check 50 buckets
20/// starting from bucket 70, then buckets 70-99 and 0-19 will be checked
21///
22/// # Arguments:
23///
24/// - `input` What will be hashed and matched against the range of the buckets
25/// - `start` the index of the bucket to start checking
26/// - `count` then number of buckets to check
27/// - `total` The total number of buckets to group inputs into
28///
29/// # Returns:
30///
31/// Returns true if the hash generated from the input belongs within the range
32/// otherwise false
33///
34/// # Errors:
35///
36/// Could error in the following cases (but not limited to)
37/// - An error occurred in the hashing process
38/// - an error occurred while checking if the hash belongs in the bucket
39pub(crate) fn bucket_sample<T: serde::Serialize>(
40    input: T,
41    start: u32,
42    count: u32,
43    total: u32,
44) -> Result<bool> {
45    let input_hash = hex::encode(truncated_hash(input)?);
46    let wrapped_start = start % total;
47    let end = wrapped_start + count;
48
49    Ok(if end > total {
50        is_hash_in_bucket(&input_hash, 0, end % total, total)?
51            || is_hash_in_bucket(&input_hash, wrapped_start, total, total)?
52    } else {
53        is_hash_in_bucket(&input_hash, wrapped_start, end, total)?
54    })
55}

57/// Sample over a list of ratios such that, over the input space, each
58/// ratio has a number of matches in correct proportion to the other ratios
59///
60/// # Arguments:
61/// - `input`: the input used in the sampling process
62/// - `ratios`: The list of ratios associated with each option
63///
64/// # Example:
65///
66/// Assuming the ratios: `[1, 2, 3, 4]`
67/// 10% of all inputs will return 0, 20% will return 1 and so on
68///
69/// # Returns
70/// Returns an index of the ratio that matched the input
71///
72/// # Errors
73/// Could return an error if the input couldn't be hashed
74pub(crate) fn ratio_sample<T: serde::Serialize>(input: T, ratios: &[u32]) -> Result<usize> {
75    if ratios.is_empty() {
76        return Err(NimbusError::EmptyRatiosError);
77    }
78    let input_hash = hex::encode(truncated_hash(input)?);
79    let ratio_total: u32 = ratios.iter().sum();
80    let mut sample_point = 0;
81    for (i, ratio) in ratios.iter().enumerate() {
82        sample_point += ratio;
83        if input_hash <= fraction_to_key(sample_point as f64 / ratio_total as f64)? {
84            return Ok(i);
85        }
86    }
87    Ok(ratios.len() - 1)
88}

90/// Provides a hash of `data`, truncated to the 6 most significant bytes
91/// For consistency with: https://searchfox.org/mozilla-central/source/toolkit/components/utils/Sampling.jsm#79
92/// # Arguments:
93/// - `data`: The data to be hashed
94///
95/// # Returns:
96/// Returns the 6 bytes associated with the SHA-256 of the data
97///
98/// # Errors:
99/// Would return an error if the hashing function fails to generate a hash
100/// that is larger than 6 bytes (Should never occur)
101pub(crate) fn truncated_hash<T: serde::Serialize>(data: T) -> Result<[u8; 6]> {
102    let mut hasher = Sha256::new();
103    let data_str = match serde_json::to_string(&data) {
104        Ok(v) => v,
105        Err(e) => {
106            return Err(NimbusError::JSONError(
107                "data_str = nimbus::sampling::truncated_hash::serde_json::to_string".into(),
108                e.to_string(),
109            ));
110        }
111    };
112    hasher.update(data_str.as_bytes());
113    Ok(hasher.finalize()[0..6].try_into()?)
114}

116/// Checks if a given hash (represented as a 6 byte hex string) fits within a bucket range
117///
118/// # Arguments:
119/// - `input_hash_num`: The hash as a 6 byte hex string (12 hex digits)
120/// - `min_bucket`: The minimum bucket number
121/// - `max_bucket`: The maximum bucket number
122/// - `bucket_count`: The number of buckets
123///
124/// # Returns
125/// Returns true if the has fits in the bucket range,
126/// otherwise false
127///
128/// # Errors:
129///
130/// Could return an error if bucket numbers are higher than the bucket count
131fn is_hash_in_bucket(
132    input_hash_num: &str,
133    min_bucket: u32,
134    max_bucket: u32,
135    bucket_count: u32,
136) -> Result<bool> {
137    let min_hash = fraction_to_key(min_bucket as f64 / bucket_count as f64)?;
138    let max_hash = fraction_to_key(max_bucket as f64 / bucket_count as f64)?;
139    Ok(min_hash.as_str() <= input_hash_num && input_hash_num < max_hash.as_str())
140}

142/// Maps from the range [0, 1] to [0, 2^48]
143///
144/// # Argument:
145/// - `fraction`: float in the range 0-1
146///
147/// # Returns
148/// returns a hex string representing the fraction multiplied to be within the
149/// [0, 2^48] range
150///
151/// # Errors
152/// returns an error if the fraction not within the 0-1 range
153fn fraction_to_key(fraction: f64) -> Result<String> {
154    if !(0.0..=1.0).contains(&fraction) {
155        return Err(NimbusError::InvalidFraction);
156    }
157    let multiplied = (fraction * (2u64.pow(HASH_BITS) - 1) as f64).floor();
158    let multiplied = format!("{:x}", multiplied as u64);
159    let padding = vec!['0'; HASH_LENGTH as usize - multiplied.len()];
160    let res = padding
161        .into_iter()
162        .chain(multiplied.chars())
163        .collect::<String>();
164    Ok(res)
165}
