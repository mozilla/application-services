15/// Replace any instance of [from] with [to] in any string within the [serde_json::Value].
16///
17/// This recursively descends into the object, looking at string values and keys.
18#[allow(dead_code)]
19pub(crate) fn replace_str(value: &mut Value, from: &str, to: &str) {
20    let replacer = create_str_replacer(from, to);
21    replace_str_with(value, &replacer);
22}
23
24/// Replace any instance of [from] with [to] in any string within the [serde_json::Value::Map].
25///
26/// This recursively descends into the object, looking at string values and keys.
27pub(crate) fn replace_str_in_map(map: &mut Map<String, Value>, from: &str, to: &str) {
28    let replacer = create_str_replacer(from, to);
29    replace_str_in_map_with(map, &replacer);
30}
31
32fn replace_str_with<F>(value: &mut Value, replacer: &F)
33where
34    F: Fn(&str) -> Option<String> + ?Sized,
35{
36    match value {
37        Value::String(s) => {
38            if let Some(r) = replacer(s) {
39                *s = r;
40            }
41        }
42
43        Value::Array(list) => {
44            for item in list.iter_mut() {
45                replace_str_with(item, replacer);
46            }
47        }
48
49        Value::Object(map) => {
50            replace_str_in_map_with(map, replacer);
51        }
52
53        _ => (),
54    };
55}
56
57pub(crate) fn replace_str_in_map_with<F>(map: &mut Map<String, Value>, replacer: &F)
58where
59    F: Fn(&str) -> Option<String> + ?Sized,
60{
61    // Replace values in place.
62    for v in map.values_mut() {
63        replace_str_with(v, replacer);
64    }
65
66    // Replacing keys in place is a little trickier.
67    let mut changes = HashMap::new();
68    for k in map.keys() {
69        if let Some(new) = replacer(k) {
70            changes.insert(k.to_owned(), new);
71        }
72    }
73
74    for (k, new) in changes {
75        let v = map.remove(&k).unwrap();
76        _ = map.insert(new, v);
77    }
78}
79
80fn create_str_replacer<'a>(from: &'a str, to: &'a str) -> impl Fn(&str) -> Option<String> + 'a {
81    move |s: &str| -> Option<String> {
82        if s.contains(from) {
83            Some(s.replace(from, to))
84        } else {
85            None
86        }
87    }
88}

90#[cfg(test)]
91mod unit_tests {
92    use super::*;
93    use serde_json::json;
94
95    #[test]
96    fn test_replace_str() {
97        let mut value = json!("{test}");
98        replace_str(&mut value, "{test}", "success");
99        assert_eq!(value, json!("success"));
100
101        let mut value = json!("{test}-postfix");
102        replace_str(&mut value, "{test}", "success");
103        assert_eq!(value, json!("success-postfix"));
104
105        let mut value = json!("prefix-{test}");
106        replace_str(&mut value, "{test}", "success");
107        assert_eq!(value, json!("prefix-success"));
108
109        let mut value = json!("prefix-{test}-postfix");
110        replace_str(&mut value, "{test}", "success");
111        assert_eq!(value, json!("prefix-success-postfix"));
112
113        let mut value = json!("prefix-{test}-multi-{test}-postfix");
114        replace_str(&mut value, "{test}", "success");
115        assert_eq!(value, json!("prefix-success-multi-success-postfix"));
116    }
117
118    #[test]
119    fn test_replace_str_in_array() {
120        let mut value = json!(["alice", "bob", "{placeholder}", "daphne"]);
121        replace_str(&mut value, "{placeholder}", "charlie");
122        assert_eq!(value, json!(["alice", "bob", "charlie", "daphne"]));
123    }
124
125    #[test]
126    fn test_replace_str_in_map() {
127        let mut value = json!({
128            "key": "{test}",
129            "not": true,
130            "or": 2,
131        });
132        replace_str(&mut value, "{test}", "success");
133        assert_eq!(
134            value,
135            json!({
136                "key": "success",
137                "not": true,
138                "or": 2,
139            })
140        );
141    }
142
143    #[test]
144    fn test_replace_str_in_map_keys() {
145        let mut value = json!({

148            "or": 2,
149        });
150        replace_str(&mut value, "{test}", "success");
151        assert_eq!(
152            value,
153            json!({
154                "success-en-US": "success",

157            })

159    }
