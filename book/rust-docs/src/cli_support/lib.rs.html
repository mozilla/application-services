30pub fn cli_data_dir() -> String {
31    ensure_cli_data_dir_exists();
32    data_path(None).to_string_lossy().to_string()
33}
34
35pub fn ensure_cli_data_dir_exists() {
36    let dir = data_path(None);
37    if !dir.exists() {
38        std::fs::create_dir(&dir).unwrap_or_else(|_| panic!("Error creating dir: {dir:?}"))
39    }
40}
41
42pub fn cli_data_subdir(relative_path: &str) -> String {
43    ensure_cli_data_dir_exists();
44    data_path(Some(relative_path)).to_string_lossy().to_string()
45}
46
47pub fn cli_data_path(filename: &str) -> String {
48    ensure_cli_data_dir_exists();
49    data_path(None).join(filename).to_string_lossy().to_string()
50}
51
52fn data_path(relative_path: Option<&str>) -> PathBuf {
53    let dir = workspace_root_dir().join(".cli-data");
54    match relative_path {
55        None => dir,
56        Some(relative_path) => dir.join(relative_path),
57    }
58}
59
60pub fn workspace_root_dir() -> PathBuf {
61    let cargo_output = std::process::Command::new(env!("CARGO"))
62        .arg("locate-project")
63        .arg("--workspace")
64        .arg("--message-format=plain")
65        .output()
66        .unwrap()
67        .stdout;
68    let cargo_toml_path = Path::new(std::str::from_utf8(&cargo_output).unwrap().trim());
69    cargo_toml_path.parent().unwrap().to_path_buf()
70}

72pub fn remote_settings_service(
73    rs_server: Option<RemoteSettingsServer>,
74) -> Arc<RemoteSettingsService> {
75    let config = match rs_server {
76        Some(rs_server) => RemoteSettingsConfig2 {
77            server: Some(rs_server),
78            ..Default::default()
79        },
80        None => RemoteSettingsConfig2::default(),
81    };
82
83    Arc::new(RemoteSettingsService::new(
84        data_path(Some("remote-settings"))
85            .to_string_lossy()
86            .to_string(),
87        config,
88    ))
89}
