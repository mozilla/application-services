30fn load_fxa_creds(path: &str) -> Result<FirefoxAccount> {
31    let mut file = fs::File::open(path)?;
32    let mut s = String::new();
33    file.read_to_string(&mut s)?;
34    Ok(FirefoxAccount::from_json(&s)?)
35}
36
37fn load_or_create_fxa_creds(path: &str, cfg: FxaConfig, scopes: &[&str]) -> Result<FirefoxAccount> {
38    match load_fxa_creds(path) {
39        Ok(account) => {
40            if !account.matches_server(&cfg.server)? {
41                bail!("Stored credentials don't match configured server.\nDelete {path} to start over or specify a different server arg")
42            }
43            Ok(account)
44        }
45        Err(e) => {
46            log::info!(
47                "Failed to load existing FxA credentials from {:?} (error: {}), launching OAuth flow",
48                path,
49                e
50            );
51            create_fxa_creds(path, cfg, scopes)
52        }
53    }
54}

56fn create_fxa_creds(path: &str, cfg: FxaConfig, scopes: &[&str]) -> Result<FirefoxAccount> {
57    let acct = FirefoxAccount::new(cfg);
58    handle_oauth_flow(path, &acct, scopes)?;
59    Ok(acct)
60}
61
62fn handle_oauth_flow(path: &str, acct: &FirefoxAccount, scopes: &[&str]) -> Result<()> {
63    let oauth_uri = acct.begin_oauth_flow(scopes, "fxa_creds")?;
64
65    if open::that(&oauth_uri).is_err() {
66        log::warn!("Failed to open a web browser D:");
67        println!("Please visit this URL, sign in, and then copy-paste the final URL below.");
68        println!("\n    {}\n", oauth_uri);
69    } else {
70        println!("Please paste the final URL below:\n");
71    }
72
73    let final_url = url::Url::parse(&prompt_string("Final URL").unwrap_or_default())?;
74    let query_params = final_url
75        .query_pairs()
76        .into_owned()
77        .collect::<HashMap<String, String>>();
78
79    acct.complete_oauth_flow(&query_params["code"], &query_params["state"])?;
80    // Device registration.
81    acct.initialize_device("CLI Device", sync15::DeviceType::Desktop, vec![])?;
82    let mut file = fs::File::create(path)?;
83    write!(file, "{}", acct.to_json()?)?;
84    file.flush()?;
85    Ok(())
86}

95pub fn get_account_and_token(
96    config: FxaConfig,
97    cred_file: &str,
98    scopes: &[&str],
99) -> Result<(FirefoxAccount, AccessTokenInfo)> {
100    // TODO: we should probably set a persist callback on acct?
101    let acct = load_or_create_fxa_creds(cred_file, config.clone(), scopes)?;
102    // `scope` could be a param, but I can't see it changing.
103    match acct.get_access_token(SYNC_SCOPE, true) {
104        Ok(t) => Ok((acct, t)),
105        Err(e) => {
106            match e {
107                // We can retry an auth error.
108                FxaError::Authentication => {
109                    println!("Saw an auth error using stored credentials - attempting to re-authenticate");
110                    println!("If fails, consider deleting {cred_file} to start from scratch");
111                    handle_oauth_flow(cred_file, &acct, scopes)?;
112                    let token = acct.get_access_token(SYNC_SCOPE, true)?;
113                    Ok((acct, token))
114                }
115                _ => Err(e.into()),
116            }
117        }
118    }
119}

121pub fn get_cli_fxa(config: FxaConfig, cred_file: &str, scopes: &[&str]) -> Result<CliFxa> {
122    let (account, token_info) = match get_account_and_token(config, cred_file, scopes) {
123        Ok(v) => v,
124        Err(e) => anyhow::bail!("Failed to use saved credentials. {}", e),
125    };
126    let tokenserver_url = Url::parse(&account.get_token_server_endpoint_url()?)?;
127
128    let client_init = Sync15StorageClientInit {
129        key_id: token_info.key.as_ref().unwrap().kid.clone(),
130        access_token: token_info.token.clone(),
131        tokenserver_url: tokenserver_url.clone(),
132    };
133
134    Ok(CliFxa {
135        account,
136        client_init,
137        tokenserver_url,
138        token_info,
139    })
140}

142pub struct CliFxa {

148
149impl CliFxa {
150    // A helper for consumers who use this with the sync manager.
151    pub fn as_auth_info(&self) -> sync_manager::SyncAuthInfo {
152        let scoped_key = self.token_info.key.as_ref().unwrap();
153        sync_manager::SyncAuthInfo {
154            kid: scoped_key.kid.clone(),
155            sync_key: scoped_key.k.clone(),
156            fxa_access_token: self.token_info.token.clone(),
157            tokenserver_url: self.tokenserver_url.to_string(),
158        }
159    }
160
161    // A helper for consumers who use this directly with sync15
162    pub fn as_key_bundle(&self) -> Result<KeyBundle> {
163        let scoped_key = self.token_info.key.as_ref().unwrap();
164        Ok(KeyBundle::from_ksync_bytes(&scoped_key.key_bytes()?)?)
165    }
