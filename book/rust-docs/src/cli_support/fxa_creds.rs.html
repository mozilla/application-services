54/// Manages FxA credentials for CLI examples and tests.
55///
56/// Typical usage:
57/// ```no_run
58/// let mut cli = CliFxa::new(get_default_fxa_config(), None)?;
59/// cli.ensure_logged_in(&[SYNC_SCOPE])?;
60/// let sync = cli.sync_info()?.expect("logged in with SYNC_SCOPE");
61/// ```
62pub struct CliFxa {
63    config: FxaConfig,
64    cred_path: String,
65    account: Option<FirefoxAccount>,
66}
67
68impl CliFxa {
69    /// Create a `CliFxa`, loading saved credentials if they exist.
70    ///
71    /// `cred_path` defaults to `CREDENTIALS_FILENAME` in the workspace root.
72    pub fn new(config: FxaConfig, cred_path: Option<&str>) -> Result<Self> {
73        let cred_path = cred_path.map(|s| s.to_owned()).unwrap_or_else(|| {
74            workspace_root_dir()
75                .join(CREDENTIALS_FILENAME)
76                .to_string_lossy()
77                .to_string()
78        });
79
80        let account = match load_account(&cred_path, &config) {
81            Ok(acct) => {
82                crate::info!("Loaded saved FxA credentials from {cred_path}");
83                match acct.check_authorization_status() {
84                    Ok(info) if info.active => {
85                        crate::info!("fxa stored credentials are active");
86                        Some(acct)
87                    }
88                    Ok(_) => {
89                        crate::warn!("fxa stored credentials are no longer active; re-login will be required");
90                        None
91                    }
92                    Err(e) => {
93                        crate::warn!("FxA: could not verify authorization status ({e}); will attempt to use stored credentials");
94                        Some(acct)
95                    }
96                }
97            }
98            Err(e) => {
99                crate::info!("No saved FxA credentials at {cred_path} ({e}); login required");
100                None
101            }
102        };
103
104        Ok(Self {
105            config,
106            cred_path,
107            account,
108        })
109    }

111    /// Ensure the user is logged in, running interactive OAuth if needed.
112    ///
113    /// Uses the state machine: `Initialize` → `BeginOAuthFlow` if needed → `CompleteOAuthFlow`.
114    /// Persists credentials after a successful login.
115    pub fn ensure_logged_in(&mut self, scopes: &[&str]) -> Result<&FirefoxAccount> {
116        let service = ""; // TODO: expose this.
117        if self.account.is_none() {
118            crate::info!("Creating new FxA account object");
119            self.account = Some(FirefoxAccount::new(self.config.clone()));
120        }
121
122        let device_config = DeviceConfig {
123            name: DEVICE_NAME.to_owned(),
124            device_type: DeviceType::Desktop,
125            capabilities: vec![],
126        };
127
128        let account = self.account.as_mut().unwrap();
129
130        let state = account.process_event(FxaEvent::Initialize { device_config })?;
131
132        match state {
133            FxaState::Connected => {
134                let have_all_scopes = account.has_scope(&scopes.join(" "));
135                crate::info!("FxA: already connected, all scopes is {have_all_scopes}");
136                if !have_all_scopes {
137                    self.handle_oauth_flow(service, scopes)?;
138                }
139            }
140            FxaState::Disconnected | FxaState::AuthIssues => {
141                crate::info!("FxA: need to authenticate (state was {state:?})");
142                self.handle_oauth_flow(service, scopes)?;
143            }
144            other => {
145                anyhow::bail!("Unexpected FxA state after Initialize: {other:?}");
146            }
147        }
148
149        Ok(self.account.as_ref().unwrap())
150    }
151
152    /// Get the account reference if logged in.
153    pub fn account(&self) -> Option<&FirefoxAccount> {
154        self.account.as_ref()
155    }

157    /// Get everything needed to sync, built fresh from the current access token.
158    ///
159    /// Returns `Ok(None)` if the sync scope key is not available (e.g., not logged in with
160    /// `SYNC_SCOPE`). Returns `Err` for real errors.
161    pub fn sync_info(&self) -> Result<Option<CliSyncInfo>> {
162        let account = self
163            .account
164            .as_ref()
165            .ok_or_else(|| anyhow::anyhow!("not logged in; call ensure_logged_in first"))?;
166
167        let token = account.get_access_token(SYNC_SCOPE, false)?;
168        let key = match token.key {
169            Some(k) => k,
170            None => {
171                crate::info!("No sync key in token — not logged in with SYNC_SCOPE");
172                return Ok(None);
173            }
174        };
175
176        let tokenserver_url = Url::parse(&account.get_token_server_endpoint_url()?)?;
177        let client_init = Sync15StorageClientInit {
178            key_id: key.kid.clone(),
179            access_token: token.token.clone(),
180            tokenserver_url: tokenserver_url.clone(),
181        };
182        let key_bundle = KeyBundle::from_ksync_bytes(&key.key_bytes()?)?;
183        let auth_info = sync_manager::SyncAuthInfo {
184            kid: key.kid.clone(),
185            sync_key: key.k.clone(),
186            fxa_access_token: token.token.clone(),
187            tokenserver_url: tokenserver_url.to_string(),
188        };
189
190        Ok(Some(CliSyncInfo {
191            client_init,
192            key_bundle,
193            auth_info,
194            tokenserver_url,
195        }))
196    }

198    /// Persist account state to disk.
199    ///
200    /// Called automatically after login. Call this explicitly after any direct
201    /// account operation (e.g. `set_device_name`, `poll_device_commands`) that
202    /// may update local state.
203    pub fn persist(&self) -> Result<()> {
204        if let Some(account) = &self.account {
205            let json = account.to_json()?;
206            let mut file = fs::File::create(&self.cred_path)?;
207            write!(file, "{json}")?;
208            file.flush()?;
209            crate::info!("Saved FxA credentials to {}", self.cred_path);
210        }
211        Ok(())
212    }
213
214    /// Run the interactive browser OAuth flow and complete the login.
215    fn handle_oauth_flow(&mut self, service: &str, scopes: &[&str]) -> Result<()> {
216        let scopes: Vec<String> = scopes.iter().map(|s| s.to_string()).collect();
217
218        let state = self
219            .account
220            .as_mut()
221            .unwrap()
222            .process_event(FxaEvent::BeginOAuthFlow {
223                service: service.to_owned(),
224                scopes,
225                entrypoint: OAUTH_ENTRYPOINT.to_owned(),
226            })?;
227
228        let oauth_url = match state {
229            FxaState::Authenticating { oauth_url, .. } => oauth_url,
230            other => anyhow::bail!("Unexpected FxA state after BeginOAuthFlow: {other:?}"),
231        };
232
233        println!("In a (probably private) browser window, please open:");
234        println!(" {oauth_url}\n");
235        println!("paste the final 'Connected' URL from a successful flow below.");
236        let final_url = Url::parse(&prompt_string("Final URL").unwrap_or_default())?;
237        let query_params: HashMap<String, String> = final_url.query_pairs().into_owned().collect();
238
239        let state = self
240            .account
241            .as_mut()
242            .unwrap()
243            .process_event(FxaEvent::CompleteOAuthFlow {
244                code: query_params["code"].clone(),
245                state: query_params["state"].clone(),
246            })?;
247
248        match state {
249            FxaState::Connected => {
250                crate::info!("FxA: OAuth flow complete, now connected");
251                self.persist()
252            }
253            FxaState::Disconnected => {
254                crate::warn!("FxA: OAuth flow failed to complete, account is not connected");
255                self.persist()
256            }
257            other => anyhow::bail!("Unexpected FxA state after CompleteOAuthFlow: {other:?}"),
258        }
259    }
260}

262fn load_account(path: &str, config: &FxaConfig) -> Result<FirefoxAccount> {
263    let s = fs::read_to_string(path)?;
264    let account = FirefoxAccount::from_json(&s)?;
265    if !account.matches_server(&config.server)? {
266        anyhow::bail!(
267            "Stored credentials don't match configured server.\n\
268             Delete {path} to start over or use a different server arg"
269        )
270    }
271    Ok(account)
