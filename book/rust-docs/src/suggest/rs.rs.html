6//! Crate-internal types for interacting with Remote Settings (`rs`). Types in
7//! this module describe records and attachments in the Suggest Remote Settings
8//! collection.
9//!
10//! To add a new suggestion `T` to this component, you'll generally need to:
11//!
12//!  1. Add a variant named `T` to [`SuggestRecord`]. The variant must have a
13//!     `#[serde(rename)]` attribute that matches the suggestion record's
14//!     `type` field.
15//!  2. Define a `DownloadedTSuggestion` type with the new suggestion's fields,
16//!     matching their attachment's schema. Your new type must derive or
17//!     implement [`serde::Deserialize`].
18//!  3. Update the database schema in the [`schema`] module to store the new
19//!     suggestion.
20//!  4. Add an `insert_t_suggestions()` method to [`db::SuggestDao`] that
21//!     inserts `DownloadedTSuggestion`s into the database.
22//!  5. Update [`store::SuggestStoreInner::ingest()`] to download, deserialize,
23//!     and store the new suggestion.
24//!  6. Add a variant named `T` to [`suggestion::Suggestion`], with the fields
25//!     that you'd like to expose to the application. These can be the same
26//!     fields as `DownloadedTSuggestion`, or slightly different, depending on
27//!     what the application needs to show the suggestion.
28//!  7. Update the `Suggestion` enum definition in `suggest.udl` to match your
29//!     new [`suggestion::Suggestion`] variant.
30//!  8. Update any [`db::SuggestDao`] methods that query the database to include
31//!     the new suggestion in their results, and return `Suggestion::T` variants
32//!     as needed.

61/// A trait for a client that downloads suggestions from Remote Settings.
62///
63/// This trait lets tests use a mock client.
64pub(crate) trait Client {
65    /// Get all records from the server
66    ///
67    /// We use this plus client-side filtering rather than any server-side filtering, as
68    /// recommended by the remote settings docs
69    /// (https://remote-settings.readthedocs.io/en/stable/client-specifications.html). This is
70    /// relatively inexpensive since we use a cache and don't fetch attachments until after the
71    /// client-side filtering.
72    ///
73    /// Records that can't be parsed as [SuggestRecord] are ignored.
74    fn get_records(&self, collection: Collection) -> Result<Vec<Record>>;
75
76    fn download_attachment(&self, record: &Record) -> Result<Vec<u8>>;
77}

79/// Implements the [Client] trait using a real remote settings client
80pub struct SuggestRemoteSettingsClient {
81    // Create a separate client for each collection name
82    amp_client: Arc<RemoteSettingsClient>,
83    other_client: Arc<RemoteSettingsClient>,
84}
85
86impl SuggestRemoteSettingsClient {
87    pub fn new(rs_service: &RemoteSettingsService) -> Self {
88        Self {
89            amp_client: rs_service.make_client(Collection::Amp.name().to_owned()),
90            other_client: rs_service.make_client(Collection::Other.name().to_owned()),
91        }
92    }
93
94    fn client_for_collection(&self, collection: Collection) -> &RemoteSettingsClient {
95        match collection {
96            Collection::Amp => &self.amp_client,
97            Collection::Other => &self.other_client,
98        }
99    }
100}
101
102impl Client for SuggestRemoteSettingsClient {
103    fn get_records(&self, collection: Collection) -> Result<Vec<Record>> {
104        let client = self.client_for_collection(collection);
105        client.sync()?;
106        let response = client.get_records(false);
107        match response {
108            Some(r) => Ok(r
109                .into_iter()
110                .filter_map(|r| Record::new(r, collection).ok())
111                .collect()),
112            None => Err(Error::RemoteSettings(RemoteSettingsError::Other {
113                reason: "Unable to get records".to_owned(),
114            })),
115        }
116    }
117
118    fn download_attachment(&self, record: &Record) -> Result<Vec<u8>> {
119        let converted_record: RemoteSettingsRecord = record.clone().into();
120        match &record.attachment {
121            Some(_) => Ok(self
122                .client_for_collection(record.collection)
123                .get_attachment(&converted_record)?),
124            None => Err(Error::MissingAttachment(record.id.to_string())),
125        }
126    }
127}

382/// Iterate over all AMP/Wikipedia-style keywords.
383pub fn iterate_keywords<'a>(
384    keywords: &'a [String],
385    full_keywords: &'a [(String, usize)],
386) -> impl Iterator<Item = AmpKeyword<'a>> {
387    let full_keywords_iter = full_keywords
388        .iter()
389        .flat_map(|(full_keyword, repeat_for)| {
390            std::iter::repeat_n(Some(full_keyword.as_str()), *repeat_for)
391        })
392        .chain(std::iter::repeat(None)); // In case of insufficient full keywords, just fill in with infinite `None`s
393                                         //
394    keywords
395        .iter()
396        .zip(full_keywords_iter)
397        .enumerate()
398        .map(move |(i, (keyword, full_keyword))| AmpKeyword {
399            rank: i,
400            keyword,
401            full_keyword,
402        })
403}
404
405impl DownloadedAmpSuggestion {
406    pub fn keywords(&self) -> impl Iterator<Item = AmpKeyword<'_>> {
407        iterate_keywords(&self.keywords, &self.full_keywords)
408    }
409
410    pub fn full_keywords_fts_column(&self) -> String {
411        full_keywords_to_fts_content(self.full_keywords.iter().map(|(s, _)| s.as_str()))
412    }
413}

514/// A single full keyword or a `(prefix, suffixes)` tuple representing multiple
515/// prefix keywords. Prefix keywords are enumerated by appending to `prefix`
516/// each possible prefix of each suffix, including the full suffix. The prefix
517/// is also enumerated by itself. Examples:
518///
519/// `FullOrPrefixKeywords::Full("some full keyword")`
520/// => "some full keyword"
521///
522/// `FullOrPrefixKeywords::Prefix(("sug", vec!["gest", "arplum"]))`
523/// => "sug"
524///    "sugg"
525///    "sugge"
526///    "sugges"
527///    "suggest"
528///    "suga"
529///    "sugar"
530///    "sugarp"
531///    "sugarpl"
532///    "sugarplu"
533///    "sugarplum"
534#[derive(Clone, Debug, Deserialize)]
535#[serde(untagged)]
536enum FullOrPrefixKeywords<T> {
537    Full(T),
538    Prefix((T, Vec<T>)),
539}
540
541impl<T> From<T> for FullOrPrefixKeywords<T> {
542    fn from(full_keyword: T) -> Self {
543        Self::Full(full_keyword)
544    }
545}
546
547impl<T> From<(T, Vec<T>)> for FullOrPrefixKeywords<T> {
548    fn from(prefix_suffixes: (T, Vec<T>)) -> Self {
549        Self::Prefix(prefix_suffixes)
550    }
551}
552
553impl FullOrPrefixKeywords<String> {
554    pub fn keywords(&self) -> Box<dyn Iterator<Item = String> + '_> {
555        match self {
556            FullOrPrefixKeywords::Full(kw) => Box::new(std::iter::once(kw.to_owned())),
557            FullOrPrefixKeywords::Prefix((prefix, suffixes)) => Box::new(
558                std::iter::once(prefix.to_owned()).chain(suffixes.iter().flat_map(|suffix| {
559                    let mut kw = prefix.clone();
560                    suffix.chars().map(move |c| {
561                        kw.push(c);
562                        kw.clone()
563                    })
564                })),
565            ),
566        }
567    }
568}
