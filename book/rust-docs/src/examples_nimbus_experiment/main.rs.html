5use clap::{Parser, Subcommand};
6use std::path::PathBuf;
7
8use nimbus::error::{info, Result};
9
10#[derive(Parser)]
11#[command(name = "Nimbus SDK Demo")]
12#[command(author = "Tarik E. <teshaq@mozilla.com>")]
13#[command(about = "A demo for the Nimbus SDK")]
14struct Args {
15    /// Custom File configuration
16    #[arg(short, long, value_name = "FILE")]
17    config: PathBuf,
18
19    /// Custom collection name
20    #[arg(short = 'n', long)]
21    collection: Option<String>,
22
23    #[arg(short, long, value_name = "SERVER_URL")]
24    /// Specifies the server to use
25    server: Option<String>,
26
27    #[arg(long, value_name = "PATH")]
28    /// Path where the database will be created"
29    db_path: Option<String>,
30
31    #[command(subcommand)]
32    subcommand: Subcommands,
33}
34
35#[derive(Subcommand)]
36enum Subcommands {
37    /// Show all experiments, followed by the enrolled experiments
38    ShowExperiments,
39
40    /// Fetch experiments from the server. Subsequent calls to apply-pending-experiments will change enrolments.
41    FetchExperiments,
42
43    /// Updates enrollments with the experiments last fetched from the server with fetch-experiments
44    ApplyPendingExperiments,
45
46    /// Equivalent to fetch-experiments and apply-pending-experiments together
47    UpdateExperiments,
48
49    /// Opts in to an experiment and branch
50    OptIn {
51        #[arg(long, value_name = "EXPERIMENT_ID")]
52        /// The ID of the experiment to opt in to
53        experiment: String,
54
55        #[arg(long, value_name = "BRANCH_ID")]
56        /// The ID of the branch to opt in to
57        branch: String,
58    },
59
60    /// Opts out of an experiment
61    OptOut {
62        #[arg(long, value_name = "EXPERIMENT_ID")]
63        /// The ID of the experiment to opt out of
64        experiment: String,
65    },
66
67    /// Opts out of all experiments
68    OptOutAll,
69
70    /// Generate a uuid that can get enrolled in experiments
71    GenUuid {
72        /// The number of experiments the uuid generated should be able to enroll in,
73        /// WARNING: This can end in an infinite loop if the number is too high
74        #[arg(long, default_value_t = 1)]
75        number: usize,
76
77        /// Sets the UUID in the database when complete.
78        #[arg(long)]
79        set: bool,
80    },
81
82    /// Brute-force an experiment a number of times, showing enrollment results
83    BruteForce {
84        #[arg(long, value_name = "EXPERIMENT_ID")]
85        /// The ID of the experiment to reset
86        experiment: String,
87
88        #[arg(short, long, default_value_t = 10000)]
89        /// The number of times to generate a UUID and attempt enrollment.
90        num: usize,
91    },
92}
93
94fn main() -> Result<()> {
95    const DEFAULT_BASE_URL: &str = "https://firefox.settings.services.mozilla.com";
96    const DEFAULT_COLLECTION_NAME: &str = "messaging-experiments";
97
98    use nimbus::{
99        metrics::{
100            EnrollmentStatusExtraDef, FeatureExposureExtraDef, MalformedFeatureConfigExtraDef,
101            MetricsHandler,
102        },
103        AppContext, AvailableRandomizationUnits, EnrollmentStatus, NimbusClient,
104        NimbusTargetingHelper, RemoteSettingsConfig, RemoteSettingsServer,
105    };
106    use std::collections::HashMap;
107    use std::io::prelude::*;
108
109    pub struct NoopMetricsHandler;
110
111    impl MetricsHandler for NoopMetricsHandler {
112        fn record_enrollment_statuses(&self, _: Vec<EnrollmentStatusExtraDef>) {
113            // do nothing
114        }
115
116        fn record_feature_activation(&self, _activation_event: FeatureExposureExtraDef) {
117            // do nothing
118        }
119
120        fn record_feature_exposure(&self, _exposure_event: FeatureExposureExtraDef) {
121            // do nothing
122        }
123
124        fn record_malformed_feature_config(&self, _event: MalformedFeatureConfigExtraDef) {
125            // do nothing
126        }
127    }
128
129    // We set the logging level to be `warn` here, meaning that only
130    // logs of `warn` or higher will be actually be shown, any other
131    // error will be omitted
132    // To manually set the log level, you can set the `RUST_LOG` environment variable
133    // Possible values are "info", "debug", "warn" and "error"
134    // Check [`env_logger`](https://docs.rs/env_logger/) for more details
135    error_support::init_for_tests_with_level(error_support::Level::Info);
136    viaduct_hyper::init_backend_hyper().expect("Error initalizing viaduct");
137
138    // Initiate the matches for the command line arguments
139    let args = Args::parse();
140
141    // Read command line arguments, or set default values
142    let mut config_file = std::fs::File::open(args.config).expect("Config file does not exist");
143    let mut config = String::new();
144    config_file.read_to_string(&mut config).unwrap();
145    let config = serde_json::from_str::<serde_json::Value>(&config).unwrap();
146
147    let context = config.get("context").unwrap();
148    let context = serde_json::from_value::<AppContext>(context.clone()).unwrap();
149    let server_url = args
150        .server
151        .as_deref()
152        .unwrap_or_else(|| match config.get("server_url") {
153            Some(v) => v.as_str().unwrap(),
154            _ => DEFAULT_BASE_URL,
155        });
156    info!("Server url is {}", server_url);
157
158    let client_id = config
159        .get("client_id")
160        .map(|v| v.to_string())
161        .unwrap_or_else(|| "no-client-id-specified".to_string());
162    info!("Client ID is {}", client_id);
163
164    let collection_name =
165        args.collection
166            .as_deref()
167            .unwrap_or_else(|| match config.get("collection_name") {
168                Some(v) => v.as_str().unwrap(),
169                _ => DEFAULT_COLLECTION_NAME,
170            });
171    info!("Collection name is {}", collection_name);
172
173    let temp_dir = std::env::temp_dir();
174    let db_path_default = temp_dir.to_str().unwrap();
175    let db_path = args
176        .db_path
177        .as_deref()
178        .unwrap_or_else(|| match config.get("db_path") {
179            Some(v) => v.as_str().unwrap(),
180            _ => db_path_default,
181        });
182    info!("Database directory is {}", db_path);
183
184    // initiate the optional config
185    let config = RemoteSettingsConfig {
186        server: Some(RemoteSettingsServer::Custom {
187            url: server_url.to_string(),
188        }),
189        server_url: None,
190        bucket_name: None,
191        collection_name: collection_name.to_string(),
192    };
193
194    // Here we initialize our main `NimbusClient` struct
195    let nimbus_client = NimbusClient::new(
196        context.clone(),
197        Default::default(),
198        Default::default(),
199        db_path,
200        Some(config),
201        Box::new(NoopMetricsHandler),
202        None,
203    )?;
204    info!("Nimbus ID is {}", nimbus_client.nimbus_id()?);
205
206    // Explicitly update experiments at least once for init purposes
207    nimbus_client.fetch_experiments()?;
208    nimbus_client.apply_pending_experiments()?;
209
210    // We match against the subcommands
211    match args.subcommand {
212        // show_enrolled shows only the enrolled experiments and the chosen branches
213        Subcommands::ShowExperiments => {
214            println!("======================================");
215            println!("Printing all experiments (regardless of enrollment)");
216            nimbus_client
217                .get_all_experiments()?
218                .iter()
219                .for_each(|e| println!("Experiment: {}", e.slug));
220            println!("======================================");
221            println!("Printing only enrolled experiments");
222            nimbus_client
223                .get_active_experiments()?
224                .iter()
225                .for_each(|e| {
226                    println!(
227                        "Enrolled in experiment: {}, in branch: {}",
228                        e.slug, e.branch_slug
229                    )
230                });
231        }
232        Subcommands::FetchExperiments => {
233            println!("======================================");
234            println!("Fetching experiments");
235            nimbus_client.fetch_experiments()?;
236        }
237        Subcommands::ApplyPendingExperiments => {
238            println!("======================================");
239            println!("Applying pending experiments");
240            nimbus_client.apply_pending_experiments()?;
241        }
242        Subcommands::UpdateExperiments => {
243            println!("======================================");
244            println!("Fetching and applying experiments");
245            nimbus_client.fetch_experiments()?;
246            nimbus_client.apply_pending_experiments()?;
247        }
248        Subcommands::OptIn { experiment, branch } => {
249            println!("======================================");
250            println!(
251                "Opting in to experiment '{}', branch '{}'",
252                experiment, branch
253            );
254            nimbus_client.opt_in_with_branch(experiment.to_string(), branch.to_string())?;
255        }
256        Subcommands::OptOut { experiment } => {
257            println!("======================================");
258            println!("Opting out of experiment '{}'", experiment);
259            nimbus_client.opt_out(experiment.to_string())?;
260        }
261        Subcommands::OptOutAll => {
262            println!("======================================");
263            println!("Opting out of ALL experiments:");
264            let experiments = nimbus_client.get_all_experiments().unwrap();
265            for experiment in experiments {
266                println!("\t'{}'", &experiment.slug);
267                nimbus_client.opt_out(experiment.slug)?;
268            }
269        }
270        // gen_uuid will generate a UUID that gets enrolled in a given number of
271        // experiments, optionally setting the generated ID in the database.
272        Subcommands::GenUuid { number, set } => {
273            let all_experiments = nimbus_client.get_all_experiments()?;
274            // XXX - this check below isn't good enough - we need to know how
275            // many of those experiments we are actually eligible for!
276            if all_experiments.len() < number {
277                println!(
278                    "Can't try to enroll in {} experiments - only {} exist",
279                    number,
280                    all_experiments.len(),
281                );
282                std::process::exit(1);
283            }
284
285            let mut num_tries = 0;
286            let aru = AvailableRandomizationUnits::default();
287            'outer: loop {
288                let uuid = uuid::Uuid::new_v4();
289                let aru = aru.apply_nimbus_id(&uuid);
290                let mut num_of_experiments_enrolled = 0;
291                let event_store = nimbus_client.event_store();
292                let th = NimbusTargetingHelper::new(&context, event_store.clone(), None);
293                for exp in &all_experiments {
294                    let enr = nimbus::evaluate_enrollment(&aru, exp, &th)?;
295                    if enr.status.is_enrolled() {
296                        num_of_experiments_enrolled += 1;
297                        if num_of_experiments_enrolled >= number {
298                            println!("======================================");
299                            println!("Generated UUID is: {}", uuid);
300                            println!("(it took {} goes to find it)", num_tries);
301                            // ideally we'd
302                            if set {
303                                println!("Setting uuid in the database...");
304                                nimbus_client.set_nimbus_id(&uuid)?;
305                            }
306                            break 'outer;
307                        }
308                    }
309                }
310                num_tries += 1;
311                if num_tries % 5000 == 0 {
312                    println!(
313                        "Made {} attempts so far; it's not looking good...",
314                        num_tries
315                    );
316                }
317            }
318        }
319        Subcommands::BruteForce {
320            experiment: experiment_id,
321            num,
322        } => {
323            println!("Brute-forcing experiment '{}' {} times", experiment_id, num);
324
325            // *sob* no way currently to get by id.
326            let find_exp = || {
327                for exp in nimbus_client
328                    .get_all_experiments()
329                    .expect("can't fetch experiments!?")
330                {
331                    if exp.slug == *experiment_id {
332                        return exp;
333                    }
334                }
335                panic!("No such experiment");
336            };
337            let exp = find_exp();
338            let mut results = HashMap::new();
339            let event_store = nimbus_client.event_store();
340            for _i in 0..num {
341                // Rather than inspecting what randomization unit is specified
342                // by the experiment just generate a new uuid for all possible
343                // options.
344                let uuid = uuid::Uuid::new_v4();
345                let aru = AvailableRandomizationUnits::with_nimbus_id(&uuid);
346                let th = NimbusTargetingHelper::new(&context, event_store.clone(), None);
347                let enrollment = nimbus::evaluate_enrollment(&aru, &exp, &th)?;
348                let key = match enrollment.status.clone() {
349                    EnrollmentStatus::Enrolled { .. } => "Enrolled",
350                    EnrollmentStatus::NotEnrolled { .. } => "NotEnrolled",
351                    EnrollmentStatus::Disqualified { .. } => "Disqualified",
352                    EnrollmentStatus::WasEnrolled { .. } => "WasEnrolled",
353                    EnrollmentStatus::Error { .. } => "Error",
354                };
355                results.insert(key, results.get(&key).unwrap_or(&0) + 1);
356            }
357            println!("Results: {:#?}", results);
358        }
359    };
360    Ok(())
361}
