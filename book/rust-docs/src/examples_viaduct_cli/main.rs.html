5use clap::{Parser, ValueEnum};
6
7use viaduct::{header_names, Client, ClientSettings, Method, Request, Response, Result};
8
9#[derive(Debug, Parser)]
10#[command(about, long_about = None)]
11struct Cli {
12    /// Backend style
13    #[arg(short, long)]
14    backend: Option<BackendStyle>,
15    /// Make a Post request
16    #[arg(short, long)]
17    post_request: bool,
18    /// Set a request timeout (ms)
19    #[arg(short, long)]
20    timeout: Option<u64>,
21}
22
23#[derive(Clone, Debug, ValueEnum)]
24enum BackendStyle {
25    // New backend
26    New,
27    // Bridged backend: initialize the new backend, but use the old API
28    Bridged,
29}
30
31fn main() -> Result<()> {
32    let cli = Cli::parse();
33    println!("{cli:?}");
34    let backend_style = cli.backend.unwrap_or(BackendStyle::New);
35    let req = if cli.post_request {
36        make_post_request()?
37    } else {
38        make_request()?
39    };
40    match backend_style {
41        BackendStyle::New => {
42            viaduct_hyper::init_backend_hyper()?;
43            let settings = ClientSettings {
44                timeout: cli.timeout.unwrap_or(0) as u32,
45                ..ClientSettings::default()
46            };
47            let client = Client::new(settings);
48            print_response(client.send_sync(req))
49        }
50        BackendStyle::Bridged => {
51            viaduct_hyper::init_backend_hyper()?;
52            if let Some(t) = cli.timeout {
53                set_old_global_timeout(t);
54            }
55            print_response(req.send())
56        }
57    }
58
59    Ok(())
60}
61
62fn set_old_global_timeout(timeout: u64) {
63    let mut s = viaduct::settings::GLOBAL_SETTINGS.write();
64    s.connect_timeout = Some(std::time::Duration::from_millis(timeout));
65    s.read_timeout = Some(std::time::Duration::from_millis(timeout));
66}
67
68fn make_request() -> Result<Request> {
69    let url = url::Url::parse("https://httpbun.org/anything")?;
70    let mut req = Request::new(Method::Get, url);
71    req = req.header(header_names::USER_AGENT, "viaduct-cli")?;
72    Ok(req)
73}
74
75#[derive(serde::Serialize)]
76struct TestPostData {
77    data: String,
78    more_data: String,
79}
80
81fn make_post_request() -> Result<Request> {
82    let url = url::Url::parse("https://httpbun.org/anything")?;
83    let mut req = Request::new(Method::Post, url);
84    req = req.header(header_names::USER_AGENT, "viaduct-cli")?;
85    let req = req.json(&TestPostData {
86        data: "Hello".to_owned(),
87        more_data: "World".to_owned(),
88    });
89    Ok(req)
90}
91
92fn print_response(response: Result<Response>) {
93    match response {
94        Ok(response) => {
95            println!("got response");
96            println!("status: {}", response.status);
97            println!("\nHEADERS:");
98            for h in response.headers {
99                println!("{}: {}", h.name, h.value);
100            }
101            println!("\nRESPONSE:");
102            println!("{}", String::from_utf8(response.body).unwrap());
103        }
104        Err(e) => {
105            println!("error: {e}");
106        }
107    }
108}
