11#[derive(Debug, Parser)]
12#[command(about, long_about = None)]
13struct Cli {
14    /// Enable verbose logging
15    #[arg(short, long)]
16    verbose: bool,
17
18    /// Backend style
19    #[arg(short, long)]
20    backend: Option<BackendStyle>,
21
22    /// Set a request timeout (ms)
23    #[arg(short, long)]
24    timeout: Option<u64>,
25
26    #[command(subcommand)]
27    command: Commands,
28}

30#[derive(Debug, Subcommand)]
31enum Commands {
32    /// Make a simple HTTP request
33    Request {
34        /// Make a Post request
35        #[arg(short, long)]
36        post: bool,
37    },
38    /// Test OHTTP flow with a relay
39    #[cfg(feature = "ohttp")]
40    Ohttp {
41        /// OHTTP relay URL
42        #[arg(
43            long,
44            default_value = "https://mozilla-ohttp-relay-test.edgecompute.app/"
45        )]
46        relay_url: String,
47
48        /// Gateway host for fetching encryption keys
49        #[arg(
50            long,
51            default_value = "stage.ohttp-gateway.nonprod.webservices.mozgcp.net"
52        )]
53        gateway_host: String,
54
55        /// Channel name for OHTTP
56        #[arg(long, default_value = "merino")]
57        channel: String,
58    },

61#[derive(Clone, Debug, ValueEnum)]
62enum BackendStyle {
63    // New backend
64    New,
65    // Bridged backend: initialize the new backend, but use the old API
66    Bridged,
67}

69fn main() -> Result<()> {
70    let cli = Cli::parse();
71
72    // Initialize logging
73    cli_support::init_logging_with(if cli.verbose {
74        "viaduct=trace"
75    } else {
76        "viaduct=info"
77    });
78
79    println!("{cli:?}");
80    let backend_style = cli.backend.unwrap_or(BackendStyle::New);
81
82    match cli.command {
83        Commands::Request { post } => {
84            let req = if post {
85                make_post_request()?
86            } else {
87                make_request()?
88            };
89
90            match backend_style {
91                BackendStyle::New => {
92                    viaduct_hyper::init_backend_hyper()?;
93                    let settings = ClientSettings {
94                        timeout: cli.timeout.unwrap_or(0) as u32,
95                        ..ClientSettings::default()
96                    };
97                    let client = Client::new(settings);
98                    print_response(client.send_sync(req));
99                }
100                BackendStyle::Bridged => {
101                    viaduct_hyper::init_backend_hyper()?;
102                    if let Some(t) = cli.timeout {
103                        set_old_global_timeout(t);
104                    }
105                    print_response(req.send());
106                }
107            }
108        }
109        #[cfg(feature = "ohttp")]
110        Commands::Ohttp {
111            relay_url,
112            gateway_host,
113            channel,
114        } => {
115            return run_ohttp_example(relay_url, gateway_host, channel, backend_style);
116        }
117    }
118
119    Ok(())
120}

122#[cfg(feature = "ohttp")]
123fn run_ohttp_example(
124    relay_url: String,
125    gateway_host: String,
126    channel: String,
127    backend_style: BackendStyle,
128) -> Result<()> {
129    // Step 1: Initialize viaduct backend
130    println!("Initializing viaduct backend...");
131
132    match backend_style {
133        BackendStyle::New => {
134            viaduct_hyper::init_backend_hyper()?;
135        }
136        BackendStyle::Bridged => {
137            println!("OHTTP is not compatible with the bridged backend. Use --backend=new or omit the backend parameter.");
138            return Ok(());
139        }
140    }
141
142    println!("Backend initialized successfully");
143
144    // Step 2: Configure the OHTTP channel
145    println!("Configuring OHTTP channel...");
146    configure_ohttp_channel(
147        channel.clone(),
148        OhttpConfig {
149            relay_url,
150            gateway_host,
151        },
152    )?;
153    println!("OHTTP channel configured");
154
155    // Step 3: Create OHTTP client
156    println!("Creating OHTTP client...");
157    let client = Client::with_ohttp_channel(&channel, ClientSettings::default())?;
158    println!("OHTTP client created");
159
160    // Step 4: Make request
161    println!("Creating request...");
162    let request = Request::get(Url::parse("https://merino.services.mozilla.com/docs")?);
163    println!("Sending request...");
164    let response = client.send_sync(request)?;
165
166    // Step 5: Handle response
167    println!("Response received!");
168    println!("Status: {}", response.status);
169    println!("Body: {}", String::from_utf8_lossy(&response.body));
170
171    Ok(())
172}

174fn set_old_global_timeout(timeout: u64) {
175    let mut s = viaduct::settings::GLOBAL_SETTINGS.write();
176    s.connect_timeout = Some(std::time::Duration::from_millis(timeout));
177    s.read_timeout = Some(std::time::Duration::from_millis(timeout));
178}

180fn make_request() -> Result<Request> {
181    let url = Url::parse("https://httpbun.org/anything")?;
182    let mut req = Request::new(Method::Get, url);
183    req = req.header(header_names::USER_AGENT, "viaduct-cli")?;
184    Ok(req)
185}

187#[derive(serde::Serialize)]
188struct TestPostData {
189    data: String,
190    more_data: String,
191}
192
193fn make_post_request() -> Result<Request> {
194    let url = Url::parse("https://httpbun.org/anything")?;
195    let mut req = Request::new(Method::Post, url);
196    req = req.header(header_names::USER_AGENT, "viaduct-cli")?;
197    let req = req.json(&TestPostData {
198        data: "Hello".to_owned(),
199        more_data: "World".to_owned(),
200    });
201    Ok(req)
202}

204fn print_response(response: Result<Response>) {
205    match response {
206        Ok(response) => {
207            println!("got response");
208            println!("status: {}", response.status);
209            println!("\nHEADERS:");
210            for h in response.headers {
211                println!("{}: {}", h.name, h.value);
212            }
213            println!("\nRESPONSE:");
214            println!("{}", String::from_utf8(response.body).unwrap());
215        }
216        Err(e) => {
217            println!("error: {e}");
218        }
219    }
220}
