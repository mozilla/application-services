5use anyhow::Result;
6use clap::{Parser, Subcommand, ValueEnum};
7use std::path::PathBuf;
8
9use dump::client::CollectionDownloader;
10use remote_settings::{RemoteSettingsConfig2, RemoteSettingsServer, RemoteSettingsService};
11
12const DEFAULT_LOG_FILTER: &str = "remote_settings=info";
13const DEFAULT_LOG_FILTER_VERBOSE: &str = "remote_settings=trace";
14
15#[derive(Debug, Parser)]
16#[command(about, long_about = None)]
17struct Cli {
18    #[arg(short = 's')]
19    server: Option<RemoteSettingsServerArg>,
20    #[arg(short = 'b')]
21    bucket: Option<String>,
22    #[arg(short = 'd')]
23    storage_dir: Option<String>,
24    #[arg(long, short, action)]
25    verbose: bool,
26    #[command(subcommand)]
27    command: Commands,
28}
29
30#[derive(Clone, Debug, ValueEnum)]
31enum RemoteSettingsServerArg {
32    Prod,
33    Stage,
34    Dev,
35}
36
37#[derive(Debug, Subcommand)]
38enum Commands {
39    /// Sync collections
40    Sync {
41        #[clap(required = true)]
42        collections: Vec<String>,
43    },
44    /// Query against ingested data
45    Get {
46        collection: String,
47        #[arg(long)]
48        sync_if_empty: bool,
49    },
50    /// Download and combine all remote settings collections
51    DumpSync {
52        /// Root path of the repository
53        #[arg(short, long, default_value = ".")]
54        path: PathBuf,
55
56        /// Dry run - don't write any files
57        #[arg(long, default_value_t = false)]
58        dry_run: bool,
59    },
60    /// Download a single collection to the dumps directory
61    DumpGet {
62        /// Bucket name
63        #[arg(long, required = true)]
64        bucket: String,
65
66        /// Collection name
67        #[arg(long, required = true)]
68        collection_name: String,
69
70        /// Root path of the repository
71        #[arg(short, long, default_value = ".")]
72        path: PathBuf,
73    },
74}
75
76fn main() -> Result<()> {
77    let cli = Cli::parse();
78    cli_support::init_logging_with(if cli.verbose {
79        DEFAULT_LOG_FILTER_VERBOSE
80    } else {
81        DEFAULT_LOG_FILTER
82    });
83    nss::ensure_initialized();
84    viaduct_hyper::viaduct_init_backend_hyper()?;
85    let service = build_service(&cli)?;
86    match cli.command {
87        Commands::Sync { collections } => sync(service, collections),
88        Commands::Get {
89            collection,
90            sync_if_empty,
91        } => {
92            get_records(service, collection, sync_if_empty);
93            Ok(())
94        }
95        Commands::DumpSync { path, dry_run } => {
96            let downloader = CollectionDownloader::new(path);
97            downloader.run(dry_run)
98        }
99        Commands::DumpGet {
100            bucket,
101            collection_name,
102            path,
103        } => {
104            let downloader = CollectionDownloader::new(path);
105            downloader.download_single(&bucket, &collection_name)
106        }
107    }
108}
109
110fn build_service(cli: &Cli) -> Result<RemoteSettingsService> {
111    let config = RemoteSettingsConfig2 {
112        server: cli.server.as_ref().map(|s| match s {
113            RemoteSettingsServerArg::Dev => RemoteSettingsServer::Dev,
114            RemoteSettingsServerArg::Stage => RemoteSettingsServer::Stage,
115            RemoteSettingsServerArg::Prod => RemoteSettingsServer::Prod,
116        }),
117        bucket_name: cli.bucket.clone(),
118        app_context: None,
119    };
120    cli_support::ensure_cli_data_dir_exists();
121    let storage_dir = cli
122        .storage_dir
123        .clone()
124        .unwrap_or_else(|| cli_support::cli_data_subdir("remote-settings-data"));
125    Ok(RemoteSettingsService::new(storage_dir, config))
126}
127
128fn sync(service: RemoteSettingsService, collections: Vec<String>) -> Result<()> {
129    // Create a bunch of clients so that sync() syncs their collections
130    let _clients = collections
131        .into_iter()
132        .map(|collection| service.make_client(collection))
133        .collect::<Vec<_>>();
134    service.sync()?;
135    Ok(())
136}
137
138fn get_records(service: RemoteSettingsService, collection: String, sync_if_empty: bool) {
139    let client = service.make_client(collection);
140    match client.get_records(sync_if_empty) {
141        Some(records) => {
142            for record in records {
143                println!("{record:?}");
144            }
145        }
146        None => println!("No cached records"),
147    }
148}
