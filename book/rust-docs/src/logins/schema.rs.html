5//! Logins Schema v4
6//! ================
7//!
8//! The schema we use is a evolution of the firefox-ios logins database format.
9//! There are three tables:
10//!
11//! - `loginsL`: The local table.
12//! - `loginsM`: The mirror table.
13//! - `loginsSyncMeta`: The table used to to store various sync metadata.
14//!
15//! ## `loginsL`
16//!
17//! This stores local login information, also known as the "overlay".
18//!
19//! `loginsL` is essentially unchanged from firefox-ios, however note the
20//! semantic change v4 makes to timestamp fields (which is explained in more
21//! detail in the [COMMON_COLS] documentation).
22//!
23//! It is important to note that `loginsL` is not guaranteed to be present for
24//! all records. Synced records may only exist in `loginsM` (although this is
25//! not guaranteed). In either case, queries should read from both `loginsL` and
26//! `loginsM`.
27//!
28//! ### `loginsL` Columns
29//!
30//! Contains all fields in [COMMON_COLS], as well as the following additional
31//! columns:
32//!
33//! - `local_modified`: A millisecond local timestamp indicating when the record
34//!   was changed locally, or NULL if the record has never been changed locally.
35//!
36//! - `is_deleted`: A boolean indicating whether or not this record is a
37//!   tombstone.
38//!
39//! - `sync_status`: A `SyncStatus` enum value, one of
40//!
41//!     - `0` (`SyncStatus::Synced`): Indicating that the record has been synced
42//!
43//!     - `1` (`SyncStatus::Changed`): Indicating that the record should be
44//!       has changed locally and is known to exist on the server.
45//!
46//!     - `2` (`SyncStatus::New`): Indicating that the record has never been
47//!       synced, or we have been reset since the last time it synced.
48//!
49//! ## `loginsM`
50//!
51//! This stores server-side login information, also known as the "mirror".
52//!
53//! Like `loginsL`, `loginM` has not changed from firefox-ios, beyond the
54//! change to store timestamps as milliseconds explained in [COMMON_COLS].
55//!
56//! Also like `loginsL`, `loginsM` is not guaranteed to have rows for all
57//! records. It should not have rows for records which were not synced!
58//!
59//! It is important to note that `loginsL` is not guaranteed to be present for
60//! all records. Synced records may only exist in `loginsM`! Queries should
61//! test against both!
62//!
63//! ### `loginsM` Columns
64//!
65//! Contains all fields in [COMMON_COLS], as well as the following additional
66//! columns:
67//!
68//! - `server_modified`: the most recent server-modification timestamp
69//!   ([sync15::ServerTimestamp]) we've seen for this record. Stored as
70//!   a millisecond value.
71//!
72//! - `is_overridden`: A boolean indicating whether or not the mirror contents
73//!   are invalid, and that we should defer to the data stored in `loginsL`.
74//!
75//! ## `loginsSyncMeta`
76//!
77//! This is a simple key-value table based on the `moz_meta` table in places.
78//! This table was added (by this rust crate) in version 4, and so is not
79//! present in firefox-ios.
80//!
81//! Currently it is used to store two items:
82//!
83//! 1. The last sync timestamp is stored under [LAST_SYNC_META_KEY], a
84//!    `sync15::ServerTimestamp` stored in integer milliseconds.
85//!
86//! 2. The persisted sync state machine information is stored under
87//!    [GLOBAL_STATE_META_KEY]. This is a `sync15::GlobalState` stored as
88//!    JSON.
89//!

102/// Every column shared by both tables except for `id`
103///
104/// Note: `timeCreated`, `timeLastUsed`, and `timePasswordChanged` are in
105/// milliseconds. This is in line with how the server and Desktop handle it, but
106/// counter to how firefox-ios handles it (hence needing to fix them up
107/// firefox-ios on schema upgrade from 3, the last firefox-ios password schema
108/// version).
109///
110/// The reason for breaking from how firefox-ios does things is just because it
111/// complicates the code to have multiple kinds of timestamps, for very little
112/// benefit. It also makes it unclear what's stored on the server, leading to
113/// further confusion.
114///
115/// However, note that the `local_modified` (of `loginsL`) and `server_modified`
116/// (of `loginsM`) are stored as milliseconds as well both on firefox-ios and
117/// here (and so they do not need to be updated with the `timeLastUsed`/
118/// `timePasswordChanged`/`timeCreated` timestamps.
119pub const COMMON_COLS: &str = "
120    guid,
121    secFields,
122    origin,
123    httpRealm,
124    formActionOrigin,
125    usernameField,
126    passwordField,
127    timeCreated,
128    timeLastUsed,
129    timePasswordChanged,
130    timesUsed,
131    timeOfLastBreach,
132    timeLastBreachAlertDismissed
133";

200// breachesL stores encrypted passwords from logins that have been marked as breached.
201// This allows cross-domain password reuse detection: if a password was breached on site A,
202// we can warn users who reuse that same password on site B.
203//
204// https://bugzilla.mozilla.org/show_bug.cgi?id=2014119:
205// The `id` column provides implicit FIFO ordering for future retention policy implementation.
206// When retention is added, old entries will be deleted keeping only the most recent N entries
207// based on id ordering.
208//
209// Note: No index on encryptedPassword is possible since passwords are encrypted (non-deterministic).
210// All query operations must be done in-memory after decryption.
211const CREATE_LOCAL_BREACHES_TABLE_SQL: &str = "
212    CREATE TABLE IF NOT EXISTS breachesL (
213        id INTEGER PRIMARY KEY AUTOINCREMENT,
214        encryptedPassword TEXT
215    )
216";

224pub(crate) fn init(db: &Connection) -> Result<()> {
225    let user_version = db.conn_ext_query_one::<i64>("PRAGMA user_version")?;
226    warn!("user_version: {}", user_version);
227    if user_version == 0 {
228        return create(db);
229    }
230    if user_version != VERSION {
231        if user_version < VERSION {
232            upgrade(db, user_version)?;
233        } else {
234            warn!(
235                "Loaded future schema version {} (we only understand version {}). \
236                 Optimistically ",
237                user_version, VERSION
238            )
239        }
240    }
241    Ok(())
242}

244// Allow the redundant Ok() here.  It will make more sense once we have an actual upgrade function.
245#[allow(clippy::unnecessary_wraps)]
246fn upgrade(db: &Connection, from: i64) -> Result<()> {
247    debug!("Upgrading schema from {} to {}", from, VERSION);
248
249    if from == VERSION {
250        return Ok(());
251    }
252
253    for version in from..VERSION {
254        upgrade_from(db, version)?;
255    }
256
257    db.execute_batch(&SET_VERSION_SQL)?;
258    Ok(())
259}
260
261fn upgrade_from(db: &Connection, from: i64) -> Result<()> {
262    debug!("- running schema upgrade {}", from);
263    // Schema upgrades.
264    match from {
265        0 => Err(Error::IncompatibleVersion(from)),
266
267        // Just one new nullable column makes this fairly easy
268        1 => Ok(db.execute_batch("ALTER TABLE loginsM ADD enc_unknown_fields TEXT;")?),
269
270        // again, easy migratable nullable columns
271        2 => Ok(db.execute_batch(
272            "ALTER TABLE loginsL ADD timeOfLastBreach INTEGER;
273        ALTER TABLE loginsM ADD timeOfLastBreach INTEGER;
274        ALTER TABLE loginsL ADD timeLastBreachAlertDismissed INTEGER;
275        ALTER TABLE loginsM ADD timeLastBreachAlertDismissed INTEGER;",
276        )?),
277
278        3 => Ok(db.execute_batch(CREATE_LOCAL_BREACHES_TABLE_SQL)?),
279
280        // next migration, add here
281        _ => Err(Error::IncompatibleVersion(from)),
282    }
283}

285pub(crate) fn create(db: &Connection) -> Result<()> {
286    debug!("Creating schema");
287    db.execute_all(&[
288        &*CREATE_LOCAL_TABLE_SQL,
289        &*CREATE_MIRROR_TABLE_SQL,
290        CREATE_OVERRIDE_ORIGIN_INDEX_SQL,
291        CREATE_DELETED_ORIGIN_INDEX_SQL,
292        CREATE_META_TABLE_SQL,
293        CREATE_LOCAL_BREACHES_TABLE_SQL,
294        &*SET_VERSION_SQL,
295    ])?;
296    Ok(())
297}
