13impl AboutBlock {
14    fn nimbus_fully_qualified_name(&self) -> String {
15        let kt_about = self.kotlin_about.as_ref().unwrap();
16
17        let class = &kt_about.class;
18        if class.starts_with('.') {
19            format!("{}{}", kt_about.package, class)
20        } else {
21            class.clone()
22        }
23    }
24
25    fn nimbus_object_name_kt(&self) -> String {
26        let fqe = self.nimbus_fully_qualified_name();
27        let last = fqe.split('.').next_back().unwrap_or(&fqe);
28        last.to_string()
29    }
30
31    fn nimbus_package_name(&self) -> Option<String> {
32        let fqe = self.nimbus_fully_qualified_name();
33        if !fqe.contains('.') {
34            return None;
35        }
36        let mut it = fqe.split('.');
37        it.next_back()?;
38        Some(it.collect::<Vec<&str>>().join("."))
39    }
40
41    fn resource_package_name(&self) -> String {
42        let kt_about = self.kotlin_about.as_ref().unwrap();
43        kt_about.package.clone()
44    }
45}

47pub(crate) fn generate_struct(manifest: &FeatureManifest, cmd: &GenerateStructCmd) -> Result<()> {
48    if manifest.about.kotlin_about.is_none() {
49        return Err(FMLError::ValidationError(
50            "about".to_string(),
51            format!(
52                "The `about` block is missing a valid `android` or `kotlin` entry: {}",
53                &cmd.manifest
54            ),
55        ));
56    }
57
58    let path = &cmd.output;
59    let path = if path.is_dir() {
60        path.join(format!("{}.kt", manifest.about.nimbus_object_name_kt()))
61    } else {
62        path.clone()
63    };
64
65    let kt = gen_structs::FeatureManifestDeclaration::new(manifest);
66
67    let contents = kt.render()?;
68
69    std::fs::write(path, contents)?;
70
71    Ok(())
72}

144    // Compile a genertaed manifest file against the mocked out Android runtime.
145    pub fn compile_manifest_kt(manifest_paths: &[String]) -> Result<tempfile::TempDir> {
146        let temp = tempfile::tempdir()?;
147        let build_dir = temp.path();
148
149        let status = Command::new("kotlinc")
150            // Our generated bindings should not produce any warnings; fail tests if they do.
151            .arg("-Werror")
152            .arg("-J-ea")
153            // Reflect $CLASSPATH from the environment, to help find `json.jar`.
154            .arg("-classpath")
155            .arg(json_jar())
156            .arg("-d")
157            .arg(build_dir)
158            .arg(variables_kt())
159            .arg(features_kt())
160            .arg(hardcoded_features_kt())
161            .arg(runtime_dir())
162            .arg(nimbus_internals_kt())
163            .args(manifest_paths)
164            .spawn()?
165            .wait()?;
166        if status.success() {
167            Ok(temp)
168        } else {
169            bail!("running `kotlinc` failed compiling a generated manifest")
170        }
171    }

173    // Given a generated manifest, run a kts script against it.
174    pub fn run_script_with_generated_code(manifests_kt: &[String], script: &str) -> Result<()> {
175        if !detect_kotlinc()? {
176            println!("SDK-446 Install kotlinc or add it the PATH to run tests");
177            return Ok(());
178        }
179        let temp_dir = compile_manifest_kt(manifests_kt)?;
180        let build_dir = temp_dir.path();
181
182        let status = Command::new("kotlinc")
183            // Our generated bindings should not produce any warnings; fail tests if they do.
184            .arg("-Werror")
185            .arg("-J-ea")
186            // Reflect $CLASSPATH from the environment, to help find `json.jar`.
187            .arg("-classpath")
188            .arg(&classpath(build_dir)?)
189            .arg("-script")
190            .arg(script)
191            .spawn()?
192            .wait()?;
193
194        drop(temp_dir);
195        if status.success() {
196            Ok(())
197        } else {
198            bail!("running `kotlinc` failed running a script")
199        }
200    }
