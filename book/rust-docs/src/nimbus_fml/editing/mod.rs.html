22#[cfg(feature = "client-lib")]
23#[derive(Debug, PartialEq, Default)]
24pub struct FmlEditorError {
25    /// The message to display to the user.
26    pub message: String,
27    /// The token to highlight, and to replace
28    pub highlight: Option<String>,
29    /// The position in the source code of the first
30    /// character of the highlight
31    pub error_span: CursorSpan,
32    /// The list of possible corrective actions the user
33    /// can take to fix this error.
34    pub corrections: Vec<CorrectionCandidate>,
35    pub line: u32,
36    pub col: u32,
37}
38
39#[cfg(feature = "client-lib")]
40#[derive(Debug, Default, PartialEq)]
41pub struct CorrectionCandidate {
42    /// The string that should be inserted into the source
43    pub insert: String,
44    /// The short display name to represent the fix.
45    pub display_name: Option<String>,
46
47    /// The span where the `insert` string should overwrite. If None,
48    /// then use the `error_span`.
49    pub insertion_span: Option<CursorSpan>,
50
51    /// The final position of the cursor after the insertion has taken place.
52    /// If None, then should be left to the editor to decide.
53    pub cursor_at: Option<CursorPosition>,
54}
55
56/// Constructors
57#[cfg(feature = "client-lib")]
58impl CorrectionCandidate {
59    /// Replace the error token with a quoted string.
60    /// The display is the unquoted string.
61    pub(crate) fn string_replacement(s: &str) -> Self {
62        CorrectionCandidate {
63            insert: format!("\"{}\"", s),
64            display_name: Some(s.to_owned()),
65            ..Default::default()
66        }
67    }
68
69    /// Replace the error token with the literal,
70    /// represented by the `s: &str`.
71    pub(crate) fn literal_replacement(s: &str) -> Self {
72        CorrectionCandidate {
73            insert: s.to_owned(),
74            ..Default::default()
75        }
76    }
77}
