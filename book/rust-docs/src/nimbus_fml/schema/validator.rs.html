14const DISALLOWED_PREFS: &[(&str, &str)] = &[
15    (
16        r#"^app\.shield\.optoutstudies\.enabled$"#,
17        "disabling Nimbus causes immediate unenrollment",
18    ),
19    (
20        r#"^datareporting\.healthreport\.uploadEnabled$"#,
21        "disabling telemetry causes immediate unenrollment",
22    ),
23    (
24        r#"^services\.settings\.server$"#,
25        "changing the Remote Settings endpoint will break clients",
26    ),
27    (r#"^nimbus\.debug$"#, "internal Nimbus preference for QA"),
28    (
29        r#"^security\.turn_off_all_security_so_that_viruses_can_take_over_this_computer$"#,
30        "this pref is automation-only and is unsafe to enable outside tests",
31    ),
32];

34pub(crate) struct SchemaValidator<'a> {
35    enum_defs: &'a BTreeMap<String, EnumDef>,
36    object_defs: &'a BTreeMap<String, ObjectDef>,
37
38    lax_gecko_pref_validation: bool,
39}

41impl<'a> SchemaValidator<'a> {
42    pub(crate) fn new(
43        enums: &'a BTreeMap<String, EnumDef>,
44        objs: &'a BTreeMap<String, ObjectDef>,
45    ) -> Self {
46        Self {
47            enum_defs: enums,
48            object_defs: objs,
49            lax_gecko_pref_validation: false,
50        }
51    }
52
53    pub(crate) fn with_lax_gecko_pref_validation(mut self, value: bool) -> Self {
54        self.lax_gecko_pref_validation = value;
55        self
56    }
57
58    fn _get_enum(&self, nm: &str) -> Option<&EnumDef> {
59        self.enum_defs.get(nm)
60    }
61
62    fn get_object(&self, nm: &str) -> Option<&ObjectDef> {
63        self.object_defs.get(nm)
64    }

66    pub(crate) fn validate_object_def(&self, object_def: &ObjectDef) -> Result<()> {
67        let obj_nm = &object_def.name;
68        for prop in &object_def.props {
69            let prop_nm = &prop.name;
70
71            // Check the types exist for this property.
72            let path = format!("objects/{obj_nm}/{prop_nm}");
73            self.validate_type_ref(&path, &prop.typ)?;
74        }
75
76        Ok(())
77    }
78
79    pub(crate) fn validate_feature_def(&self, feature_def: &FeatureDef) -> Result<()> {
80        let feat_nm = &feature_def.name;
81        let mut string_aliases: HashSet<_> = Default::default();
82
83        for prop in &feature_def.props {
84            let prop_nm = &prop.name;
85            let prop_t = &prop.typ;
86
87            let path = format!("features/{feat_nm}/{prop_nm}");
88
89            // Check the types exist for this property.
90            self.validate_type_ref(&path, prop_t)?;
91
92            // Check pref is not in the disallowed prefs list.
93            if let Some(pref) = &prop.gecko_pref {
94                for (pref_str, error) in DISALLOWED_PREFS {
95                    let regex = Regex::new(pref_str)?;
96                    if regex.is_match(&pref.pref()) {
97                        return Err(FMLError::ValidationError(
98                            path,
99                            format!(
100                                "Cannot use pref `{}` in experiments, reason: {}",
101                                pref.pref(),
102                                error
103                            ),
104                        ));
105                    }
106                }
107            }
108
109            // Check pref support for this type.
110            if prop.gecko_pref.is_some()
111                && !prop
112                    .typ
113                    .supports_gecko_prefs(self.lax_gecko_pref_validation)
114            {
115                return Err(FMLError::ValidationError(
116                    path,
117                    "Pref keys can only be used with Option<Boolean>, Option<Int>, and Option<String>  variables"
118                        .to_string(),
119                ));
120            }
121
122            // Check string-alias definition.
123            if let Some(sa) = &prop.string_alias {
124                // Check that the string-alias has only been defined once in this feature.
125                if !string_aliases.insert(sa) {
126                    return Err(FMLError::ValidationError(
127                        path,
128                        format!("The string-alias {sa} should only be declared once per feature"),
129                    ));
130                }
131
132                // Check that the string-alias is actually used in this property type.
133                let types = prop_t.all_types();
134                if !types.contains(sa) {
135                    return Err(FMLError::ValidationError(
136                        path,
137                        format!(
138                            "The string-alias {sa} must be part of the {} type declaration",
139                            prop_nm
140                        ),
141                    ));
142                }
143            }
144        }
145
146        // Now check that that there is a path from this feature to any objects using the
147        // string-aliases defined in this feature.
148        let types = feature_def.all_types();
149        self.validate_string_alias_declarations(
150            &format!("features/{feat_nm}"),
151            feat_nm,
152            &types,
153            &string_aliases,
154        )?;
155
156        Ok(())
157    }
158
159    pub(crate) fn validate_prefs(&self, feature_manifest: &FeatureManifest) -> Result<()> {
160        let prefs = feature_manifest
161            .iter_gecko_prefs()
162            .map(|p| p.pref())
163            .collect::<Vec<String>>();
164        for pref in prefs.clone() {
165            if prefs
166                .iter()
167                .map(|p| if p == &pref { 1 } else { 0 })
168                .sum::<i32>()
169                > 1
170            {
171                let path = format!(r#"prefs/"{}""#, pref);
172                return Err(FMLError::ValidationError(
173                    path,
174                    "Prefs can only be include once per feature manifest".into(),
175                ));
176            }
177        }
178
179        Ok(())
180    }
181
182    fn validate_string_alias_declarations(
183        &self,
184        path: &str,
185        feature: &str,
186        types: &HashSet<TypeRef>,
187        string_aliases: &HashSet<&TypeRef>,
188    ) -> Result<()> {
189        let unaccounted: Vec<_> = types
190            .iter()
191            .filter(|t| matches!(t, TypeRef::StringAlias(_)))
192            .filter(|t| !string_aliases.contains(t))
193            .collect();
194
195        if !unaccounted.is_empty() {
196            let t = unaccounted.first().unwrap();
197            return Err(FMLError::ValidationError(
198                path.to_string(),
199                format!("A string-alias {t} is used by– but has not been defined in– the {feature} feature"),
200            ));
201        }
202        for t in types {
203            if let TypeRef::Object(nm) = t {
204                if let Some(obj) = self.get_object(nm) {
205                    let types = obj.all_types();
206                    self.validate_string_alias_declarations(
207                        &format!("objects/{nm}"),
208                        feature,
209                        &types,
210                        string_aliases,
211                    )?;
212                }
213            }
214        }
215        Ok(())
216    }
217
218    fn validate_type_ref(&self, path: &str, type_ref: &TypeRef) -> Result<()> {
219        match type_ref {
220            TypeRef::Enum(name) => {
221                if !self.enum_defs.contains_key(name) {
222                    return Err(FMLError::ValidationError(
223                        path.to_string(),
224                        format!("Found enum reference with name: {name}, but no definition"),
225                    ));
226                }
227            }
228            TypeRef::Object(name) => {
229                if !self.object_defs.contains_key(name) {
230                    return Err(FMLError::ValidationError(
231                        path.to_string(),
232                        format!("Found object reference with name: {name}, but no definition"),
233                    ));
234                }
235            }
236            TypeRef::EnumMap(key_type, value_type) => match key_type.as_ref() {
237                TypeRef::Enum(_) | TypeRef::String | TypeRef::StringAlias(_) => {
238                    self.validate_type_ref(path, key_type)?;
239                    self.validate_type_ref(path, value_type)?;
240                }
241                _ => {
242                    return Err(FMLError::ValidationError(
243                        path.to_string(),
244                        format!(
245                            "Map key must be a String, string-alias or enum, found: {key_type:?}",
246                        ),
247                    ))
248                }
249            },
250            TypeRef::List(list_type) => self.validate_type_ref(path, list_type)?,
251            TypeRef::StringMap(value_type) => self.validate_type_ref(path, value_type)?,
252            TypeRef::Option(option_type) => {
253                if let TypeRef::Option(_) = option_type.as_ref() {
254                    return Err(FMLError::ValidationError(
255                        path.to_string(),
256                        "Found nested optional types".into(),
257                    ));
258                } else {
259                    self.validate_type_ref(path, option_type)?
260                }
261            }
262            _ => (),
263        };
264        Ok(())
265    }
266}

475mod string_aliases {
476    use std::slice;
477
478    use serde_json::json;

480    use crate::intermediate_representation::PropDef;

482    use super::*;

484    fn with_objects(objects: &[ObjectDef]) -> BTreeMap<String, ObjectDef> {
485        let mut obj_defs: BTreeMap<_, _> = Default::default();
486        for o in objects {
487            obj_defs.insert(o.name(), o.clone());
488        }
489        obj_defs
490    }
491
492    fn with_feature(props: &[PropDef]) -> FeatureDef {
493        FeatureDef::new("test-feature", "", props.into(), false)
494    }
495
496    #[test]
497    fn test_validate_feature_schema() -> Result<()> {
498        let name = TypeRef::StringAlias("PersonName".to_string());
499        let all_names = {
500            let t = TypeRef::List(Box::new(name.clone()));
501            let v = json!(["Alice", "Bonnie", "Charlie", "Denise", "Elise", "Frankie"]);
502            PropDef::with_string_alias("all-names", &t, &v, &name)
503        };
504
505        let all_names2 = {
506            let t = TypeRef::List(Box::new(name.clone()));
507            let v = json!(["Alice", "Bonnie"]);
508            PropDef::with_string_alias("all-names-duplicate", &t, &v, &name)
509        };
510
511        let enums = Default::default();
512        let objects = Default::default();
513        let validator = SchemaValidator::new(&enums, &objects);
514
515        // -> Verify that only one property per feature can define the same string-alias.
516        let fm = with_feature(&[all_names.clone(), all_names2.clone()]);
517        assert!(validator.validate_feature_def(&fm).is_err());
518
519        let newest_member = {
520            let t = &name;
521            let v = json!("Alice"); // it doesn't matter for this test what the value is.
522            PropDef::new("newest-member", t, &v)
523        };
524
525        // -> Verify that a property in a feature can validate against the a string-alias
526        // -> in the same feature.
527        // { all-names: ["Alice"], newest-member: "Alice" }
528        let fm = with_feature(&[all_names.clone(), newest_member.clone()]);
529        validator.validate_feature_def(&fm)?;
530
531        // { newest-member: "Alice" }
532        // We have a reference to a team mate, but no definitions.
533        // Should error out.
534        let fm = with_feature(slice::from_ref(&newest_member));
535        assert!(validator.validate_feature_def(&fm).is_err());
536
537        // -> Validate a property in a nested object can validate against a string-alias
538        // -> in a feature that uses the object.
539        let team_def = ObjectDef::new("Team", slice::from_ref(&newest_member));
540        let team = {
541            let t = TypeRef::Object("Team".to_string());
542            let v = json!({ "newest-member": "Alice" });
543
544            PropDef::new("team", &t, &v)
545        };
546
547        // { all-names: ["Alice"], team: { newest-member: "Alice" } }
548        let fm = with_feature(&[all_names.clone(), team.clone()]);
549        let objs = with_objects(slice::from_ref(&team_def));
550        let validator = SchemaValidator::new(&enums, &objs);
551        validator.validate_feature_def(&fm)?;
552
553        // { team: { newest-member: "Alice" } }
554        let fm = with_feature(slice::from_ref(&team));
555        let objs = with_objects(slice::from_ref(&team_def));
556        let validator = SchemaValidator::new(&enums, &objs);
557        assert!(validator.validate_feature_def(&fm).is_err());
558
559        // -> Validate a property in a deeply nested object can validate against a string-alias
560        // -> in a feature that uses the object.
561
562        let match_def = ObjectDef::new("Match", slice::from_ref(&team));
563        let match_ = {
564            let t = TypeRef::Object("Match".to_string());
565            let v = json!({ "team": { "newest-member": "Alice" }});
566
567            PropDef::new("match", &t, &v)
568        };
569
570        // { all-names: ["Alice"], match: { team: { newest-member: "Alice" }} }
571        let fm = with_feature(&[all_names.clone(), match_.clone()]);
572        let objs = with_objects(&[team_def.clone(), match_def.clone()]);
573        let validator = SchemaValidator::new(&enums, &objs);
574        validator.validate_feature_def(&fm)?;
575
576        // { match: {team: { newest-member: "Alice" }} }
577        let fm = with_feature(slice::from_ref(&match_));
578        let validator = SchemaValidator::new(&enums, &objs);
579        assert!(validator.validate_feature_def(&fm).is_err());
580
581        Ok(())
582    }
